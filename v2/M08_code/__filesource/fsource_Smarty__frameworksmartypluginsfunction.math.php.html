18            return;

20

22    
23        // make sure parenthesis are balanced
24        if (substr_count($equation,"(") != substr_count($equation,")")) {
25            $smarty->trigger_error("math: unbalanced parenthesis");
26            return;
27        }
28    
29        // match all vars in equation, make sure all are passed
30        preg_match_all("![a-zA-Z][a-zA-Z0-9_]*!",$equation, $match);
31        $allowed_funcs = array('int','abs','ceil','cos','exp','floor','log','log10',
32                               'max','min','pi','pow','rand','round','sin','sqrt','srand','tan');
33    
34        foreach($match[0] as $curr_var) {
35            if (!in_array($curr_var,array_keys($params)) && !in_array($curr_var, $allowed_funcs)) {
36                $smarty->trigger_error("math: parameter $curr_var not passed as argument");
37                return;
38            }
39        }
40    
41        foreach($params as $key => $val) {
42            if ($key != "equation" && $key != "format" && $key != "assign") {
43                // make sure value is not empty
44                if (strlen($val)==0) {
45                    $smarty->trigger_error("math: parameter $key is empty");
46                    return;
47                }
48                if (!is_numeric($val)) {
49                    $smarty->trigger_error("math: parameter $key: is not numeric");
50                    return;
51                }
52                $equation = preg_replace("/\b$key\b/",$val, $equation);
53            }
54        }
55    
56        eval("\$smarty_math_result = ".$equation.";");
57    
58        if (empty($params['format'])) {
59            if (empty($params['assign'])) {
60                return $smarty_math_result;
61            } else {
62                $smarty->assign($params['assign'],$smarty_math_result);
63            }
64        } else {
65            if (empty($params['assign'])){
66                printf($params['format'],$smarty_math_result);
67            } else {
68                $smarty->assign($params['assign'],sprintf($params['format'],$smarty_math_result));
69            }
70        }
71    }
