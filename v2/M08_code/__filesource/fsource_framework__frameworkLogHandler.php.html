4      * Writes logs file to preserve important information about
5      * framework, can be extended for saving and application actions.
6      *
7      * It's created with an array configuration and an indentifier
8      * that defines the log format (what must be saved in logs).
9      * Can handle multiple log resources of different types.
10     * Saves log information on filesystems in CVS format in
11     * different files for each log type.
12     * Saves log requests with the addLogEntry() method

38        /**
39    	 * Creates a LogHandler object.
40    	 *
41    	 * @param string $logging_path filesystem path where logs are saved
42    	 *              (system should have read-write rights on this directory)
43    	 * @param string $type_indentifier log type identifier for this handler object
44    	 * @param array $type_definition ordered array of column names/types to add in CSV file
45    	 */
46        function LogHandler($type_indentifier, $logging_path, $type_definition)
47        {
48            $this->identifier = $type_indentifier;
49            $this->path = $logging_path;
50            $this->file_name = date('Y').'-'.date('m').'-'.date('d').'-'.
51                                $type_indentifier.'.log.csv';
52            $this->count_values = count($type_definition);
53            $this->definition = $type_definition;
54        }

58        /**
59    	 * Adds an entry in log file.
60    	 * 
61    	 * If entry parameter is undefined default value is an empty string
62    	 *
63    	 * @param array $entry associative array of column<->values to add
64    	 * @return boolean true: successfull, false: error
65    	 */
66        function addLogEntry($entry)
67        {
68            $string='';
69            for ($i=0; $i < $this->count_values; $i++)
70            {
71                if (!array_key_exists($this->definition[$i],$entry) )
72                {
73                    $curr_value = '';
74                }
75                else
76                {
77                    $curr_value = $entry[$this->definition[$i]];
78                }
79                
80                $curr_value = str_replace($this->csv_text_delimiter, $this->csv_text_delimiter.$this->csv_text_delimiter, $curr_value);
81                $curr_value = str_replace("\n",'|', $curr_value);
82                $string .= $this->csv_text_delimiter.$curr_value.$this->csv_text_delimiter;
83                if ($i < $this->count_values -1) 
84                {
85                    $string .= $this->csv_separator;
86                }
87            }
88    
89            $string .= "\n";
90            
91            $this->_addCsvLine($this->path.$this->file_name, $string);
92            
93        }

129       /**
130   	 * Adds a new entry in the csv log file
131   	 * It takes care about file "lock" between multiple requests.
132   	 *
133   	 * @param string $file $logging_path filesystem path where logs are saved
134   	 *              (system should have read-write rights on this directory)
135   	 * @param string $string line in csv format "column1";"column2";..;"columnN"\n
136   	 * @access private
137   	 */
138       function _addCsvLine($full_file_name, $string)
139       {
140           $addHeader = false;
141           if ( !file_exists($full_file_name) == true ){
142               $addHeader = true;    
143           }
144           
145           $fp = fopen ($full_file_name, "a");
146           
147           if ($fp === false) return false; 
148           
149           flock ($fp,2);
150           if ( $addHeader == true )
151           {
152               $header = $this->_getHeaderLine();
153               fwrite($fp, $header);
154           } 
155           fwrite($fp, $string);
156           fflush ($fp);
157           flock($fp,3);
158           fclose($fp);
159       }
