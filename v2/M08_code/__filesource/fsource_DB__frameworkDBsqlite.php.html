180       /**
181        * Connect to a database represented by a file.
182        *
183        * @param $dsn the data source name; the file is taken as
184        *         database; "sqlite://root:@host/test.db"
185        * @param $persistent (optional) whether the connection should
186        *         be persistent
187        * @access public
188        * @return int DB_OK on success, a DB error on failure
189        */
190       function connect($dsninfo, $persistent = false) {
191           $ret = DB_OK;
192           $file = $dsninfo['database'];
193   
194           if (!DB::assertExtension('sqlite'))
195               return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
196   
197           if (isset($file)) {
198               if (!file_exists($file)) {
199                   touch($file );
200                   chmod($file, (is_numeric($dsninfo['mode']) ? $dsninfo['mode'] : 0644 ));
201                   if (!file_exists($file)) {
202                       return $this->sqliteRaiseError(DB_ERROR_NOT_FOUND );
203                   }
204               }
205               if (!is_file($file)) {
206                   return $this->sqliteRaiseError(DB_ERROR_INVALID );
207               }
208               if (!is_readable($file)) {
209                   return $this->sqliteRaiseError(DB_ERROR_ACCESS_VIOLATION );
210               }
211           } else {
212               return $this->sqliteRaiseError(DB_ERROR_ACCESS_VIOLATION );
213           }
214   
215           $connect_function = $persistent ? 'sqlite_open' : 'sqlite_popen';
216           if (!($conn = @$connect_function($dsninfo['database']) )) {
217               return $this->sqliteRaiseError(DB_ERROR_NODBSELECTED);
218           }
219           $this->connection = $conn;
220           $this->dsn = $dsninfo;
221   
222           return DB_OK;
223       }

243       /**
244        * Send a query to SQLite and returns the results as a SQLite resource
245        * identifier.
246        *
247        * @param the SQL query
248        * @access public
249        * @return mixed returns a valid SQLite result for successful SELECT
250        *  queries, DB_OK for other successful queries. A DB error is
251        *  returned on failure.
252        */
253       function simpleQuery($query) {
254           $ismanip = DB::isManip($query);
255           $this->last_query = $query;
256           $query = $this->_modifyQuery($query);
257   
258           $result = @sqlite_query($query, $this->connection);
259           $this->result = $result;
260           if (!$this->result ) {
261               $errno = sqlite_last_error($this->connection );
262               if (!$errno) {
263                   return null;
264               }
265               return $this->sqliteRaiseError($errno);
266           }
267   
268           if (gettype($result) === "resource") {
269               $numRows = $this->numRows($result);
270               if (is_object($numRows )) {
271                   return $numRows;
272               }
273               return $result;
274           }
275           return DB_OK;
276       }

295       /**
296        * Fetch and return a row of data (it uses fetchInto for that)
297        *
298        * @param   $result     SQLite result identifier
299        * @param   $fetchmode  format of fetched row array
300        * @param   $rownum     the absolute row number to fetch
301        * @return  array       a row of data, or false on error
302        */
303       function fetchRow($result, $fetchmode=DB_FETCHMODE_DEFAULT, $rownum=null) {
304           if ($fetchmode == DB_FETCHMODE_DEFAULT) {
305               $fetchmode = $this->fetchmode;
306           }
307           $res = $this->fetchInto($this->result, $arr, $fetchmode, $rownum );
308           if (!$res) {
309               $errno = sqlite_last_error($this->connection );
310               if (!$errno) {
311                   return null;
312               }
313               return $this->raiseError($errno);
314           }
315           if ($res !== DB_OK) {
316               return $res;
317           }
318           return $arr;
319       }

324       /**
325        * Fetch a row and insert the data into an existing array. Availabe modes
326        * are SQLITE_ASSOC, SQLITE_NUM and SQLITE_BOTH. An object type is not
327        * available.
328        *
329        * @param $result    SQLite result identifier
330        * @param $arr       (reference) array where data from the row is stored
331        * @param $fetchmode how the array data should be indexed
332        * @param $rownum    the row number to fetch
333        * @access public
334        *
335        * @return int DB_OK on success, a DB error on failure
336        */
337       function fetchInto($result, &$arr, $fetchmode, $rownum=null) {
338           if ($rownum !== null) {
339               if (!@sqlite_seek($this->result, $rownum)) {
340                   return null;
341               }
342           }
343           if ($fetchmode & DB_FETCHMODE_ASSOC ) {
344               $arr = sqlite_fetch_array($this->result, SQLITE_ASSOC );
345           } else {
346               $arr = sqlite_fetch_array($this->result, SQLITE_NUM );
347           }
348           if (!$arr) {
349               $errno = sqlite_last_error($this->connection );
350               if (!$errno) {
351                   return null;
352               }
353               return $this->sqliteRaiseError($errno);
354           }
355           return DB_OK;
356       }

362        * Free the internal resources associated with $result.
363        *
364        * @param $result SQLite result identifier or DB statement identifier
365        * @access public
366        * @return bool TRUE on success, FALSE if $result is invalid
367        */
368       function freeResult($result) {
369           if(is_resource($result)) {
370               unset($result);
371               return true;
372           }
373           // $result is a prepared query handle
374           $result = (int)$result;
375           if (!isset($this->prepare_tokens[$result])) {
376               return false;
377           }
378           $this->prepare_types = array();
379           $this->prepare_tokens = array();
380           return true;
381       }

442       /**
443       * Returns the query needed to get some backend info. Refer to
444       * the online manual at http://sqlite.org/sqlite.html.
445       *
446       * @param string $type What kind of info you want to retrieve
447       * @return string The SQL query string
448       */
449       function getSpecialQuery($type, $args=array()) {
450           $query = "";
451           if(!is_array($args))
452               return $this->raiseError('no key specified', null, null, null,
453                                        'Argument has to be an array.');
454           switch (strtolower($type)) {
455           case 'master':
456               $query .= "SELECT * FROM sqlite_master;";
457               break;
458           case 'tables':
459               $query .= "SELECT name FROM sqlite_master WHERE type='table' ";
460               $query .= "UNION ALL SELECT name FROM sqlite_temp_master ";
461               $query .= "WHERE type='table' ORDER BY name;";
462               break;
463           case 'schema':
464               $query .= "SELECT sql FROM (SELECT * FROM sqlite_master UNION ALL ";
465               $query .= "SELECT * FROM sqlite_temp_master) ";
466               $query .= "WHERE type!='meta' ORDER BY tbl_name, type DESC, name;";
467               break;
468           case 'schemax':
469           case 'schema_x':
470               /**
471               * Use like:
472               * $res = $db->query( $db->getSpecialQuery("schema_x", array("table" => "table3" )) );
473               */
474               $query .= "SELECT sql FROM (SELECT * FROM sqlite_master UNION ALL ";
475               $query .= "SELECT * FROM sqlite_temp_master) ";
476               $query .= sprintf("WHERE tbl_name LIKE '%s' AND type!='meta' ", $args['table'] );
477               $query .= "ORDER BY type DESC, name;";
478               break;
479           case 'alter':
480               /**
481               * SQLite does not support ALTER TABLE; this is a helper query to handle this. 'table'
482               * represents the table name, 'rows' the news rows to create, 'save' the row(s) to keep
483               * _with_ the data.
484               *
485               * Use like:
486               * $args = array(
487               *     'table' => $table, 
488               *     'rows'  => "id INTEGER PRIMARY KEY, firstname TEXT, surname TEXT, datetime TEXT",
489               *     'save'  => "NULL, titel, content, datetime"
490               * );
491               * );
492               * $res = $db->query( $db->getSpecialQuery("alter", $args ) );
493               */
494               $rows = strtr($args['rows'], $this->keywords );
495   
496               $query .= "BEGIN TRANSACTION;";
497               $query .= "CREATE TEMPORARY TABLE {$args['table']}_backup ({$args['rows']});";
498               $query .= "INSERT INTO {$args['table']}_backup SELECT {$args['save']} FROM {$args['table']};";
499               $query .= "DROP TABLE {$args['table']};";
500               $query .= "CREATE TABLE {$args['table']} ({$args['rows']});";
501               $query .= "INSERT INTO {$args['table']} SELECT {$rows} FROM {$args['table']}_backup;";
502               $query .= "DROP TABLE {$args['table']}_backup;";
503               $query .= "COMMIT;";
504   
505               // This is a dirty hack, since the above query will no get executed with a single
506               // query call; so here the query method will be called directly and return a select instead.
507               $q = explode(";", $query );
508               for($i=0; $i<8; $i++)
509                   $result = $this->query( $q[$i] );
510               $query = "SELECT * FROM {$args['table']};";
511               break;
512           default:
513               return null;
514           }
515           return $query;
516       }

521       /**
522       * Get the file stats for the current database. Possible arguments are
523       * dev, ino, mode, nlink, uid, gid, rdev, size, atime, mtime, ctime, blksize, blocks
524       * or a numeric key between 0 and 12.
525       *
526       * @param string $arg Array key for stats()
527       * @return mixed array on an unspecified key, integer on a passed arg and
528       *  FALSE at a stats error.
529       */
530       function getDbFileStats($arg="" ) {
531           $stats = stat($this->dsn['database'] );
532           if ($stats == false )
533               return false;
534           if (is_array($stats)) {
535               if(is_numeric($arg) ) {
536                   if(((int)$arg <= 12) & ((int)$arg >= 0))
537                       return false;
538                   return $stats[$arg ];
539               }
540               if (array_key_exists(trim($arg), $stats)) {
541                   return $stats[$arg ];
542               }
543           }
544           return $stats;
545       }

550       /**
551       * "DELETE FROM table" gives 0 affected rows in SQLite. This little hack
552       * lets you know how many rows were deleted.
553       *
554       * @param string $query The SQL query string
555       * @return string The SQL query string
556       */
557       function _modifyQuery($query ) {
558           if ($this->options['optimize'] == 'portability') {
559               if (preg_match('/^\s*DELETE\s+FROM\s+(\S+)\s*$/i', $query)) {
560                   $query = preg_replace('/^\s*DELETE\s+FROM\s+(\S+)\s*$/',
561                                         'DELETE FROM \1 WHERE 1=1', $query);
562               }
563           }
564           return $query;
565       }

576       function sqliteRaiseError($errno = null) {
