12     *           Gerard <gerard@interfold.com>
13     *           Jason Sweat <jsweat_php@yahoo.com>
14     * Purpose:  cycle through given values
15     * Input:    name = name of cycle (optional)
16     *           values = comma separated list of values to cycle,
17     *                    or an array of values to cycle
18     *                    (this can be left out for subsequent calls)
19     *
20     *           reset = boolean - resets given var to true
21     *			 print = boolean - print var or not. default is true
22     *           advance = boolean - whether or not to advance the cycle
23     *           delimiter = the value delimiter, default is ","
24     *           assign = boolean, assigns to template var instead of
25     *                    printed.
26     * 
27     * Examples: {cycle values="#eeeeee,#d0d0d0d"}
28     *           {cycle name=row values="one,two,three" reset=true}
29     *           {cycle name=row}
30     * -------------------------------------------------------------
31     *
32     * @package Smarty
33     */
34    function smarty_function_cycle($params, &$smarty)
35    {
36        static $cycle_vars;
37        
38        extract($params);
39    
40        if (empty($name)) {
41            $name = 'default';
42        }
43    
44        if (!isset($print)) {
45            $print = true;
46        }
47    
48        if (!isset($advance)) {
49            $advance = true;        
50        }    
51    
52        if (!isset($reset)) {
53            $reset = false;        
54        }        
55                
56        if (!in_array('values', array_keys($params))) {
57            if(!isset($cycle_vars[$name]['values'])) {
58                $smarty->trigger_error("cycle: missing 'values' parameter");
59                return;
60            }
61        } else {
62            if(isset($cycle_vars[$name]['values'])
63                && $cycle_vars[$name]['values'] != $values ) {
64                $cycle_vars[$name]['index'] = 0;
65            }
66            $cycle_vars[$name]['values'] = $values;
67        }
68    
69        if (isset($delimiter)) {
70            $cycle_vars[$name]['delimiter'] = $delimiter;
71        } elseif (!isset($cycle_vars[$name]['delimiter'])) {
72            $cycle_vars[$name]['delimiter'] = ',';        
73        }
74        
75        if(!is_array($cycle_vars[$name]['values'])) {
76            $cycle_array = explode($cycle_vars[$name]['delimiter'],$cycle_vars[$name]['values']);
77        } else {
78            $cycle_array = $cycle_vars[$name]['values'];    
79        }
80        
81        if(!isset($cycle_vars[$name]['index']) || $reset ) {
82            $cycle_vars[$name]['index'] = 0;
83        }
84        
85        if (isset($assign)) {
86            $print = false;
87            $smarty->assign($assign, $cycle_array[$cycle_vars[$name]['index']]);
88        }
89            
90        if($print) {
91            $retval = $cycle_array[$cycle_vars[$name]['index']];
92        } else {
93            $retval = null;
94        }
95    
96        if($advance) {
97            if ( $cycle_vars[$name]['index'] >= count($cycle_array) -1 ) {
98                $cycle_vars[$name]['index'] = 0;            
99            } else {
100               $cycle_vars[$name]['index']++;
101           }
102       }
103       
104       return $retval;
105   }
