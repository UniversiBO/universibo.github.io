102       /**
103        * Utility method to build a "WHERE" clause to locate ourselves in
104        * the table.
105        *
106        * XXX future improvement: use rowids?
107        *
108        * @access private
109        */
110       function _makeWhere($keyval = null)
111       {
112           if (is_array($this->_keycolumn)) {
113               if ($keyval === null) {
114                   for ($i = 0; $i < sizeof($this->_keycolumn); $i++) {
115                       $keyval[] = $this->{$this->_keycolumn[$i]};
116                   }
117               }
118               $whereclause = '';
119               for ($i = 0; $i < sizeof($this->_keycolumn); $i++) {
120                   if ($i > 0) {
121                       $whereclause .= ' AND ';
122                   }
123                   $whereclause .= $this->_keycolumn[$i];
124                   if (is_null($keyval[$i])) {
125                       // there's not much point in having a NULL key,
126                       // but we support it anyway
127                       $whereclause .= ' IS NULL';
128                   } else {
129                       $whereclause .= ' = ' . $this->_dbh->quote($keyval[$i]);
130                   }
131               }
132           } else {
133               if ($keyval === null) {
134                   $keyval = @$this->{$this->_keycolumn};
135               }
136               $whereclause = $this->_keycolumn;
137               if (is_null($keyval)) {
138                   // there's not much point in having a NULL key,
139                   // but we support it anyway
140                   $whereclause .= ' IS NULL';
141               } else {
142                   $whereclause .= ' = ' . $this->_dbh->quote($keyval);
143               }
144           }
145           return $whereclause;
146       }

151       /**
152        * Method used to initialize a DB_storage object from the
153        * configured table.
154        *
155        * @param $keyval mixed the key[s] of the row to fetch (string or array)
156        *
157        * @return int DB_OK on success, a DB error if not
158        */
159       function setup($keyval)
160       {
161           $qval = $this->_dbh->quote($keyval);
162           $whereclause = $this->_makeWhere($keyval);
163           $query = 'SELECT * FROM ' . $this->_table . ' WHERE ' . $whereclause;
164           $sth = $this->_dbh->query($query);
165           if (DB::isError($sth)) {
166               return $sth;
167           }
168           $row = $sth->fetchRow(DB_FETCHMODE_ASSOC);
169           if (DB::isError($row)) {
170               return $row;
171           }
172           if (empty($row)) {
173               return $this->raiseError(null, DB_ERROR_NOT_FOUND, null, null,
174                                        $query, null, true);
175           }
176           foreach ($row as $key => $value) {
177               $this->_properties[$key] = true;
178               $this->$key = $value;
179           }
180           return DB_OK;
181       }

186       /**
187        * Create a new (empty) row in the configured table for this
188        * object.
189        */
190       function insert($newpk)
191       {
192           if (is_array($this->_keycolumn)) {
193               $primarykey = $this->_keycolumn;
194           } else {
195               $primarykey = array($this->_keycolumn);
196           }
197           settype($newpk, "array");
198           for ($i = 0; $i < sizeof($primarykey); $i++) {
199               $pkvals[] = $this->_dbh->quote($newpk[$i]);
200           }
201   
202           $sth = $this->_dbh->query("INSERT INTO $this->_table (" .
203                                     implode(",", $primarykey) . ") VALUES(" .
204                                     implode(",", $pkvals) . ")");
205           if (DB::isError($sth)) {
206               return $sth;
207           }
208           if (sizeof($newpk) == 1) {
209               $newpk = $newpk[0];
210           }
211           $this->setup($newpk);
212       }

217       /**
218        * Output a simple description of this DB_storage object.
219        * @return string object description
220        */
221       function toString()
222       {
223           $info = get_class($this);
224           $info .= " (table=";
225           $info .= $this->_table;
226           $info .= ", keycolumn=";
227           if (is_array($this->_keycolumn)) {
228               $info .= "(" . implode(",", $this->_keycolumn) . ")";
229           } else {
230               $info .= $this->_keycolumn;
231           }
232           $info .= ", dbh=";
233           if (is_object($this->_dbh)) {
234               $info .= $this->_dbh->toString();
235           } else {
236               $info .= "null";
237           }
238           $info .= ")";
239           if (sizeof($this->_properties)) {
240               $info .= " [loaded, key=";
241               $keyname = $this->_keycolumn;
242               if (is_array($keyname)) {
243                   $info .= "(";
244                   for ($i = 0; $i < sizeof($keyname); $i++) {
245                       if ($i > 0) {
246                           $info .= ",";
247                       }
248                       $info .= $this->$keyname[$i];
249                   }
250                   $info .= ")";
251               } else {
252                   $info .= $this->$keyname;
253               }
254               $info .= "]";
255           }
256           if (sizeof($this->_changes)) {
257               $info .= " [modified]";
258           }
259           return $info;
260       }

354       /**
355        * Modify an attriute value.
356        */
357       function set($property, $newvalue)
358       {
359           // only change if $property is known and object is not
360           // read-only
361           if ($this->_readonly) {
362               return $this->raiseError(null, DB_WARNING_READ_ONLY, null,
363                                        null, null, null, true);
364           }
365           if (@isset($this->_properties[$property])) {
366               if (empty($this->_validator)) {
367                   $valid = true;
368               } else {
369                   $valid = @call_user_func($this->_validator,
370                                            $this->_table,
371                                            $property,
372                                            $newvalue,
373                                            $this->$property,
374                                            $this);
375               }
376               if ($valid) {
377                   $this->$property = $newvalue;
378                   @$this->_changes[$property]++;
379               } else {
380                   return $this->raiseError(null, DB_ERROR_INVALID, null,
381                                            null, "invalid field: $property",
382                                            null, true);
383               }
384               return true;
385           }
386           return $this->raiseError(null, DB_ERROR_NOSUCHFIELD, null,
387                                    null, "unknown field: $property",
388                                    null, true);
389       }

431       /**
432        * Stores changes to this object in the database.
433        *
434        * @return DB_OK or a DB error
435        */
436       function store()
437       {
438           while (list($name, $changed) = each($this->_changes)) {
439               $params[] = &$this->$name;
440               $vars[] = $name . ' = ?';
441           }
442           if ($vars) {
443               $query = 'UPDATE ' . $this->_table . ' SET ' .
444                   implode(', ', $vars) . ' WHERE ' .
445                   $this->_makeWhere();
446               $stmt = $this->_dbh->prepare($query);
447               $res = $this->_dbh->execute($stmt, $params);
448               if (DB::isError($res)) {
449                   return $res;
450               }
451               $this->_changes = array();
452           }
453           return DB_OK;
454       }

459       /**
460        * Remove the row represented by this object from the database.
461        *
462        * @return mixed DB_OK or a DB error
463        */
464       function remove()
465       {
466           if ($this->_readonly) {
467               return $this->raiseError(null, DB_WARNING_READ_ONLY, null,
468                                        null, null, null, true);
469           }
470           $query = 'DELETE FROM ' . $this->_table .' WHERE '.
471               $this->_makeWhere();
472           $res = $this->_dbh->query($query);
473           if (DB::isError($res)) {
474               return $res;
475           }
476           foreach ($this->_properties as $prop => $foo) {
477               unset($this->$prop);
478           }
479           $this->_properties = array();
480           $this->_changes = array();
481           return DB_OK;
482       }
