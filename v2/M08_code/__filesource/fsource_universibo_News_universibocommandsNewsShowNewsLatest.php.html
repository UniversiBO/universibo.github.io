157       {

159            $db =& FrontController::getDbConnection('main');

161           $query = 'SELECT A.id_news FROM news A, news_canale B 
162   					WHERE A.id_news = B.id_news AND eliminata!='.$db->quote( NEWS_ELIMINATA ).
163                       'AND ( data_scadenza IS NULL OR \''.time().'\' < data_scadenza ) AND B.id_canale = '.$db->quote($id_canale).' 
164   					ORDER BY A.data_inserimento DESC';
165           $res =& $db->limitQuery($query, 0 , $num);
