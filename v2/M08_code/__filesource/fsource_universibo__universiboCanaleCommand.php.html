28    	 * Restituisce l'id_canale corrente, se non è specificato nella richiesta HTTP-GET si considera
29    	 * default l'homepage id_canale = 1
30    	 *
31    	 * @static
32    	 * @return int 
33    	 */
34        function getRequestIdCanale()
35        {
36            if (!array_key_exists('id_canale', $_GET ) )
37            {
38                if ($this->frontController->getCommandRequest() == 'ShowHome') return 1;
39                else Error::throw(_ERROR_DEFAULT,array('msg'=>'il parametro id_canale non è specificato nella richiesta','file'=>__FILE__,'line'=>__LINE__));
40            }
41    
42            if (!ereg('^([0-9]+)$', $_GET['id_canale'] ) )
43                Error::throw(_ERROR_DEFAULT,array('msg'=>'il parametro id_canale è sintatticamente non valido','file'=>__FILE__,'line'=>__LINE__));
44    
45            return intval($_GET['id_canale']);
46        }

82        {
83    
84            $tipo_canale =  Canale::getTipoCanaleFromId ( $this->getRequestIdCanale() );
85            if ($tipo_canale === false )
86                Error::throw(_ERROR_DEFAULT,array('msg'=>'Il canale richiesto non è presente','file'=>__FILE__,'line'=>__LINE__));
87            
88            $dispatch_array = array (    CANALE_DEFAULT   => 'Canale',
89                                        CANALE_HOME      => 'Canale',
90                                        CANALE_FACOLTA   => 'Facolta',
91                                        CANALE_CDL       => 'Cdl',
92                                        CANALE_ESAME_ING => 'EsameIng',
93                                        CANALE_ESAME_ECO => 'EsameEco');
94    
95            $class_name = $dispatch_array[$tipo_canale];
96            require_once($class_name.PHP_EXTENSION);
97            
98            $this->requestCanale =& call_user_func(array($class_name,'factoryCanale'), $this->getRequestIdCanale());
99    
100           //$this->requestCanale =& $class_name::factoryCanale( $this->getRequestIdCanale() );
101                 
102           if ( $this->requestCanale === false ) 
103               Error::throw(_ERROR_DEFAULT,array('msg'=>'Il canale richiesto non è presente','file'=>__FILE__,'line'=>__LINE__));
104           
105           $canale = $this->getRequestCanale();
106           $canale->addVisite();
107
