27    function smarty_function_html_checkboxes($params, &$smarty)
28    {
29       require_once $smarty->_get_plugin_filepath('shared','escape_special_chars');
30    
31       $name = 'checkbox';
32       $values = null;
33       $options = null;
34       $selected = null;
35       $separator = '';
36       $output = null;
37    
38       $extra = '';
39    
40       foreach($params as $_key => $_val) {
41          switch($_key) {
42          case 'name':
43          case 'separator':
44             $$_key = $_val;
45             break;
46    
47          case 'options':
48             $$_key = (array)$_val;
49             break;
50    
51          case 'values':
52          case 'output':
53             $$_key = array_values((array)$_val);
54         break;
55    
56          case 'checked':
57          case 'selected':
58             $selected = array_values((array)$_val);
59             break;
60    
61          case 'checkboxes':
62             $smarty->trigger_error('html_checkboxes: the use of the "checkboxes" attribute is deprecated, use "options" instead', E_USER_WARNING);
63             $options = (array)$_val;
64             break;
65    
66          default:
67             $extra .= ' '.$_key.'="'.smarty_function_escape_special_chars((string)$_val).'"';
68             break;
69          }
70       }
71    
72       if (!isset($options) && !isset($values))
73          return ''; /* raise error here? */
74    
75       settype($selected, 'array');
76       $_html_result = '';
77    
78       if (is_array($options)) {
79    
80          foreach ($options as $_key=>$_val)
81             $_html_result .= smarty_function_html_checkboxes_output($name, $_key, $_val, $selected, $extra, $separator);
82    
83    
84       } else {
85          foreach ($values as $_i=>$_key) {
86             $_val = isset($output[$_i]) ? $output[$_i] : '';

88          }
89

97       $_output = '<input type="checkbox" name="'

100   
101      if (in_array($value, $selected)) {

103      }
