82        /**
83         * Connect to a database and log in as the specified user.
84         *
85         * @param $dsn the data source name (see DB::parseDSN for syntax)
86         * @param $persistent (optional) whether the connection should
87         *         be persistent
88         *
89         * @return int DB_OK on success, a DB error code on failure

91        function connect($dsninfo, $persistent = false)
92        {
93            if (!DB::assertExtension('oci8')) {
94                return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
95            }
96            $this->dsn = $dsninfo;
97            $user = $dsninfo['username'];
98            $pw = $dsninfo['password'];
99            $hostspec = $dsninfo['hostspec'];
100   
101           $connect_function = $persistent ? 'OCIPLogon' : 'OCILogon';
102   
103           if ($hostspec) {
104               $conn = @$connect_function($user,$pw,$hostspec);
105           } elseif ($user || $pw) {
106               $conn = @$connect_function($user,$pw);
107           } else {
108               $conn = false;
109           }
110           if ($conn == false) {
111               $error = OCIError();
112               $error = (is_array($error)) ? $error['message'] : null;
113               return $this->raiseError(DB_ERROR_CONNECT_FAILED, null, null,
114                                        null, $error);
115           }
116           $this->connection = $conn;
117           return DB_OK;
118       }

139        * Send a query to oracle and return the results as an oci8 resource

141        *
142        * @param $query the SQL query
143        *
144        * @return int returns a valid oci8 result for successful SELECT

149       {
150           $this->last_query = $query;
151           $query = $this->modifyQuery($query);
152           $result = @OCIParse($this->connection, $query);
153           if (!$result) {

192        *
193        * @param $result oci8 result identifier
194        * @param $arr (reference) array where data from the row is stored
195        * @param $fetchmode how the array data should be indexed

200       function fetchInto($result, &$arr, $fetchmode = DB_FETCHMODE_DEFAULT, $rownum=NULL)
201       {
202           if ($rownum !== NULL) {
203               return $this->raiseError(DB_ERROR_NOT_CAPABLE);
204           }
205           if ($fetchmode & DB_FETCHMODE_ASSOC) {
206               $moredata = @OCIFetchInto($result,$arr,OCI_ASSOC+OCI_RETURN_NULLS+OCI_RETURN_LOBS);
207               if ($moredata && $this->options['optimize'] == 'portability') {
208                   $arr = array_change_key_case($arr, CASE_LOWER);
209               }
210           } else {
211               $moredata = @OCIFetchInto($result,$arr,OCI_RETURN_NULLS+OCI_RETURN_LOBS);
212           }
213           if (!$moredata) {
214               return NULL;
215           }
216           return DB_OK;
217       }

267           // emulate numRows for Oracle.  yuck.

330       // {{{ prepare()

332       /**
333        * Prepares a query for multiple execution with execute().  With
334        * oci8, this is emulated.
335        * @param $query query to be prepared
336        *
337        * @return DB statement resource
338        */
339       function prepare($query)
340       {
341           $tokens = split('[\&\?]', $query);
342           $token = 0;
343           $types = array();
344           $qlen = strlen($query);
345           for ($i = 0; $i < $qlen; $i++) {
346               switch ($query[$i]) {
347                   case '?':
348                       $types[$token++] = DB_PARAM_SCALAR;
349                       break;
350                   case '&':
351                       $types[$token++] = DB_PARAM_OPAQUE;
352                       break;
353               }
354           }
355           $binds = sizeof($tokens) - 1;
356           $newquery = '';
357           for ($i = 0; $i < $binds; $i++) {
358               $newquery .= $tokens[$i] . ":bind" . $i;
359           }
360           $newquery .= $tokens[$i];
361           $this->last_query = $query;
362           $newquery = $this->modifyQuery($newquery);
363           $stmt = @OCIParse($this->connection, $newquery);
364           $this->prepare_types[$stmt] = $types;
365           $this->manip_query[(int)$stmt] = DB::isManip($query);
366           return $stmt;
367       }

370       // {{{ execute()
371   
372       /**
373        * Executes a DB statement prepared with prepare().
374        *
375        * @param $stmt a DB statement resource (returned from prepare())
376        * @param $data data to be used in execution of the statement
377        *
378        * @return int returns an oci8 result resource for successful
379        *  SELECT queries, DB_OK for other successful queries.  A DB error
380        *  code is returned on failure.
381        */
382       function &execute($stmt, $data = false)
383       {
384           $types=&$this->prepare_types[$stmt];
385           if (($size = sizeof($types)) != sizeof($data)) {
386               return $this->raiseError(DB_ERROR_MISMATCH);
387           }
388           for ($i = 0; $i < $size; $i++) {
389               if (is_array($data)) {
390                   $pdata[$i] = &$data[$i];
391               }
392               else {
393                   $pdata[$i] = &$data;
394               }
395               if ($types[$i] == DB_PARAM_OPAQUE) {
396                   $fp = fopen($pdata[$i], "r");
397                   $pdata[$i] = '';
398                   if ($fp) {
399                       while (($buf = fread($fp, 4096)) != false) {
400                           $pdata[$i] .= $buf;
401                       }
402                       fclose($fp);
403                   }
404               }
405               if (!@OCIBindByName($stmt, ":bind" . $i, $pdata[$i], -1)) {
406                   return $this->oci8RaiseError($stmt);
407               }
408           }
409           if ($this->autoCommit) {
410               $success = @OCIExecute($stmt, OCI_COMMIT_ON_SUCCESS);
411           } else {
412               $success = @OCIExecute($stmt, OCI_DEFAULT);
413           }
414           if (!$success) {
415               return $this->oci8RaiseError($stmt);
416           }
417           $this->last_stmt = $stmt;
418           if ($this->manip_query[(int)$stmt]) {
419               return DB_OK;
420           } else {
421               return new DB_result($this, $stmt);
422           }
423       }

510       /**
511       * Emulate the row limit support altering the query
512       *
513       * @param string $query The query to treat
514       * @param int    $from  The row to start to fetch from
515       * @param int    $count The offset
516       * @return string The modified query
517       *
518       * @author Tomas V.V.Cox <cox@idecnet.com>
519       */
520       function modifyLimitQuery($query, $from, $count)
521       {
522           // Let Oracle return the name of the columns instead of
523           // coding a "home" SQL parser
524           $q_fields = "SELECT * FROM ($query) WHERE NULL = NULL";
525           if (!$result = OCIParse($this->connection, $q_fields)) {
526               return $this->oci8RaiseError();
527           }
528           if (!OCIExecute($result, OCI_DEFAULT)) {
529               return $this->oci8RaiseError($result);
530           }
531           $ncols = OCINumCols($result);
532           $cols  = array();
533           for ( $i = 1; $i <= $ncols; $i++ ) {
534               $cols[] = OCIColumnName($result, $i);
535           }
536           $fields = implode(', ', $cols);
537           // XXX Test that (tip by John Lim)
538           //if(preg_match('/^\s*SELECT\s+/is', $query, $match)) {
539           //    // Introduce the FIRST_ROWS Oracle query optimizer
540           //    $query = substr($query, strlen($match[0]), strlen($query));
541           //    $query = "SELECT /* +FIRST_ROWS */ " . $query;
542           //}
543   
544           // Construct the query
545           // more at: http://marc.theaimsgroup.com/?l=php-db&m=99831958101212&w=2
546           // Perhaps this could be optimized with the use of Unions
547           $from += 1; // in Oracle rownum starts at 1
548           $query = "SELECT $fields FROM".
549                    "  (SELECT rownum as linenum, $fields FROM".
550                    "      ($query)".
551                    "  WHERE rownum <= ". ($from + $count) .
552                    ") WHERE linenum >= $from";
553           return $query;
554       }

559       /**
560        * Get the next value in a sequence.  We emulate sequences
561        * for MySQL.  Will create the sequence if it does not exist.
562        *
563        * @access public
564        *
565        * @param $seq_name the name of the sequence
566        *
567        * @param $ondemand whether to create the sequence table on demand
568        *  (default is true)
569        *
570        * @return a sequence integer, or a DB error
571        */
572       function nextId($seq_name, $ondemand = true)
573       {
574           $seqname = $this->getSequenceName($seq_name);
575           $repeat = 0;
576           do {
577               $this->expectError(DB_ERROR_NOSUCHTABLE);
578               $result = $this->query("SELECT ${seqname}.nextval FROM dual");
579               $this->popExpect();
580               if ($ondemand && DB::isError($result) &&
581                   $result->getCode() == DB_ERROR_NOSUCHTABLE) {
582                   $repeat = 1;
583                   $result = $this->createSequence($seq_name);
584                   if (DB::isError($result)) {
585                       return $this->raiseError($result);
586                   }
587               } else {
588                   $repeat = 0;
589               }
590           } while ($repeat);
591           if (DB::isError($result)) {
592               return $this->raiseError($result);
593           }
594           $arr = $result->fetchRow(DB_FETCHMODE_ORDERED);
595           return $arr[0];
596       }

654       // {{{ tableInfo()
655   
656       function tableInfo($result, $mode = null)
657       {
658           $count = 0;
659           $res   = array();
660           /*
661            * depending on $mode, metadata returns the following values:
662            *
663            * - mode is false (default):
664            * $res[]:
665            *   [0]["table"]       table name
666            *   [0]["name"]        field name
667            *   [0]["type"]        field type
668            *   [0]["len"]         field length
669            *   [0]["nullable"]    field can be null (boolean)
670            *   [0]["format"]      field precision if NUMBER
671            *   [0]["default"]     field default value
672            *
673            * - mode is DB_TABLEINFO_ORDER
674            * $res[]:
675            *   ["num_fields"]     number of fields
676            *   [0]["table"]       table name
677            *   [0]["name"]        field name
678            *   [0]["type"]        field type
679            *   [0]["len"]         field length
680            *   [0]["nullable"]    field can be null (boolean)
681            *   [0]["format"]      field precision if NUMBER
682            *   [0]["default"]     field default value
683            *   ['order'][field name] index of field named "field name"
684            *   The last one is used, if you have a field name, but no index.
685            *   Test:  if (isset($result['order']['myfield'])) { ...
686            *
687            * - mode is DB_TABLEINFO_ORDERTABLE
688            *    the same as above. but additionally
689            *   ["ordertable"][table name][field name] index of field
690            *      named "field name"
691            *
692            *      this is, because if you have fields from different
693            *      tables with the same field name * they override each
694            *      other with DB_TABLEINFO_ORDER
695            *
696            *      you can combine DB_TABLEINFO_ORDER and
697            *      DB_TABLEINFO_ORDERTABLE with DB_TABLEINFO_ORDER |
698            *      DB_TABLEINFO_ORDERTABLE * or with DB_TABLEINFO_FULL
699            */
700   
701           // if $result is a string, we collect info for a table only
702           if (is_string($result)) {
703               $result = strtoupper($result);
704               $q_fields = "select column_name, data_type, data_length, data_precision,
705                            nullable, data_default from user_tab_columns
706                            where table_name='$result' order by column_id";
707               if (!$stmt = OCIParse($this->connection, $q_fields)) {
708                   return $this->oci8RaiseError();
709               }
710               if (!OCIExecute($stmt, OCI_DEFAULT)) {
711                   return $this->oci8RaiseError($stmt);
712               }
713               while (OCIFetch($stmt)) {
714                   $res[$count]['table']       = $result;
715                   $res[$count]['name']        = @OCIResult($stmt, 1);
716                   $res[$count]['type']        = @OCIResult($stmt, 2);
717                   $res[$count]['len']         = @OCIResult($stmt, 3);
718                   $res[$count]['format']      = @OCIResult($stmt, 4);
