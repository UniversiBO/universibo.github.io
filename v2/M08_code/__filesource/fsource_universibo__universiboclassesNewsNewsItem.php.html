358   	 *
359   	 * @static
360   	 * @param array $id_notizie array elenco di id della news
361   	 * @return NewsItems 
362   	 */
363        function &selectNewsItems ($id_notizie)
364        {
365            
366            $db =& FrontController::getDbConnection('main');
367           
368           if ( count($id_notizie) == 0 )
369               return array();
370           
371           //esegue $db->quote() su ogni elemento dell'array
372           array_walk($id_notizie, array($db, 'quote'));
373           $values = implode(',',$id_notizie);
374           
375           $query = 'SELECT titolo, notizia, data_inserimento, data_scadenza, flag_urgente, eliminata, A.id_utente, id_news, username FROM news A, utente B WHERE A.id_utente = B.id_utente AND id_news IN ('.$values.') AND eliminata!='.$db->quote(NEWS_ELIMINATA);
376           $res =& $db->query($query);
377           if (DB::isError($res)) 
378               Error::throw(_ERROR_CRITICAL,array('msg'=>DB::errorMessage($res),'file'=>__FILE__,'line'=>__LINE__)); 
379       
380           $rows = $res->numRows();
381   
382           if( $rows = 0) return false;
383           $news_list = array();
384       
385           while ( $res->fetchInto($row) )
386           {
387               $news_list[]=& new NewsItem($row[7],$row[0],$row[1],$row[2],$row[3],$row[4],$row[5],$row[6],$row[8]);
388           }
389           
390           $res->free();
391           
392           return $news_list;
393        }
