23     function smarty_outputfilter_trimwhitespace($source, &$smarty)
24     {
25        // Pull out the script blocks
26        preg_match_all("!<script[^>]+>.*?</script>!is", $source, $match);
27        $_script_blocks = $match[0];
28        $source = preg_replace("!<script[^>]+>.*?</script>!is",
29        '@@@SMARTY:TRIM:SCRIPT@@@', $source);
30    
31        // Pull out the pre blocks
32        preg_match_all("!<pre>.*?</pre>!is", $source, $match);
33        $_pre_blocks = $match[0];
34        $source = preg_replace("!<pre>.*?</pre>!is",
35        '@@@SMARTY:TRIM:PRE@@@', $source);
36

42    
43        // remove all leading spaces, tabs and carriage returns NOT
44        // preceeded by a php close tag.
45              $source = trim(preg_replace('/((?<!\?>)\n)[\s]+/m', '\1', $source));
46    
47        // replace script blocks
48        smarty_outputfilter_trimwhitespace_replace("@@@SMARTY:TRIM:SCRIPT@@@",$_script_blocks, $source);

50        // replace pre blocks
