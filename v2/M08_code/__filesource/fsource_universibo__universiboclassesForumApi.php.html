31        /**
32    	 * esegue la codifica esadecimale di un ipv4 nel formato separato da punti
33    	 * es: '127.0.0.1' -> '7f000001'
34    	 *
35    	 * @access private
36    	 * @param string codifica separata da punti di un numero ip
37    	 * @return string codifica esadecimale del numero ip
38    	 */
39        function _encodeIp($dotquad_ip)
40        {
41            $ip_sep = explode('.', $dotquad_ip);
42            return sprintf('%02x%02x%02x%02x', $ip_sep[0], $ip_sep[1], $ip_sep[2], $ip_sep[3]);
43        }

49        function getForumSid()
50        {
51            //echo $_SESSION['phpbb_sid'];

59    	 * Inserisce le informazioni di sessione e cookie per mantenere traccia dell'utente

83            if( $rows != 3) 
84                Error::throw(_ERROR_DEFAULT,array('msg'=>'Impossibile trovare le informazioni di configurazione del forum','file'=>__FILE__,'line'=>__LINE__)); 
85            while (    $res->fetchInto($row) )
86            {
87                ${$row[0]} = $row[1];
88            }
89            //echo $cookie_domain;

91            //echo $cookie_secure;
92    
93            $phpbb2_cookie = array();
94            $phpbb2_cookie['autologinid'] = ''; 
95            $phpbb2_cookie['userid'] = (string)$user->getIdUser() ; 
96            $cookie_value = serialize($phpbb2_cookie);
97            setcookie ('phpbb2_data', $cookie_value, time()+3600, $cookie_path, $cookie_domain , $cookie_secure);

103           $res = $db->query($query);
104           if (DB::isError($res))

106   
107           $_SESSION['phpbb_sid'] = $sid;
