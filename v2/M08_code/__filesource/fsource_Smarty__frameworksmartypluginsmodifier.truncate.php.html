15    function smarty_modifier_truncate($string, $length = 80, $etc = '...',
16                                      $break_words = false)

18        if ($length == 0)
19            return '';
20    
21        if (strlen($string) > $length) {
22            $length -= strlen($etc);    
23            if (!$break_words)
24            $string = preg_replace('/\s+?(\S+)?$/', '', substr($string, 0, $length+1));
