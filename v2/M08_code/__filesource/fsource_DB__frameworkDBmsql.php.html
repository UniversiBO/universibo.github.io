60        // {{{ connect()
61    
62        function connect($dsninfo, $persistent = false)
63        {
64            if (!DB::assertExtension('msql'))
65                return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
66    
67            $this->dsn = $dsninfo;
68            $user = $dsninfo['username'];
69            $pw = $dsninfo['password'];
70            $dbhost = $dsninfo['hostspec'] ? $dsninfo['hostspec'] : 'localhost';
71    
72            $connect_function = $persistent ? 'msql_pconnect' : 'msql_connect';
73    
74            if ($dbhost && $user && $pw) {
75                $conn = $connect_function($dbhost, $user, $pw);
76            } elseif ($dbhost && $user) {
77                $conn = $connect_function($dbhost,$user);
78            } else {
79                $conn = $connect_function($dbhost);
80            }
81            if (!$conn) {
82                $this->raiseError(DB_ERROR_CONNECT_FAILED);
83            }
84            if (!@msql_select_db($dsninfo['database'], $conn)){
85                return $this->raiseError(DB_ERROR_NODBSELECTED);
86            }
87            $this->connection = $conn;
88            return DB_OK;
89        }
90

94        function disconnect()
95        {
96            $ret = @msql_close($this->connection);
97            $this->connection = null;
98            return $ret;

104       function simpleQuery($query)
105       {
106           $this->last_query = $query;
107           $query = $this->modifyQuery($query);
108           $result = @msql_query($query, $this->connection);
109           if (!$result) {
110               return $this->raiseError();
111           }
112           // Determine which queries that should return data, and which
113           // should return an error code only.
114           return DB::isManip($query) ? DB_OK : $result;
115       }

138       function fetchInto($result, &$ar, $fetchmode, $rownum=null)
139       {
140           if ($rownum !== null) {
141               if (!@msql_data_seek($result, $rownum)) {
142                   return null;
143               }
144           }
145           if ($fetchmode & DB_FETCHMODE_ASSOC) {
146               $ar = @msql_fetch_array($result, MSQL_ASSOC);
147           } else {
148               $ar = @msql_fetch_row($result);
149           }
150           if (!$ar) {
151               if ($error = msql_error()) {
152                   return $this->raiseError($error);
153               } else {
154                   return null;
155               }
156           }
157           return DB_OK;
158       }

163       function freeResult($result)
164       {
165           if (is_resource($result)) {
166               return @msql_free_result($result);
167           }
168           if (!isset($this->prepare_tokens[$result])) {
169               return false;
170           }
171           unset($this->prepare_tokens[$result]);
172           unset($this->prepare_types[$result]);
173           return true;
174       }
