89     * Author: Deepak Dutta, http://www.eocene.net
90     * Unrestricted license, subject to no modifcation to the line above
91     * Please include any modification history
92     * 10/01/2002 Initial creation
93     * TemplateEngine class is used to parse the template
94     *
95     * PUBLIC PROPERTIES
96     *	var $fileName		Name of the file including the path
97     *  var $fileContents	Entire contents of the file
98     *  var $variables		array of all the variables (anything between {} in the template)
99     *  var $loops			name of all the loops
100    *  var $blockNames		name of all the blocks
101    *  var $unresolvedVars	variables that were found in template but not in $variables
102    * PUBLIC METHODS
103    *	setVar(&$varName,&$varValue)		sets variables
104    *	setLoop(&$loopName, &$loopData)		sets loops
105    *	setBlock(&$aName)					sets blocks
106    *	process(&$fileName,&$templateRoot)	process the template. Should be called after setting var, loops, and blocks
107    * 										The processed template can be extracted from $fileContents
108   */
109   /**
110   class TemplateEngine{
111   	var $fileName;
112   	var $fileContents;
113   	var $variables = array();
114   	var $loops = array();
115   	var $blockNames = array();
116   	//Use it to set variables
117   	function setVar(&$varName, &$varValue){
118   		if (is_array($varValue)) {
119   			foreach($varValue as $key=>$value){
120   				$newName=$varName . '.' . $key;
121   				$this->setVar($newName, $value);
122   			}
123   		}
124   		else {
125   			$this->variables[$varName] = $varValue;
126   		}
127   	}
128   	//Use it to set loops
129   	function setLoop(&$loopName, &$loopData){
130   		if (!$loopData) $loopData = 0;
131   		$this->loops[$loopName] = $loopData;
132   	}
133   	//Use it to set blocks
134   	function setBlock(&$aName){
135   		$this->blockNames[$aName]="";
136   	}
137   	//Call this method to process template after everything is set
138   	//$fileName has full path, $templateRoot is used to resolve includes
139   	function process(&$fileName,&$templateRoot){
140   		$this->fileName=$fileName;
141   		$this->fileContents=&$this->getFileContents($this->fileName);
142   		$this->processInclude($templateRoot);	
143   		$this->processBlocks();
144   		$this->processLoops();
145   		$this->processVariables();
146   		$this->processPlugs();		
147   	}
148   	function &getFileContents(&$fileName){
149   		if (!file_exists($fileName))
150   		  $this->error("TemplateEngine: file $fileName does not exist.");
151   		$fileContents = fread($fp = fopen($fileName, 'r'), filesize($fileName));
152   		fclose($fp);
153   		return $fileContents;
154   	}
155   	function processInclude(&$templateRoot){
156   		$fileContents = &$this->fileContents;
157   		while(is_long($position = strpos($fileContents, '<include filename="'))){
158   			$position += 19;
159   			$endPosition = strpos($fileContents, '"/>', $position);
160   			$includeFileName = substr($fileContents, $position, $endPosition-$position);
161   			$replaceThis = '<include filename="'.$includeFileName.'"/>';
162   			$includeFileName=&$this->getIncludedFileWithPath($includeFileName,$templateRoot);
163   			$includeContents = &$this->getFileContents($includeFileName);
164   			$fileContents = str_replace($replaceThis, $includeContents, $fileContents);
165   		}
166   	}
167   	
168   	function processPlugs(){
169   		$fileContents = &$this->fileContents;
170   		while(is_long($position = strpos($fileContents, '<plug command="'))){
171   			$position += 15;
172   			$endPosition = strpos($fileContents, '"/>', $position);
173   			$plugAction = substr($fileContents, $position, $endPosition-$position);
174   			$replaceThis = '<plug command="'.$plugAction.'"/>';
175   			
176   			global $fc;
177   			$plugContents = &$fc->getPlugContents($plugAction);
178   			$fileContents = str_replace($replaceThis, $plugContents, $fileContents);
179   		}
180   	}
181   	
182   	function processLoops(){
183   		$fileContents = &$this->fileContents;
184   		while (list($loopName, $loopArgs) = each($this->loops) ){
185   			$startTag = strpos($fileContents, '<loop start="'.$loopName.'"/>');
186   			$startPosition = $startTag + strlen('<loop start="'.$loopName.'"/>');
187   			if (!$startPosition) continue;
188   			$endPosition = strpos($fileContents, '<loop end="'.$loopName.'"/>');
189   			$loopContents = substr($fileContents, $startPosition, $endPosition-$startPosition);
190   			$originalLoopContents = $loopContents;
191   			
192   			$startTag = substr($fileContents, $startTag, strlen('<loop start="'.$loopName.'"/>'));
193   			$endTag = substr($fileContents, $endPosition, strlen('<loop end="'.$loopName.'"/>'));
194   			if($loopContents == '') continue;
195   		
196   			$parsedContents = '';
197   			if (is_array($loopArgs)){
198   				$theKeys = array_keys($loopArgs);
199   				$numKeys = count($theKeys);
200   				for($i = 0; ($i< $numKeys); $i++){	
201   					$holdIt = $loopContents;
202   					foreach( $loopArgs[$theKeys[$i]] as $k=>$v){
203   						$holdIt = str_replace('{'. $loopName. '.' .$k. '}',$v,$holdIt);
204   					}
205   					$parsedContents .= $holdIt;
206   				}	
207   			} 	
208   			$fileContents =str_replace($startTag.$originalLoopContents.$endTag,$parsedContents,$fileContents);
209   		}
210   	}
211   	function processVariables(){
212   		if(empty($this->variables)) return;
213   		
214   		$fileContents=&$this->fileContents;
215   		$allStringsBeforeParen=explode('{',$fileContents);
216   		$numSizeBeforeParen=count($allStringsBeforeParen);
217   		for($i=1;$i<$numSizeBeforeParen;$i++){
218   			$templateVariable=explode('}',$allStringsBeforeParen[$i]);
219   			if(count($templateVariable)==0){
220   				$this->error("No closing } for ".$allStringsBeforeParen[$i-1]);
221   			}
222   			if(isset($this->variables[$templateVariable[0]])){
223   				$templateVariable[0] = $this->variables[$templateVariable[0]];
224   				$allStringsBeforeParen[$i]=implode("",$templateVariable);
225   			}
226   			else{
227   				$allStringsBeforeParen[$i]=implode("}",$templateVariable);
228   				$allStringsBeforeParen[$i] ="{".$allStringsBeforeParen[$i];
229   			}
230   		}
231   		$fileContents=implode("",$allStringsBeforeParen);
232   	}
233   	function processBlocks(){
234   		$templateString=&$this->fileContents;
235   		$blockStartAt=strpos($templateString, '<block start="',0);
236   		if(!is_long($blockStartAt)) return;		
237   	
238   		while(is_long($blockStartAt)){
239   			$this->processTemplateStringForBlock($templateString,$blockStartAt);
240   			$templateString=&$this->fileContents;
241   			$blockStartAt=strpos($templateString, '<block start="',0);	
242   		}
243   	}
244   	
245   	function processTemplateStringForBlock(&$templateString,$blockStartAt){
246   		if(!is_long($blockStartAt)) return;
247   		
248   		$blockNameStartAt=$blockStartAt+14;
249   		$blockNameEndAt=strpos($templateString,'"/>',$blockNameStartAt);
250   		$blockNameLength=$blockNameEndAt-$blockNameStartAt;
251   		$blockName=substr($templateString,$blockNameStartAt,$blockNameLength);
252   		if($blockNameLength>50 || $blockNameLength<=0)
253   			$this->error("Block name $blockName should be between 1 and 50 characters");
254   		
255   		$startTag='<block start="'.$blockName.'"/>';
256   		$endTag='<block end="'.$blockName.'"/>';
257   		$blockEndTagStartsAt=strpos($templateString,$endTag,$blockNameStartAt);
258   		if(!is_long($blockEndTagStartsAt))
259   			$this->error("Cannot find end tag for block name $blockName");
260   		$blockEndAt=$blockEndTagStartsAt+strlen($endTag);
261   		$blockLength=$blockEndAt-$blockStartAt;
262   		if(isset($this->blockNames[$blockName])){
263   			$searchString=array();
264   			$searchString[0]=$startTag;
265   			$searchString[1]=$endTag;
266   			$templateString=str_replace($searchString,'',$templateString);	
267   		}
268   		else{
269   			$templateString=substr_replace($templateString,'',$blockStartAt,$blockLength);
270   		}
271   	}

280   		return $templateRoot.'/'.$fileName;
281   	}
