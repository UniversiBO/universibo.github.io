35    class DB_dbase extends DB_common

37        // {{{ properties

39        var $connection;
40        var $phptype, $dbsyntax;
41        var $prepare_tokens = array();
42        var $prepare_types = array();
43        var $transaction_opcount = 0;
44        var $res_row = array();

50        /**
51         * DB_mysql constructor.
52         *
53         * @access public
54         */
55    
56        function DB_dbase()
57        {
58            $this->DB_common();
59            $this->phptype = 'dbase';
60            $this->dbsyntax = 'dbase';
61            $this->features = array(
62                'prepare'       => false,
63                'pconnect'      => false,
64                'transactions'  => false,
65                'limit'         => false
66            );
67            $this->errorcode_map = array();
68        }
69    
70        // }}}
71        // {{{ connect()
72    
73        function connect($dsninfo, $persistent = false)
74        {
75            if (!DB::assertExtension('dbase')) {
76                return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
77            }
78            $this->dsn = $dsninfo;
79            ob_start();
80            $conn  = dbase_open($dsninfo['database'], 0);
81            $error = ob_get_contents();
82            ob_end_clean();
83            if (!$conn) {
84                return $this->raiseError(DB_ERROR_CONNECT_FAILED, null,
85                                         null, null, strip_tags($error));
86            }
87            $this->connection = $conn;
88            return DB_OK;
89        }
90    
91        // }}}
92        // {{{ disconnect()
93    
94        function disconnect()
95        {
96            $ret = dbase_close($this->connection);
97            $this->connection = null;
98            return $ret;
99        }
100   
101       // }}}
102       // {{{ &query()
103   
104       function &query($query = null)
105       {
106           // emulate result resources
107           $this->res_row[$this->result] = 0;
108           return new DB_result($this, $this->result++);
109       }
110   
111       // }}}
112       // {{{ fetchInto()
113   
114       function fetchInto($res, &$row, $fetchmode, $rownum = null)
115       {
116           if ($rownum === null) {
117               $rownum = $this->res_row[$res]++;
118           }
119           if ($fetchmode & DB_FETCHMODE_ASSOC) {
120               $row = @dbase_get_record_with_names($this->connection, $rownum);
121           } else {
122               $row = @dbase_get_record($this->connection, $rownum);
123           }
124           if (!$row) {
125               return null;
126           }
127           return DB_OK;
128       }
129   
130       // }}}

136       }
