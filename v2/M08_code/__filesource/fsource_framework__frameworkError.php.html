121           if ( $error_category === NULL ) $error_category = $this->error_category;

164       function retrieve($error_category)
165       {
166   
167           global $_Error_repository
168   
169           $count = count($_Error_repository);
170           for ($i=0; $i<$count; $i++)
171           {
172               //var_dump($_Error_repository);
173               if ( $_Error_repository[$i] !== NULL  &&  $_Error_repository[$i]->error_category === $error_category )
174               {
175                   $current_error = $_Error_repository[$i];
176                   $_Error_repository[$i] = NULL;
177                   return $current_error;
178               }
179           
180           }
181           
182           return false;
