128   	*
129   	* Action is in receiver.php?do=someAction
130   	* A class SomeAction should exists and someAction should be
131       * associated with this class in the config file.  Class SomeAction
132       * is the command controller that can serve one or more commands.
133   	*	
134   	* @access public
135   	*/
136       function executeCommand(){
137   
138           include_once ('BaseCommand'.PHP_EXTENSION);    
139   
140           //$command_request=$this->getCommandRequest();
141           $command_class=$this->getCommandClass();
142           include_once($this->paths['commands'].'/'.$command_class.PHP_EXTENSION);
143           
144           $command = new $command_class;
145           
146           $command->initCommand($this);
147           $response = $command->execute();
148           $command->shutdownCommand();
149           
150           
151           if ($response == NULL) $response='default';
152           if (array_key_exists($response, $this->commandTemplate))
153           {
154               $template = $this->commandTemplate[$response];
155               
156               $templateEngine =& $this->getTemplateEngine();
157               if (!$templateEngine->template_exists($template))
158                   Error::throw(_ERROR_CRITICAL,array('msg'=>'Non è presente il file relativo al template specificato','file'=>__FILE__,'line'=>__LINE__));
159               
160               $templateEngine->display($template);

690
