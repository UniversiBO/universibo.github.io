18    	 * Handler per errori di categoria ERROR_CRITICAL
19    	 * Salva l'indicazione dell'errore sul LogHandler, insieme a tutti i parametri della richiesta,
20    	 * Mostra sul browser un messaggio di errore e interrompe l'esecuzione
21    	 *
22    	 * @param $param mixed,array() Tipo restituito da chi cattura l'errore,
23    	 *  questo handler è in grado di gestire un parametro array avente la seguente struttura
24    	 *  $param = array(  "msg"=>"messaggio di errore da mostrare",
25    	 *  					"file"=>"file in cui è avvenuto l'errore",
26    	 *  					"line"=>"linea di codice in cui è avvenuto l'errore",
27    	 *  					"log"=>"(opzionale)se impostato viene salvato come messaggio sul LogHandler"
28    	 *  					)
29    	 */
30        function critical_handler($param)
31        {
32            $param['log']= true;
33            die( 'Errore Critico: '.$param['msg']. '<br />
34    		file: '.$param['file']. '<br />
35    		line: '.$param['line']. '<br />
36    		log: '.$param['log']. '<br />');
37    
38            //header('Redirect: http://location/error_page.php');
39        }

41        /**
42    	 * Handler per errori di categoria ERROR_DEFAULT
43    	 * Salva opzionalmente l'indicazione dell'errore sul LogHandler, insieme a tutti i parametri della richiesta,
44    	 * Redirige il browser ad una pagina dell'applicazione in cui viene mostrato il messaggio di errore
45    	 *
46    	 * @param $param mixed,array() Tipo restituito da chi cattura l'errore,
47    	 *  questo handler è in grado di gestire un parametro array avente la seguente struttura
48    	 *  $param = array(  "msg"=>"messaggio di errore da mostrare",
49    	 *  					"file"=>"file in cui è avvenuto l'errore",
50    	 *  					"line"=>"linea di codice in cui è avvenuto l'errore",
51    	 *  					"log"=>"(opzionale)se impostato viene salvato come messaggio sul LogHandler"
52    	 *  					)
53    	 */
54        function default_handler($param)
55        {
56            $_SESSION['error_param'] = $param;
57    
58            FrontController::redirectCommand('ShowError');
59            exit(1);
60        }

78

92            $template->append('error_notice', $param['msg']);

101       }
