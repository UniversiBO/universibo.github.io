151       function simpleQuery($query)
152       {
153           $ismanip = DB::isManip($query);
154           $this->last_query = $query;
155           $query = $this->modifyQuery($query);
156           $result = @ibase_query($this->connection, $query);
157           if (!$result) {
158               return $this->ibaseRaiseError();
159           }
160           if ($this->autocommit && $ismanip) {
161               ibase_commit($this->connection);
162           }
163           // Determine which queries that should return data, and which
164           // should return an error code only.
165           return DB::isManip($query) ? DB_OK : $result;

218       function fetchInto($result, &$ar, $fetchmode, $rownum = null)
219       {
220           if ($rownum !== NULL) {
221               return $this->ibaseRaiseError(DB_ERROR_NOT_CAPABLE);
222           }
223           if ($fetchmode & DB_FETCHMODE_ASSOC) {
224               if (function_exists('ibase_fetch_assoc')) {
225                   $ar = @ibase_fetch_assoc($result);
226               } else {
227                   $ar = get_object_vars(ibase_fetch_object($result));
228               }
229               if ($ar && $this->options['optimize'] == 'portability') {
230                   $ar = array_change_key_case($ar, CASE_LOWER);
231               }
232           } else {
233               $ar = @ibase_fetch_row($result);
234           }
235           if (!$ar) {
236               if ($errmsg = ibase_errmsg()) {
237                   return $this->ibaseRaiseError(null, $errmsg);
238               } else {
239                   return null;
240               }
241           }
242           return DB_OK;
243       }

286       /**
287        * Prepares a query for multiple execution with execute().
288        * @param $query query to be prepared
289        *
290        * @return DB statement resource
291        */
292       function prepare($query)
293       {
294           $tokens = split('[\&\?]', $query);
295           $token = 0;
296           $types = array();
297           $qlen = strlen($query);
298           for ($i = 0; $i < $qlen; $i++) {
299               switch ($query[$i]) {
300                   case '?':
301                       $types[$token++] = DB_PARAM_SCALAR;
302                       break;
303                   case '&':
304                       $types[$token++] = DB_PARAM_OPAQUE;
305                       break;
306               }
307           }
308           $newquery = strtr($query, '&', '?');
309           $this->last_query = $query;
310           $newquery = $this->modifyQuery($newquery);
311           $stmt = ibase_prepare($this->connection, $newquery);
312           $this->prepare_types[(int)$stmt] = $types;
313           $this->manip_query[(int)$stmt]   = DB::isManip($query);
314           return $stmt;
315       }

320       /**
321        * Executes a DB statement prepared with prepare().
322        *
323        * @param $stmt a DB statement resource (returned from prepare())
324        * @param $data data to be used in execution of the statement
325        *
326        * @return int returns an oci8 result resource for successful
327        *  SELECT queries, DB_OK for other successful queries.  A DB error
328        *  code is returned on failure.
329        */
330       function &execute($stmt, $data = false)
331       {
332           $types=&$this->prepare_types[$stmt];
333           if (($size = sizeof($types)) != sizeof($data)) {
334               return $this->raiseError(DB_ERROR_MISMATCH);
335           }
336           $pdata[0] = $stmt;
337           for ($j = 0; $j < $size; $j++) {
338               $i = $j + 1;
339               if (is_array($data)) {
340                   $pdata[$i] = &$data[$j];
341               } else {
342                   $pdata[$i] = &$data;
343               }
344               if ($types[$j] == DB_PARAM_OPAQUE) {
345                   $fp = fopen($pdata[$i], 'r');
346                   $pdata[$i] = '';
347                   if ($fp) {
348                       while (($buf = fread($fp, 4096)) != false) {
349                           $pdata[$i] .= $buf;
350                       }
351                       fclose($fp);
352                   }
353               }
354           }
355           $res = call_user_func_array('ibase_execute', $pdata);
356           if (!$res) {
357               return $this->ibaseRaiseError();
358           }
359           /* XXX need this?
360           if ($this->autocommit && $this->manip_query[(int)$stmt]) {
361               ibase_commit($this->connection);
362           }*/
363           if ($this->manip_query[(int)$stmt]) {
364               return DB_OK;
365           } else {
366               return new DB_result($this, $res);
367           }
368       }

425       /**
426        * Get the next value in a sequence.
427        *
428        * If the sequence does not exist, it will be created,
429        * unless $ondemand is false.
430        *
431        * @access public
432        * @param string $seq_name the name of the sequence
433        * @param bool $ondemand whether to create the sequence on demand
434        * @return a sequence integer, or a DB error
435        */
436       function nextId($seq_name, $ondemand = true)
437       {
438           $sqn = strtoupper(preg_replace('/[^a-z0-9_]/i', '_', $seq_name));
439           $repeat = 0;
440           do {
441               $this->pushErrorHandling(PEAR_ERROR_RETURN);
442               $result = $this->query("SELECT GEN_ID(${sqn}_SEQ, 1) FROM RDB\$GENERATORS"
443                                      ." WHERE RDB\$GENERATOR_NAME='${sqn}_SEQ'");
444               $this->popErrorHandling();
445               if ($ondemand && DB::isError($result)) {
446                   $repeat = 1;
447                   $result = $this->createSequence($seq_name);
448                   if (DB::isError($result)) {
449                       return $result;
450                   }
451               } else {
452                   $repeat = 0;
453               }
454           } while ($repeat);
455           if (DB::isError($result)) {
456               return $this->raiseError($result);
457           }
458           $arr = $result->fetchRow(DB_FETCHMODE_ORDERED);
459           $result->free();
460           return $arr[0];
461       }

502        /**
503         * get the Flags of a Field
504         *
505         * @param string $field_name the name of the field
506         * @param string $table_name the name of the table
507         *
508         * @return string The flags of the field ("primary_key", "unique_key", "not_null"
509         *                 "default", "computed" and "blob" are supported)
510         * @access private
511         */
512        function _ibaseFieldFlags($field_name, $table_name)
513        {
514   
515            $sql = 'SELECT  R.RDB$CONSTRAINT_TYPE CTYPE'
516                   .' FROM  RDB$INDEX_SEGMENTS I'
517                   .' JOIN  RDB$RELATION_CONSTRAINTS R ON I.RDB$INDEX_NAME=R.RDB$INDEX_NAME'
518                  .' WHERE  I.RDB$FIELD_NAME=\''.$field_name.'\''
519                     .' AND R.RDB$RELATION_NAME=\''.$table_name.'\'';
520            $result = ibase_query($this->connection, $sql);
521            if (empty($result)) {
522                return $this->ibaseRaiseError();
523            }
524            if ($obj = @ibase_fetch_object($result)) {
525                ibase_free_result($result);
526                if (isset($obj->CTYPE)  && trim($obj->CTYPE) == 'PRIMARY KEY') {
527                    $flags = 'primary_key ';
528                }
529                if (isset($obj->CTYPE)  && trim($obj->CTYPE) == 'UNIQUE') {
530                    $flags .= 'unique_key ';
531                }
532            }
533   
534            $sql = 'SELECT  R.RDB$NULL_FLAG AS NFLAG,'
535                         .' R.RDB$DEFAULT_SOURCE AS DSOURCE,'
536                         .' F.RDB$FIELD_TYPE AS FTYPE,'
537                         .' F.RDB$COMPUTED_SOURCE AS CSOURCE'
538                   .' FROM  RDB$RELATION_FIELDS R '
539                   .' JOIN  RDB$FIELDS F ON R.RDB$FIELD_SOURCE=F.RDB$FIELD_NAME'
540                  .' WHERE  R.RDB$RELATION_NAME=\''.$table_name.'\''
541                    .' AND  R.RDB$FIELD_NAME=\''.$field_name.'\'';
542            $result = ibase_query($this->connection, $sql);
543            if (empty($result)) {
544                return $this->ibaseRaiseError();
545            }
546            if ($obj = @ibase_fetch_object($result)) {
547                ibase_free_result($result);
548                if (isset($obj->NFLAG)) {
549                    $flags .= 'not_null ';
550                }
551                if (isset($obj->DSOURCE)) {
552                    $flags .= 'default ';
553                }
554                if (isset($obj->CSOURCE)) {
555                    $flags .= 'computed ';
556                }
557                if (isset($obj->FTYPE)  && $obj->FTYPE == 261) {
558                    $flags .= 'blob ';
559                }
560            }
561   
562            return trim($flags);
563        }

566        // {{{ tableInfo()
567   
568        /**
569         * Returns information about a table or a result set
570         *
571         * NOTE: doesn't support 'flags'and 'table' if called from a db_result
572         *
573         * @param  mixed $resource Interbase result identifier or table name
574         * @param  int $mode A valid tableInfo mode (DB_TABLEINFO_ORDERTABLE or
575         *                    DB_TABLEINFO_ORDER)
576         *
577         * @return array An array with all the information
578         */
579        function tableInfo($result, $mode = null)
580        {
581            $count = 0;
582            $id    = 0;
583            $res   = array();
584   
585            /*
586             * depending on $mode, metadata returns the following values:
587             *
588             * - mode is false (default):
589             * $result[]:
590             *   [0]["table"]  table name
591             *   [0]["name"]   field name
592             *   [0]["type"]   field type
593             *   [0]["len"]    field length
594             *   [0]["flags"]  field flags
595             *
596             * - mode is DB_TABLEINFO_ORDER
597             * $result[]:
598             *   ["num_fields"] number of metadata records
599             *   [0]["table"]  table name
600             *   [0]["name"]   field name
601             *   [0]["type"]   field type
602             *   [0]["len"]    field length
603             *   [0]["flags"]  field flags
604             *   ["order"][field name]  index of field named "field name"
605             *   The last one is used, if you have a field name, but no index.
606             *   Test:  if (isset($result['meta']['myfield'])) { ...
607             *
608             * - mode is DB_TABLEINFO_ORDERTABLE
609             *    the same as above. but additionally
610             *   ["ordertable"][table name][field name] index of field
611             *      named "field name"
612             *
613             *      this is, because if you have fields from different
614             *      tables with the same field name * they override each
615             *      other with DB_TABLEINFO_ORDER
616             *
617             *      you can combine DB_TABLEINFO_ORDER and
618             *      DB_TABLEINFO_ORDERTABLE with DB_TABLEINFO_ORDER |
619             *      DB_TABLEINFO_ORDERTABLE * or with DB_TABLEINFO_FULL
620             */
621   
622            // if $result is a string, then we want information about a
623            // table without a resultset
624   
625            if (is_string($result)) {
626                $id = ibase_query($this->connection,"SELECT * FROM $result");
627                if (empty($id)) {
628                    return $this->ibaseRaiseError();
629                }
630            } else { // else we want information about a resultset
631                $id = $result;
632                if (empty($id)) {
633                    return $this->ibaseRaiseError();
634                }
635            }
636   
637            $count = @ibase_num_fields($id);
638   
639            // made this IF due to performance (one if is faster than $count if's)
640            if (empty($mode)) {
641   
642                for ($i=0; $i<$count; $i++) {
643                    $info = @ibase_field_info($id, $i);
644                    $res[$i]['table'] = (is_string($result)) ? $result : '';
645                    $res[$i]['name']  = $info['name'];
646                    $res[$i]['type']  = $info['type'];
647                    $res[$i]['len']   = $info['length'];
648                    $res[$i]['flags'] = (is_string($result)) ? $this->_ibaseFieldFlags($info['name'], $result) : '';
649                }
650   
651            } else { // full
652                $res["num_fields"]= $count;
653   
654                for ($i=0; $i<$count; $i++) {
655                    $info = @ibase_field_info($id, $i);
656                    $res[$i]['table'] = (is_string($result)) ? $result : '';
657                    $res[$i]['name']  = $info['name'];
658                    $res[$i]['type']  = $info['type'];
659                    $res[$i]['len']   = $info['length'];
660                    $res[$i]['flags'] = (is_string($result)) ? $this->_ibaseFieldFlags($info['name'], $result) : '';
661                    if ($mode & DB_TABLEINFO_ORDER) {
662                        $res['order'][$res[$i]['name']] = $i;
663                    }
664                    if ($mode & DB_TABLEINFO_ORDERTABLE) {
665                        $res['ordertable'][$res[$i]['table']][$res[$i]['name']] = $i;
666                    }
667                }
668            }
669   
670            // free the result only if we were called on a table
671            if (is_string($result)) {
672                ibase_free_result($id);
673            }
674            return $res;
675        }

698       function ibaseRaiseError($db_errno = null, $native_errmsg = null)
699       {
700           if ($native_errmsg === null) {
701               $native_errmsg = ibase_errmsg();
702           }
703           // memo for the interbase php module hackers: we need something similar
704           // to mysql_errno() to retrieve error codes instead of this ugly hack
705           if (preg_match('/^([^0-9\-]+)([0-9\-]+)\s+(.*)$/', $native_errmsg, $m)) {
706               $native_errno = (int)$m[2];
707           } else {
708               $native_errno = null;
709           }
710           // try to map the native error to the DB one
711           if ($db_errno === null) {
712               if ($native_errno) {
713                   // try to interpret Interbase error code (that's why we need ibase_errno()
714                   // in the interbase module to return the real error code)
715                   switch ($native_errno) {
716                       case -204:
717                           if (is_int(strpos($m[3], 'Table unknown'))) {
718                               $db_errno = DB_ERROR_NOSUCHTABLE;
719                           }
720                       break;
721                       default:
722                           $db_errno = $this->errorCode($native_errno);
723                   }
724               } else {
725                   $error_regexps = array(
726                       '/[tT]able .* already exists/' => DB_ERROR_ALREADY_EXISTS,
727                       '/violation of FOREIGN KEY constraint/' => DB_ERROR_CONSTRAINT,
728                       '/conversion error from string/' => DB_ERROR_INVALID_NUMBER,
729                       '/arithmetic exception, numeric overflow, or string truncation/' => DB_ERROR_DIVZERO
730                   );
731                   foreach ($error_regexps as $regexp => $code) {
732                       if (preg_match($regexp, $native_errmsg, $m)) {
733                           $db_errno = $code;
734                           $native_errno = null;
735                           break;
736                       }
737                   }
738               }
739           }
740           return $this->raiseError($db_errno, null, null, null, $native_errmsg);
741       }
