78        /**
79         * Connect to a database and log in as the specified user.
80         *
81         * @param $dsn the data source name (see DB::parseDSN for syntax)
82         * @param $persistent (optional) whether the connection should
83         *         be persistent
84         * @access public
85         * @return int DB_OK on success, a DB error on failure
86         */
87        function connect($dsninfo, $persistent = false)
88        {
89            if (!DB::assertExtension('sybase') && !DB::assertExtension('sybase_ct'))
90                return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
91    
92            $this->dsn = $dsninfo;
93            $user = $dsninfo['username'];
94            $pw   = $dsninfo['password'];
95    
96            $interface = $dsninfo['hostspec'] ? $dsninfo['hostspec'] : 'localhost';
97            $connect_function = $persistent ? 'sybase_pconnect' : 'sybase_connect';
98    
99            if ($interface && $user && $pw) {
100               $conn = $connect_function($interface, $user, $pw);
101           } else {
102               $conn = FALSE;
103           }
104   
105           if (!$conn) {
106               return $this->raiseError(DB_ERROR_CONNECT_FAILED);
107           }
108   
109           $this->connection = $conn;
110           return DB_OK;
111       }

146       function errorCode($errormsg = '')
147       {
148           static $error_regexps;
149           if (empty($error_regexps)) {
150               $error_regexps = array(
151                   '/Incorrect syntax near [\"\'].+[\"\']\./'
152                       => 102,
153                   '/^Unclosed quote before the character string [\"\'].*[\"\']\./'
154                       => 105,
155                   '/Incorrect syntax near the keyword [\"\'].+[\"\']\./'
156                       => 156,
157                   '/Implicit conversion from datatype [\"\'].+[\"\'] to [\"\'].+[\"\'] is not allowed\./'
158                       => 257,
159                   '/Cannot drop the table [\"\'].+[\"\'], because it doesn\'t exist in the system catalogs\./'
160                       => 3701,
161                   '/Only the owner of object [\"\'].+[\"\'] or a user with System Administrator \(SA\) role can run this command\./'
162                       => 10304,
163                   '/^.+ permission denied on object .+, database .+, owner .+/'
164                       => 10330,
165                   '/^.* permission denied, database .+, owner .+/'
166                       => 10331,
167                   '/[^.*] not found\./'
168                       => 208
169               );
170           }
171           while (list($regexp, $code) = each($error_regexps)) {
172               if (preg_match($regexp, $errormsg)) {
173                   $error['code'] = $code;
174               }
175           }
176           $error['userinfo'] = $errormsg;
177           if (isset($error['code']) && isset($this->errorcode_map[$error['code']])) {
178               $error['message'] = $this->errorcode_map[$error['code']];
179           } else {
180               return DB_ERROR;
181           }
182           return $error;
183       }

189       {
190           $native = $this->errorNative();
191           $error = $this->errorCode($native);

197       /**
198        * Send a query to Sybase and return the results as a Sybase resource
199        * identifier.
200        *
201        * @param the SQL query
202        *
203        * @access public
204        *
205        * @return mixed returns a valid Sybase result for successful SELECT
206        *  queries, DB_OK for other successful queries.  A DB error is
207        *  returned on failure.
208        */
209       function simpleQuery($query)
210       {
211           $this->last_query = $query;
212           $query = $this->modifyQuery($query);
213           $result = @sybase_query($query, $this->connection);
214           if (!$result) {
215               return $this->sybaseRaiseError();
216           }
217           if (is_resource($result)) {
218               $numrows = $this->numRows($result);
219               if (is_object($numrows)) {
220                   return $numrows;
221               }
222               $this->num_rows[$result] = $numrows;
223               return $result;
224           }
225           // Determine which queries that should return data, and which
226           // should return an error code only.
227           return DB::isManip($query) ? DB_OK : $result;
228       }

284       /**
285        * Free the internal resources associated with $result.
286        *
287        * @param $result Sybase result identifier or DB statement identifier
288        *
289        * @access public
290        *
291        * @return bool TRUE on success, FALSE if $result is invalid
292        */
293       function freeResult($result)
294       {
295           if (is_resource($result)) {
296               return @sybase_free_result($result);
297           }
298           if (!isset($this->prepare_tokens[(int)$result])) {
299               return false;
300           }
301           unset($this->prepare_tokens[(int)$result]);
302           unset($this->prepare_types[(int)$result]);
303           return true;
304       }

357       function affectedRows()
358       {
359           if (DB::isManip($this->last_query)) {
360               $result = @sybase_affected_rows($this->connection);
361           } else {
362               $result = 0;
363           }
