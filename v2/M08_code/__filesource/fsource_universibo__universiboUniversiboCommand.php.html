126   	 * Inizializza le informazioni utente dell' UniversiboCommand
127   	 *
128   	 * @private
129   	 */
130       function _setUpUserUniversibo()
131       {
132           
133           if (! $this->sessionUserExists() )
134           {
135               $this->sessionUser = new User(0, USER_OSPITE);
136               $this->setSessionIdUtente(0);
137           }
138           elseif ( $this->getSessionIdUtente() >= 0 )
139           {
140               $this->sessionUser =& User::selectUser( $this->getSessionIdUtente() );
141   //			echo $this->sessionUser->getUsername();
142           }
143           else 
144               Error::throw(_ERROR_CRITICAL,array('msg'=>'id_utente registrato nella sessione non valido','file'=>__FILE__,'line'=>__LINE__));
145               
146       }
