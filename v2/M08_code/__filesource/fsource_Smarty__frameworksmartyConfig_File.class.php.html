74        /**
75    	 * Constructs a new config file class.
76    	 *
77    	 * @param string $config_path (optional) path to the config files
78    	 */
79        function Config_File($config_path = NULL)
80        {
81            if (substr(PHP_OS, 0, 3) == "WIN" || substr(PHP_OS, 0, 4) == "OS/2")
82                $this->_separator = "\\";
83            else
84                $this->_separator = "/";
85    
86            if (isset($config_path))
87                $this->set_path($config_path);
88        }

91        /**
92    	 * Set the path where configuration files can be found.
93    	 *
94    	 * @param string $config_path path to the config files
95    	 */
96        function set_path($config_path)
97        {
98            if (!empty($config_path)) {
99                if (!is_string($config_path) || !file_exists($config_path) || !is_dir($config_path)) {
100                   $this->_trigger_error_msg("Bad config file path '$config_path'");
101                   return;
102               }
103   
104               $this->_config_path = $config_path . $this->_separator;
105           }
106       }

109       /**
110   	 * Retrieves config info based on the file, section, and variable name.
111   	 *
112   	 * @param string $file_name config file to get info for
113   	 * @param string $section_name (optional) section to get info for
114   	 * @param string $var_name (optional) variable to get info for
115   	 * @return string|arraya value or array of values
116   	 */
117       function &get($file_name, $section_name = NULL, $var_name = NULL)
118       {
119           if (empty($file_name)) {
120               $this->_trigger_error_msg('Empty config file name');
121               return;
122           } else {
123               $file_name = $this->_config_path . $file_name;
124               if (!isset($this->_config_data[$file_name]))
125                   $this->load_file($file_name, false);
126           }
127           
128           if (!empty($var_name)) {
129               if (empty($section_name)) {
130                   return $this->_config_data[$file_name]["vars"][$var_name];
131               } else {
132                   if(isset($this->_config_data[$file_name]["sections"][$section_name]["vars"][$var_name]))
133                       return $this->_config_data[$file_name]["sections"][$section_name]["vars"][$var_name];
134                   else
135                       return array();
136               }
137           } else {
138               if (empty($section_name)) {
139                   return (array)$this->_config_data[$file_name]["vars"];
140               } else {
141                   if(isset($this->_config_data[$file_name]["sections"][$section_name]["vars"]))
142                       return (array)$this->_config_data[$file_name]["sections"][$section_name]["vars"];
143                   else
144                       return array();
145               }
146           }
147       }

231       /**
232   	 * Load a configuration file manually.
233   	 *
234   	 * @param string $file_name file name to load
235   	 * @param boolean $prepend_path whether current config path should be
236        *                               prepended to the filename
237   	 */
238       function load_file($file_name, $prepend_path = true)
239       {
240           if ($prepend_path && $this->_config_path != "")
241               $config_file = $this->_config_path . $file_name;
242           else
243               $config_file = $file_name;
244   
245           ini_set('track_errors', true);
246           $fp = @fopen($config_file, "r");
247           if (!is_resource($fp)) {
248               $this->_trigger_error_msg("Could not open config file '$config_file'");
249               return false;
250           }
251   
252           $contents = fread($fp, filesize($config_file));
253           fclose($fp);
254           
255           if($this->fix_newlines) {
256               // fix mac/dos formatted newlines
257               $contents = preg_replace('!\r\n?!',"\n",$contents);
258           }
259   
260           $config_data = array();
261   
262           /* Get global variables first. */
263           if (preg_match("/^(.*?)(\n\[|\Z)/s", $contents, $match))
264               $config_data["vars"] = $this->_parse_config_block($match[1]);
265           
266           /* Get section variables. */
267           $config_data["sections"] = array();
268           preg_match_all("/^\[(.*?)\]/m", $contents, $match);
269           foreach ($match[1] as $section) {
270               if ($section{0} == '.' && !$this->read_hidden)
271                   continue;
272               if (preg_match("/\[".preg_quote($section)."\](.*?)(\n\[|\Z)/s", $contents, $match))
273                   if ($section{0} == '.')
274                       $section = substr($section, 1);
275                   $config_data["sections"][$section]["vars"] = $this->_parse_config_block($match[1]);
276           }
277   
278           $this->_config_data[$config_file] = $config_data;
279           
280           return true;
281       }

284       /**
285        * @var string $config_block 
286        */
287       function _parse_config_block($config_block)
288       {
289           $vars = array();
290   
291           /* First we grab the multi-line values. */
292           if (preg_match_all("/^([^=\n]+)=\s*\"{3}(.*?)\"{3}\s*$/ms", $config_block, $match, PREG_SET_ORDER)) {
293               for ($i = 0; $i < count($match); $i++) {
294                   $this->_set_config_var($vars, trim($match[$i][1]), $match[$i][2], false);
295               }
296               $config_block = preg_replace("/^[^=\n]+=\s*\"{3}.*?\"{3}\s*$/ms", "", $config_block);
297           }
298           
299           
300           $config_lines = preg_split("/\n+/", $config_block);
301   
302           foreach ($config_lines as $line) {
303               if (preg_match("/^\s*(\.?\w+)\s*=(.*)/", $line, $match)) {
304                   $var_value = preg_replace('/^([\'"])(.*)\1$/', '\2', trim($match[2]));
305                   $this->_set_config_var($vars, trim($match[1]), $var_value, $this->booleanize);
306               }
307           }
308   
309           return $vars;
310       }

312       /**
313        * @param array &$container 
314        * @param string $var_name 
315        * @param mixed $var_value 
316        * @param boolean $booleanize determines whether $var_value is converted to
317        *                             to true/false
318        */
319       function _set_config_var(&$container, $var_name, $var_value, $booleanize)
320       {
321           if ($var_name{0} == '.') {
322               if (!$this->read_hidden)
323                   return;
324               else
325                   $var_name = substr($var_name, 1);
326           }
327   
328           if (!preg_match("/^[a-zA-Z_]\w*$/", $var_name)) {
329               $this->_trigger_error_msg("Bad variable name '$var_name'");
330               return;
331           }
332   
333           if ($booleanize) {
334               if (preg_match("/^(on|true|yes)$/i", $var_value))
335                   $var_value = true;
336               else if (preg_match("/^(off|false|no)$/i", $var_value))
337                   $var_value = false;
338           }
339                   
340           if (!isset($container[$var_name]) || $this->overwrite)
341               $container[$var_name] = $var_value;
342           else {
343               settype($container[$var_name], 'array');
344               $container[$var_name][] = $var_value;
345           }
346       }
