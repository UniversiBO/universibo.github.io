385       /**
386       * Prepares a query for multiple execution with execute().
387       * With some database backends, this is emulated.
388       * prepare() requires a generic query as string like
389       * "INSERT INTO numbers VALUES(?,?,?)". The ? are wildcards.
390       * Types of wildcards:
391       *   ? - a quoted scalar value, i.e. strings, integers
392       *   & - requires a file name, the content of the file
393       *       insert into the query (i.e. saving binary data
394       *       in a db)
395       *   ! - value is inserted 'as is'
396       *
397       * @param string the query to prepare
398       *
399       * @return resource handle for the query
400       *
401       * @access public
402       * @see execute
403       */
404   
405       function prepare($query)
406       {
407           $tokens = split("[\&\?\!]", $query);
408           $token = 0;
409           $types = array();
410           $qlen = strlen($query);
411           for ($i = 0; $i < $qlen; $i++) {
412               switch ($query[$i]) {
413                   case '?':
414                       $types[$token++] = DB_PARAM_SCALAR;
415                       break;
416                   case '&':
417                       $types[$token++] = DB_PARAM_OPAQUE;
418                       break;
419                   case '!':
420                       $types[$token++] = DB_PARAM_MISC;
421                       break;
422               }
423           }
424   
425           $this->prepare_tokens[] = &$tokens;
426           end($this->prepare_tokens);
427   
428           $k = key($this->prepare_tokens);
429           $this->prepare_types[$k] = $types;
430           $this->prepared_queries[$k] = &$query;
431   
432           return $k;
433       }

482       /**
483       * Make automaticaly an sql query for prepare()
484       *
485       * Example : buildManipSQL('table_sql', array('field1', 'field2', 'field3'), DB_AUTOQUERY_INSERT)
486       *           will return the string : INSERT INTO table_sql (field1,field2,field3) VALUES (?,?,?)
487       * NB : - This belongs more to a SQL Builder class, but this is a simple facility
488       *      - Be carefull ! If you don't give a $where param with an UPDATE query, all
489       *        the records of the table will be updated !
490       *
491       * @param string $table name of the table
492       * @param array $table_fields ordered array containing the fields names
493       * @param int $mode type of query to make (DB_AUTOQUERY_INSERT or DB_AUTOQUERY_UPDATE)
494       * @param string $where in case of update queries, this string will be put after the sql WHERE statement
495       * @return string sql query for prepare()
496       * @access public
497       */
498       function buildManipSQL($table, $table_fields, $mode, $where = false)
499       {
500           if (count($table_fields) == 0) {
501               $this->raiseError(DB_ERROR_NEED_MORE_DATA);
502           }
503           $first = true;
504           switch ($mode) {
505               case DB_AUTOQUERY_INSERT:
506                   $values = '';
507                   $names = '';
508                   while (list(, $value) = each($table_fields)) {
509                       if ($first) {
510                           $first = false;
511                       } else {
512                           $names .= ',';
513                           $values .= ',';
514                       }
515                       $names .= $value;
516                       $values .= '?';
517                   }
518                   return "INSERT INTO $table ($names) VALUES ($values)";
519                   break;
520               case DB_AUTOQUERY_UPDATE:
521                   $set = '';
522                   while (list(, $value) = each($table_fields)) {
523                       if ($first) {
524                           $first = false;
525                       } else {
526                           $set .= ',';
527                       }
528                       $set .= "$value = ?";
529                   }
530                   $sql = "UPDATE $table SET $set";
531                   if ($where) {
532                       $sql .= " WHERE $where";
533                   }
534                   return $sql;
535                   break;
536               default:
537                   $this->raiseError(DB_ERROR_SYNTAX);
538           }
539       }

560       {

564           }

567           if (DB::isError($result) || $result === DB_OK) {
568               return $result;
569           } else {
570               return new DB_result($this, $result);
571           }

575       // {{{ executeEmulateQuery()
576   
577       /**
578       * Emulates the execute statement, when not supported
579       *
580       * @param resource $stmt query handle from prepare()
581       * @param array    $data numeric array containing the
582       *                        data to insert into the query
583       *
584       * @return mixed a string containing the real query run when emulating
585       *  prepare/execute.  A DB error code is returned on failure.
586       *
587       * @access private
588       * @see execute()
589       */
590   
591       function executeEmulateQuery($stmt, $data = false)
592       {
593           $p = &$this->prepare_tokens;
594   
595           if (!isset($this->prepare_tokens[$stmt]) ||
596               !is_array($this->prepare_tokens[$stmt]) ||
597               !sizeof($this->prepare_tokens[$stmt]))
598           {
599               return $this->raiseError(DB_ERROR_INVALID);
600           }
601   
602           $qq = &$this->prepare_tokens[$stmt];
603           $qp = sizeof($qq) - 1;
604   
605           if ((!$data && $qp > 0) ||
606               (!is_array($data) && $qp > 1) ||
607               (is_array($data) && $qp > sizeof($data)))
608           {
609               $this->last_query = $this->prepared_queries[$stmt];
610               return $this->raiseError(DB_ERROR_NEED_MORE_DATA);
611           }
612   
613           $realquery = $qq[0];
614           for ($i = 0; $i < $qp; $i++) {
615               $type = $this->prepare_types[$stmt][$i];
616               if ($type == DB_PARAM_OPAQUE) {
617                   if (is_array($data)) {
618                       $fp = fopen($data[$i], 'r');
619                   } else {
620                       $fp = fopen($data, 'r');
621                   }
622   
623                   $pdata = '';
624   
625                   if ($fp) {
626                       while (($buf = fread($fp, 4096)) != false) {
627                           $pdata .= $buf;
628                       }
629                       fclose($fp);
630                   }
631               } else {
632                   if (is_array($data)) {
633                       $pdata = &$data[$i];
634                   } else {
635                       $pdata = &$data;
636                   }
637               }
638   
639               $realquery .= ($type != DB_PARAM_MISC) ? $this->quote($pdata) : $pdata;
640               $realquery .= $qq[$i + 1];
641           }
642   
643           return $realquery;
644       }

672                   return $res;
673               }
674           }
675           return DB_OK;

739       /**
740        * Send a query to the database and return any results with a
741        * DB_result object.
742        *
743        * @access public
744        *
745        * @param string $query  the SQL query or the statement to prepare
746        * @param string $params the data to be added to the query
747        * @return mixed a DB_result object or DB_OK on success, a DB
748        *                 error on failure
749        *
750        * @see DB::isError
751        * @see DB_common::prepare
752        * @see DB_common::execute
753        */
754       function &query($query, $params = array()) {
755           if (sizeof($params) > 0) {
756               $sth = $this->prepare($query);
757               if (DB::isError($sth)) {
758                   return $sth;
759               }
760               $ret = $this->execute($sth, $params);
761               $this->freePrepared($sth);
762               return $ret;
763           } else {
764               $result = $this->simpleQuery($query);
765               if (DB::isError($result) || $result === DB_OK) {
766                   return $result;
767               } else {
768                   return new DB_result($this, $result);
769               }
770           }
771       }

962       // {{{ getAssoc()
963   
964       /**
965        * Fetch the entire result set of a query and return it as an
966        * associative array using the first column as the key.
967        *
968        * If the result set contains more than two columns, the value
969        * will be an array of the values from column 2-n.  If the result
970        * set contains only two columns, the returned value will be a
971        * scalar with the value of the second column (unless forced to an
972        * array with the $force_array parameter).  A DB error code is
973        * returned on errors.  If the result set contains fewer than two
974        * columns, a DB_ERROR_TRUNCATED error is returned.
975        *
976        * For example, if the table "mytable" contains:
977        *
978        *  ID      TEXT       DATE
979        * --------------------------------
980        *  1       'one'      944679408
981        *  2       'two'      944679408
982        *  3       'three'    944679408
983        *
984        * Then the call getAssoc('SELECT id,text FROM mytable') returns:
985        *   array(
986        *     '1' => 'one',
987        *     '2' => 'two',
988        *     '3' => 'three',
989        *   )
990        *
991        * ...while the call getAssoc('SELECT id,text,date FROM mytable') returns:
992        *   array(
993        *     '1' => array('one', '944679408'),
994        *     '2' => array('two', '944679408'),
995        *     '3' => array('three', '944679408')
996        *   )
997        *
998        * If the more than one row occurs with the same value in the
999        * first column, the last row overwrites all previous ones by
1000       * default.  Use the $group parameter if you don't want to
1001       * overwrite like this.  Example:
1002       *
1003       * getAssoc('SELECT category,id,name FROM mytable', false, null,
1004       *          DB_FETCHMODE_ASSOC, true) returns:
1005       *   array(
1006       *     '1' => array(array('id' => '4', 'name' => 'number four'),
1007       *                  array('id' => '6', 'name' => 'number six')
1008       *            ),
1009       *     '9' => array(array('id' => '4', 'name' => 'number four'),
1010       *                  array('id' => '6', 'name' => 'number six')
1011       *            )
1012       *   )
1013       *
1014       * Keep in mind that database functions in PHP usually return string
1015       * values for results regardless of the database's internal type.
1016       *
1017       * @param string $query the SQL query
1018       *
1019       * @param boolean $force_array used only when the query returns
1020       *  exactly two columns.  If true, the values of the returned array
1021       *  will be one-element arrays instead of scalars.
1022       *
1023       * @param array $params array if supplied, prepare/execute will be used
1024       *         with this array as execute parameters
1025       *
1026       * @param boolean $group if true, the values of the returned array
1027       *                        is wrapped in another array.  If the same
1028       *                        key value (in the first column) repeats
1029       *                        itself, the values will be appended to
1030       *                        this array instead of overwriting the
1031       *                        existing values.
1032       *
1033       * @access public
1034       *
1035       * @return array associative array with results from the query.
1036       */
1037  
1038      function &getAssoc($query, $force_array = false, $params = array(),
1039                         $fetchmode = DB_FETCHMODE_ORDERED, $group = false)
1040      {
1041          settype($params, "array");
1042          if (sizeof($params) > 0) {
1043              $sth = $this->prepare($query);
1044  
1045              if (DB::isError($sth)) {
1046                  return $sth;
1047              }
1048  
1049              $res = $this->execute($sth, $params);
1050              $this->freePrepared($sth);
1051          } else {
1052              $res = $this->query($query);
1053          }
1054  
1055          if (DB::isError($res)) {
1056              return $res;
1057          }
1058  
1059          $cols = $res->numCols();
1060  
1061          if ($cols < 2) {
1062              return $this->raiseError(DB_ERROR_TRUNCATED);
1063          }
1064  
1065          $results = array();
1066  
1067          if ($cols > 2 || $force_array) {
1068              // return array values
1069              // XXX this part can be optimized
1070              if ($fetchmode == DB_FETCHMODE_ASSOC) {
1071                  while (is_array($row = $res->fetchRow(DB_FETCHMODE_ASSOC))) {
1072                      reset($row);
1073                      $key = current($row);
1074                      unset($row[key($row)]);
1075                      if ($group) {
1076                          $results[$key][] = $row;
1077                      } else {
1078                          $results[$key] = $row;
1079                      }
1080                  }
1081              } else {
1082                  while (is_array($row = $res->fetchRow(DB_FETCHMODE_ORDERED))) {
1083                      // we shift away the first element to get
1084                      // indices running from 0 again
1085                      $key = array_shift($row);
1086                      if ($group) {
1087                          $results[$key][] = $row;
1088                      } else {
1089                          $results[$key] = $row;
1090                      }
1091                  }
1092              }
1093              if (DB::isError($row)) {
1094                  $results = $row;
1095              }
1096          } else {
1097              // return scalar values
1098              while (is_array($row = $res->fetchRow(DB_FETCHMODE_ORDERED))) {
1099                  if ($group) {
1100                      $results[$row[0]][] = $row[1];
1101                  } else {
1102                      $results[$row[0]] = $row[1];
1103                  }
1104              }
1105              if (DB::isError($row)) {
1106                  $results = $row;
1107              }
1108          }
1109  
1110          $res->free();
1111  
1112          return $results;
1113      }
