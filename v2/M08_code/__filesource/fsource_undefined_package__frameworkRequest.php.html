41        function _createVariable(&$key,&$value){
42            if(is_array($value)){
43                $this->_createArrayVariable($key,$value);
44            }
45            else{            
46                $this->_createNonArrayVariable($key,$value);
47            }

53        function _createArrayVariable(&$key,&$values){
54            while (list($arrayKey,$arrayValue)=each($values)){
55                $values[$arrayKey]=$this->_getProcessedString($arrayValue);
56            }
57            $this->{$key}=$values;
58            array_push($this->param,$key);
59        }
