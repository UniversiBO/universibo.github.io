48                $hours[$i] = sprintf('%02d', $hours[$i]);

58            if (null !== $all_extra){

61            $html_result .= '>'."\n";

63                                                               'values'          => $hours,
64                                                               'selected'      => strftime($hour_fmt, $time),

73                $minutes[] = sprintf('%02d', $all_minutes[$i]);
