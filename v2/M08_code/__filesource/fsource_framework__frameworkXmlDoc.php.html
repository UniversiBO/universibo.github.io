35        function parse($xmlFile){
36            $parser=xml_parser_create();
37            xml_set_object($parser,$this);
38            xml_parser_set_option($parser, XML_OPTION_CASE_FOLDING, false);
39            xml_set_element_handler($parser, "_startElement", "_endElement");
40            xml_set_character_data_handler($parser, "_characterData");
41    
42            if (!($fp = fopen($xmlFile, "r"))) {
43                $this->error="Fatal: could not open XML input";
44                return;
45            }
46            $xmlFileSize=filesize($xmlFile); 
47            $data = fread($fp, $xmlFileSize+1); 
48            if (!xml_parse($parser, $data, TRUE)){ 
49                 $errorString=xml_error_string(xml_get_error_code($parser)); 
50                 $lineNumber=xml_get_current_line_number($parser); 
51                 $this->error="Fatal: XML error: $errorString at line $lineNumber"; 
52                 return; 
53            }
54            xml_parser_free($parser);
55        }
