71        /**
72         * Connect to a database and log in as the specified user.
73         *
74         * @param $dsn the data source name (see DB::parseDSN for syntax)
75         * @param $persistent (optional) whether the connection should
76         *         be persistent
77         *
78         * @return int DB_OK on success, a DB error code on failure
79         */
80        function connect($dsninfo, $persistent = false)
81        {
82            if (!DB::assertExtension('pgsql'))
83                return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
84    
85            $this->dsn = $dsninfo;
86            $protocol = (isset($dsninfo['protocol'])) ? $dsninfo['protocol'] : 'tcp';
87            $connstr = '';
88    
89            if ($protocol == 'tcp') {
90                if (!empty($dsninfo['hostspec'])) {
91                    $connstr = 'host=' . $dsninfo['hostspec'];
92                }
93                if (!empty($dsninfo['port'])) {
94                    $connstr .= ' port=' . $dsninfo['port'];
95                }
96            }
97    
98            if (isset($dsninfo['database'])) {
99                $connstr .= ' dbname=\'' . addslashes($dsninfo['database']) . '\'';
100           }
101           if (!empty($dsninfo['username'])) {
102               $connstr .= ' user=\'' . addslashes($dsninfo['username']) . '\'';
103           }
104           if (!empty($dsninfo['password'])) {
105               $connstr .= ' password=\'' . addslashes($dsninfo['password']) . '\'';
106           }
107           if (!empty($dsninfo['options'])) {
108               $connstr .= ' options=' . $dsninfo['options'];
109           }
110           if (!empty($dsninfo['tty'])) {
111               $connstr .= ' tty=' . $dsninfo['tty'];
112           }
113   
114           $connect_function = $persistent ? 'pg_pconnect' : 'pg_connect';
115           // catch error
116           ob_start();
117           $conn = $connect_function($connstr);
118           $error = ob_get_contents();
119           ob_end_clean();
120           if ($conn == false) {
121               return $this->raiseError(DB_ERROR_CONNECT_FAILED, null,
122                                        null, null, strip_tags($error));
123           }
124           $this->connection = $conn;
125           return DB_OK;
126       }

146       /**
147        * Send a query to PostgreSQL and return the results as a
148        * PostgreSQL resource identifier.
149        *
150        * @param $query the SQL query
151        *
152        * @return int returns a valid PostgreSQL result for successful SELECT
153        *  queries, DB_OK for other successful queries.  A DB error code
154        *  is returned on failure.
155        */
156       function simpleQuery($query)
157       {
158           $ismanip = DB::isManip($query);
159           $this->last_query = $query;
160           $query = $this->modifyQuery($query);
161           if (!$this->autocommit && $ismanip) {
162               if ($this->transaction_opcount == 0) {
163                   $result = @pg_exec($this->connection, "begin;");
164                   if (!$result) {
165                       return $this->pgsqlRaiseError();
166                   }
167               }
168               $this->transaction_opcount++;
169           }
170           $result = @pg_exec($this->connection, $query);
171           if (!$result) {
172               return $this->pgsqlRaiseError();
173           }
174           // Determine which queries that should return data, and which
175           // should return an error code only.
176           if ($ismanip) {
177               $this->affected = @pg_cmdtuples($result);
178               return DB_OK;
179           } elseif (preg_match('/^\s*\(?\s*SELECT\s+/si', $query) &&
180                     !preg_match('/^\s*\(?\s*SELECT\s+INTO\s/si', $query)) {
181               /* PostgreSQL commands:
182                  ABORT, ALTER, BEGIN, CLOSE, CLUSTER, COMMIT, COPY,
183                  CREATE, DECLARE, DELETE, DROP TABLE, EXPLAIN, FETCH,
184                  GRANT, INSERT, LISTEN, LOAD, LOCK, MOVE, NOTIFY, RESET,
185                  REVOKE, ROLLBACK, SELECT, SELECT INTO, SET, SHOW,
186                  UNLISTEN, UPDATE, VACUUM
187               */
188               $this->row[$result] = 0; // reset the row counter.
189               $numrows = $this->numrows($result);
190               if (is_object($numrows)) {
191                   return $numrows;
192               }
193               $this->num_rows[$result] = $numrows;
194               $this->affected = 0;
195               return $result;
196           } else {
197               $this->affected = 0;
198               return DB_OK;
199           }
200       }

222       /**
223        * Map native error codes to DB's portable ones.  Requires that
224        * the DB implementation's constructor fills in the $errorcode_map
225        * property.
226        *
227        * @param $nativecode the native error code, as returned by the backend
228        *  database extension (string or integer)
229        *
230        * @return int a portable DB error code, or FALSE if this DB
231        *  implementation has no mapping for the given error code.
232        */
233   
234       function errorCode($errormsg)
235       {
236           static $error_regexps;
237           if (empty($error_regexps)) {
238               $error_regexps = array(
239                   '/(Table does not exist\.|Relation [\"\'].*[\"\'] does not exist|sequence does not exist|class ".+" not found)$/' => DB_ERROR_NOSUCHTABLE,
240                   '/table [\"\'].*[\"\'] does not exist/' => DB_ERROR_NOSUCHTABLE,
241                   '/Relation [\"\'].*[\"\'] already exists|Cannot insert a duplicate key into (a )?unique index.*/'      => DB_ERROR_ALREADY_EXISTS,
242                   '/divide by zero$/'                     => DB_ERROR_DIVZERO,
243                   '/pg_atoi: error in .*: can\'t parse /' => DB_ERROR_INVALID_NUMBER,
244                   '/ttribute [\"\'].*[\"\'] not found$|Relation [\"\'].*[\"\'] does not have attribute [\"\'].*[\"\']/' => DB_ERROR_NOSUCHFIELD,
245                   '/parser: parse error at or near \"/'   => DB_ERROR_SYNTAX,
246                   '/referential integrity violation/'     => DB_ERROR_CONSTRAINT
247               );
248           }
249           foreach ($error_regexps as $regexp => $code) {
250               if (preg_match($regexp, $errormsg)) {
251                   return $code;
252               }
253           }
254           // Fall back to DB_ERROR if there was no mapping.
255           return DB_ERROR;
256       }

261       /**
262        * Fetch a row and insert the data into an existing array.
263        *
264        * @param $result PostgreSQL result identifier
265        * @param $row (reference) array where data from the row is stored
266        * @param $fetchmode how the array data should be indexed
267        * @param $rownum the row number to fetch
268        *
269        * @return int DB_OK on success, a DB error code on failure
270        */
271       function fetchInto($result, &$row, $fetchmode, $rownum=null)
272       {
273           $rownum = ($rownum !== null) ? $rownum : $this->row[$result];
274           if ($rownum >= $this->num_rows[$result]) {
275               return null;
276           }
277           if ($fetchmode & DB_FETCHMODE_ASSOC) {
278               $row = @pg_fetch_array($result, $rownum, PGSQL_ASSOC);
279           } else {
280               $row = @pg_fetch_row($result, $rownum);
281           }
282           if (!$row) {
283               $err = pg_errormessage($this->connection);
284               if (!$err) {
285                   return null;
286               }
287               return $this->pgsqlRaiseError();
288           }
289           $this->row[$result] = ++$rownum;
290           return DB_OK;
291       }

297        * Free the internal resources associated with $result.
298        *
299        * @param $result int PostgreSQL result identifier or DB statement identifier

303       function freeResult($result)
304       {
305           if (is_resource($result)) {
306               return @pg_freeresult($result);
307           }
308           if (!isset($this->prepare_tokens[(int)$result])) {
309               return false;
310           }
311           unset($this->prepare_tokens[(int)$result]);
312           unset($this->prepare_types[(int)$result]);
313           unset($this->row[(int)$result]);
314           unset($this->num_rows[(int)$result]);
315           $this->affected = 0;
316           return true;
317       }

321       /**
322       * Quote the given string so it can be safely used within string delimiters
323       * in a query.
324       * @param $string mixed Data to be quoted
325       * @return mixed "NULL" string, quoted string or original data
326       */
327       function quote($str = null)
328       {
329           switch (strtolower(gettype($str))) {
330               case 'null':
331                   return 'NULL';
332               case 'integer':
333               case 'double' :
334                   return $str;
335               case 'boolean':
336                   return $str ? 'TRUE' : 'FALSE';
337               case 'string':
338               default:
339                   $str = str_replace("'", "''", $str);
340                   //PostgreSQL treats a backslash as an escape character.
341                   $str = str_replace('\\', '\\\\', $str);
342                   return "'$str'";
343           }
344       }

465       /**
466        * Get the next value in a sequence.
467        *
468        * We are using native PostgreSQL sequences. If a sequence does
469        * not exist, it will be created, unless $ondemand is false.
470        *
471        * @access public
472        * @param string $seq_name the name of the sequence
473        * @param bool $ondemand whether to create the sequence on demand
474        * @return a sequence integer, or a DB error
475        */
476       function nextId($seq_name, $ondemand = true)
477       {
478           $seqname = $this->getSequenceName($seq_name);
479           $repeat = false;
480           do {
481               $this->pushErrorHandling(PEAR_ERROR_RETURN);
482               $result = $this->query("SELECT NEXTVAL('${seqname}')");
483               $this->popErrorHandling();
484               if ($ondemand && DB::isError($result) &&
485                   $result->getCode() == DB_ERROR_NOSUCHTABLE) {
486                   $repeat = true;
487                   $this->pushErrorHandling(PEAR_ERROR_RETURN);
488                   $result = $this->createSequence($seq_name);
489                   $this->popErrorHandling();
490                   if (DB::isError($result)) {
491                       return $this->raiseError($result);
492                   }
493               } else {
494                   $repeat = false;
495               }
496           } while ($repeat);
497           if (DB::isError($result)) {
498               return $this->raiseError($result);
499           }
500           $arr = $result->fetchRow(DB_FETCHMODE_ORDERED);
501           $result->free();
502           return $arr[0];
503       }

564       /**
565        * Flags of a Field
566        *
567        * @param int $resource PostgreSQL result identifier
568        * @param int $num_field the field number
569        *
570        * @return string The flags of the field ("not_null", "default_xx", "primary_key",
571        *                 "unique" and "multiple_key" are supported)
572        * @access private
573        */
574       function _pgFieldFlags($resource, $num_field, $table_name)
575       {
576           $field_name = @pg_fieldname($resource, $num_field);
577   
578           $result = @pg_exec($this->connection, "SELECT f.attnotnull, f.atthasdef
579                                   FROM pg_attribute f, pg_class tab, pg_type typ
580                                   WHERE tab.relname = typ.typname
581                                   AND typ.typrelid = f.attrelid
582                                   AND f.attname = '$field_name'
583                                   AND tab.relname = '$table_name'");
584           if (@pg_numrows($result) > 0) {
585               $row = @pg_fetch_row($result, 0);
586               $flags  = ($row[0] == 't') ? 'not_null ' : '';
587   
588               if ($row[1] == 't') {
589                   $result = @pg_exec($this->connection, "SELECT a.adsrc
590                                       FROM pg_attribute f, pg_class tab, pg_type typ, pg_attrdef a
591                                       WHERE tab.relname = typ.typname AND typ.typrelid = f.attrelid
592                                       AND f.attrelid = a.adrelid AND f.attname = '$field_name'
593                                       AND tab.relname = '$table_name' AND f.attnum = a.adnum");
594                   $row = @pg_fetch_row($result, 0);
595                   $num = str_replace('\'', '', $row[0]);
596   
597                   $flags .= "default_$num ";
598               }
599           }
600           $result = @pg_exec($this->connection, "SELECT i.indisunique, i.indisprimary, i.indkey
601                                   FROM pg_attribute f, pg_class tab, pg_type typ, pg_index i
602                                   WHERE tab.relname = typ.typname
603                                   AND typ.typrelid = f.attrelid
604                                   AND f.attrelid = i.indrelid
605                                   AND f.attname = '$field_name'
606                                   AND tab.relname = '$table_name'");
607           $count = @pg_numrows($result);
608   
609           for ($i = 0; $i < $count ; $i++) {
610               $row = @pg_fetch_row($result, $i);
611               $keys = explode(" ", $row[2]);
612   
613               if (in_array($num_field + 1, $keys)) {
614                   $flags .= ($row[0] == 't') ? 'unique ' : '';
615                   $flags .= ($row[1] == 't') ? 'primary ' : '';
616                   if (count($keys) > 1)
617                       $flags .= 'multiple_key ';
618               }
619           }
620   
621           return trim($flags);
622       }

627       /**
628        * Returns information about a table or a result set
629        *
630        * NOTE: doesn't support table name and flags if called from a db_result
631        *
632        * @param  mixed $resource PostgreSQL result identifier or table name
633        * @param  int $mode A valid tableInfo mode (DB_TABLEINFO_ORDERTABLE or
634        *                    DB_TABLEINFO_ORDER)
635        *
636        * @return array An array with all the information
637        */
638       function tableInfo($result, $mode = null)
639       {
640           $count = 0;
641           $id    = 0;
642           $res   = array();
643   
644           /*
645            * depending on $mode, metadata returns the following values:
646            *
647            * - mode is false (default):
648            * $result[]:
649            *   [0]["table"]  table name
650            *   [0]["name"]   field name
651            *   [0]["type"]   field type
652            *   [0]["len"]    field length
653            *   [0]["flags"]  field flags
654            *
655            * - mode is DB_TABLEINFO_ORDER
656            * $result[]:
657            *   ["num_fields"] number of metadata records
658            *   [0]["table"]  table name
659            *   [0]["name"]   field name
660            *   [0]["type"]   field type
661            *   [0]["len"]    field length
662            *   [0]["flags"]  field flags
663            *   ["order"][field name]  index of field named "field name"
664            *   The last one is used, if you have a field name, but no index.
665            *   Test:  if (isset($result['meta']['myfield'])) { ...
666            *
667            * - mode is DB_TABLEINFO_ORDERTABLE
668            *    the same as above. but additionally
669            *   ["ordertable"][table name][field name] index of field
670            *      named "field name"
671            *
672            *      this is, because if you have fields from different
673            *      tables with the same field name * they override each
674            *      other with DB_TABLEINFO_ORDER
675            *
676            *      you can combine DB_TABLEINFO_ORDER and
677            *      DB_TABLEINFO_ORDERTABLE with DB_TABLEINFO_ORDER |
678            *      DB_TABLEINFO_ORDERTABLE * or with DB_TABLEINFO_FULL
679            */
680   
681           // if $result is a string, then we want information about a
682           // table without a resultset
683   
684           if (is_string($result)) {
685               $id = @pg_exec($this->connection,"SELECT * FROM $result LIMIT 0");
686               if (empty($id)) {
687                   return $this->pgsqlRaiseError();
688               }
689           } else { // else we want information about a resultset
690               $id = $result;
691               if (empty($id)) {
692                   return $this->pgsqlRaiseError();
693               }
694           }
695   
696           $count = @pg_numfields($id);
697   
698           // made this IF due to performance (one if is faster than $count if's)
699           if (empty($mode)) {
700   
701               for ($i=0; $i<$count; $i++) {
702                   $res[$i]['table'] = (is_string($result)) ? $result : '';
703                   $res[$i]['name']  = @pg_fieldname ($id, $i);
704                   $res[$i]['type']  = @pg_fieldtype ($id, $i);
705                   $res[$i]['len']   = @pg_fieldsize ($id, $i);
706                   $res[$i]['flags'] = (is_string($result)) ? $this->_pgFieldflags($id, $i, $result) : '';
707               }
708   
709           } else { // full
710               $res["num_fields"]= $count;
711   
712               for ($i=0; $i<$count; $i++) {
713                   $res[$i]['table'] = (is_string($result)) ? $result : '';
714                   $res[$i]['name']  = @pg_fieldname ($id, $i);
715                   $res[$i]['type']  = @pg_fieldtype ($id, $i);
716                   $res[$i]['len']   = @pg_fieldsize ($id, $i);
717                   $res[$i]['flags'] = (is_string($result)) ? $this->_pgFieldFlags($id, $i, $result) : '';
718                   if ($mode & DB_TABLEINFO_ORDER) {
719                       $res['order'][$res[$i]['name']] = $i;
720                   }
721                   if ($mode & DB_TABLEINFO_ORDERTABLE) {
722                       $res['ordertable'][$res[$i]['table']][$res[$i]['name']] = $i;
723                   }
724               }
725           }
726   
727           // free the result only if we were called on a table
728           if (is_string($result) && is_resource($id)) {
729               @pg_freeresult($id);
730           }
731           return $res;
732       }

737       /**
738       * Returns the query needed to get some backend info
739       * @param string $type What kind of info you want to retrieve
740       * @return string The SQL query string
741       */
742       function getSpecialQuery($type)
743       {
744           switch ($type) {
745               case 'tables': {
746                   $sql = "SELECT c.relname as \"Name\"
747                           FROM pg_class c, pg_user u
748                           WHERE c.relowner = u.usesysid AND c.relkind = 'r'
749                           AND not exists (select 1 from pg_views where viewname = c.relname)
750                           AND c.relname !~ '^pg_'
751                           UNION
752                           SELECT c.relname as \"Name\"
753                           FROM pg_class c
754                           WHERE c.relkind = 'r'
755                           AND not exists (select 1 from pg_views where viewname = c.relname)
756                           AND not exists (select 1 from pg_user where usesysid = c.relowner)
757                           AND c.relname !~ '^pg_'";
758                   break;
759               }
760               case 'views': {
761                   // Table cols: viewname | viewowner | definition
762                   $sql = "SELECT viewname FROM pg_views";
763                   break;
764               }
765               case 'users': {
766                   // cols: usename |usesysid|usecreatedb|usetrace|usesuper|usecatupd|passwd  |valuntil
767                   $sql = 'SELECT usename FROM pg_user';
768                   break;
769               }
770               case 'databases': {
771                   $sql = 'SELECT datname FROM pg_database';
772                   break;
773               }
774               case 'functions': {
775                   $sql = 'SELECT proname FROM pg_proc';
776                   break;
777               }
778               default:
779                   return null;
780           }
781           return $sql;
782       }
