93        /**
94         * Connect to a database and log in as the specified user.
95         *
96         * @param $dsn the data source name (see DB::parseDSN for syntax)
97         * @param $persistent (optional) whether the connection should
98         *         be persistent
99         * @access public
100        * @return int DB_OK on success, a DB error on failure
101        */
102   
103       function connect($dsninfo, $persistent = false)
104       {
105           if (!DB::assertExtension('fbsql'))
106               return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
107   
108           $this->dsn = $dsninfo;
109           $dbhost = $dsninfo['hostspec'] ? $dsninfo['hostspec'] : 'localhost';
110           $user = $dsninfo['username'];
111           $pw = $dsninfo['password'];
112   
113           $connect_function = $persistent ? 'fbsql_pconnect' : 'fbsql_connect';
114   
115           if ($dbhost && $user && $pw) {
116               $conn = @$connect_function($dbhost, $user, $pw);
117           } elseif ($dbhost && $user) {
118               $conn = @$connect_function($dbhost, $user);
119           } elseif ($dbhost) {
120               $conn = @$connect_function($dbhost);
121           } else {
122               $conn = false;
123           }
124           if (empty($conn)) {
125               if (empty($php_errormsg)) {
126                   return $this->raiseError(DB_ERROR_CONNECT_FAILED);
127               } else {
128                   return $this->raiseError(DB_ERROR_CONNECT_FAILED, null, null,
129                                            null, $php_errormsg);
130               }
131           }
132   
133           if ($dsninfo['database']) {
134               if (!fbsql_select_db($dsninfo['database'], $conn)) {
135                   return $this->fbsqlRaiseError();
136               }
137           }
138   
139           $this->connection = $conn;
140           return DB_OK;
141       }

163       /**
164        * Send a query to fbsql and return the results as a fbsql resource
165        * identifier.
166        *
167        * @param the SQL query
168        *
169        * @access public
170        *
171        * @return mixed returns a valid fbsql result for successful SELECT
172        *  queries, DB_OK for other successful queries.  A DB error is
173        *  returned on failure.
174        */
175       function simpleQuery($query)
176       {
177           $this->last_query = $query;
178           $query = $this->modifyQuery($query);
179           $result = @fbsql_query("$query;", $this->connection);
180           if (!$result) {
181               return $this->fbsqlRaiseError();
182           }
183           // Determine which queries that should return data, and which
184           // should return an error code only.
185           if (DB::isManip($query)) {
186               return DB_OK;
187           }
188           $numrows = $this->numrows($result);
189           if (is_object($numrows)) {
190               return $numrows;
191           }
192           $this->num_rows[$result] = $numrows;
193           return $result;
194       }

216       /**
217        * Fetch a row and insert the data into an existing array.
218        *
219        * @param $result fbsql result identifier
220        * @param $arr (reference) array where data from the row is stored
221        * @param $fetchmode how the array data should be indexed
222        * @param   $rownum the row number to fetch
223        * @access public
224        *
225        * @return int DB_OK on success, a DB error on failure
226        */
227       function fetchInto($result, &$arr, $fetchmode, $rownum=null)
228       {
229           if ($rownum !== null) {
230               if (!@fbsql_data_seek($result, $rownum)) {
231                   return null;
232               }
233           }
234           if ($fetchmode & DB_FETCHMODE_ASSOC) {
235               $arr = @fbsql_fetch_array($result, FBSQL_ASSOC);
236           } else {
237               $arr = @fbsql_fetch_row($result);
238           }
239           if (!$arr) {
240               $errno = @fbsql_errno($this->connection);
241               if (!$errno) {
242                   return NULL;
243               }
244               return $this->fbsqlRaiseError($errno);
245           }
246           return DB_OK;
247       }

252       /**
253        * Free the internal resources associated with $result.
254        *
255        * @param $result fbsql result identifier or DB statement identifier
256        *
257        * @access public
258        *
259        * @return bool TRUE on success, FALSE if $result is invalid
260        */
261       function freeResult($result)
262       {
263           if (is_resource($result)) {
264               return fbsql_free_result($result);
265           }
266   
267           if (!isset($this->prepare_tokens[(int)$result])) {
268               return false;
269           }
270   
271           unset($this->prepare_tokens[(int)$result]);
272           unset($this->prepare_types[(int)$result]);
273   
274           return true;
275       }

389       /**
390        * Get the next value in a sequence.  We emulate sequences
391        * for fbsql.  Will create the sequence if it does not exist.
392        *
393        * @access public
394        *
395        * @param $seq_name the name of the sequence
396        *
397        * @param $ondemand whether to create the sequence table on demand
398        *  (default is true)
399        *
400        * @return a sequence integer, or a DB error
401        */
402       function nextId($seq_name, $ondemand = true)
403       {
404           $sqn = preg_replace('/[^a-z0-9_]/i', '_', $seq_name);
405           $repeat = 0;
406           do {
407               $seqname = sprintf($this->getOption("seqname_format"), $sqn);
408               $result = $this->query("INSERT INTO ${seqname} VALUES(NULL)");
409               if ($ondemand && DB::isError($result) &&
410                   $result->getCode() == DB_ERROR_NOSUCHTABLE) {
411                   $repeat = 1;
412                   $result = $this->createSequence($seq_name);
413                   if (DB::isError($result)) {
414                       return $result;
415                   }
416               } else {
417                   $repeat = 0;
418               }
419           } while ($repeat);
420           if (DB::isError($result)) {
421               return $result;
422           }
423           return fbsql_insert_id($this->connection);
424       }

451       function modifyQuery($query)
452       {
453           if ($this->options['optimize'] == 'portability') {
454               // "DELETE FROM table" gives 0 affected rows in fbsql.
455               // This little hack lets you know how many rows were deleted.
456               if (preg_match('/^\s*DELETE\s+FROM\s+(\S+)\s*$/i', $query)) {
457                   $query = preg_replace('/^\s*DELETE\s+FROM\s+(\S+)\s*$/',
458                                         'DELETE FROM \1 WHERE 1=1', $query);
459               }
460           }
461           return $query;
462       }

479       function tableInfo($result, $mode = null) {
480           $count = 0;
481           $id    = 0;
482           $res   = array();
483   
484           /*
485            * depending on $mode, metadata returns the following values:
486            *
487            * - mode is false (default):
488            * $result[]:
489            *   [0]["table"]  table name
490            *   [0]["name"]   field name
491            *   [0]["type"]   field type
492            *   [0]["len"]    field length
493            *   [0]["flags"]  field flags
494            *
495            * - mode is DB_TABLEINFO_ORDER
496            * $result[]:
497            *   ["num_fields"] number of metadata records
498            *   [0]["table"]  table name
499            *   [0]["name"]   field name
500            *   [0]["type"]   field type
501            *   [0]["len"]    field length
502            *   [0]["flags"]  field flags
503            *   ["order"][field name]  index of field named "field name"
504            *   The last one is used, if you have a field name, but no index.
505            *   Test:  if (isset($result['meta']['myfield'])) { ...
506            *
507            * - mode is DB_TABLEINFO_ORDERTABLE
508            *    the same as above. but additionally
509            *   ["ordertable"][table name][field name] index of field
510            *      named "field name"
511            *
512            *      this is, because if you have fields from different
513            *      tables with the same field name * they override each
514            *      other with DB_TABLEINFO_ORDER
515            *
516            *      you can combine DB_TABLEINFO_ORDER and
517            *      DB_TABLEINFO_ORDERTABLE with DB_TABLEINFO_ORDER |
518            *      DB_TABLEINFO_ORDERTABLE * or with DB_TABLEINFO_FULL
519            */
520   
521           // if $result is a string, then we want information about a
522           // table without a resultset
523           if (is_string($result)) {
524               $id = @fbsql_list_fields($this->dsn['database'],
525                                        $result, $this->connection);
526               if (empty($id)) {
527                   return $this->fbsqlRaiseError();
528               }
529           } else { // else we want information about a resultset
530               $id = $result;
531               if (empty($id)) {
532                   return $this->fbsqlRaiseError();
533               }
534           }
535   
536           $count = @fbsql_num_fields($id);
537   
538           // made this IF due to performance (one if is faster than $count if's)
539           if (empty($mode)) {
540               for ($i=0; $i<$count; $i++) {
541                   $res[$i]['table'] = @fbsql_field_table ($id, $i);
542                   $res[$i]['name']  = @fbsql_field_name  ($id, $i);
543                   $res[$i]['type']  = @fbsql_field_type  ($id, $i);
544                   $res[$i]['len']   = @fbsql_field_len   ($id, $i);
545                   $res[$i]['flags'] = @fbsql_field_flags ($id, $i);
546               }
547           } else { // full
548               $res["num_fields"]= $count;
549   
550               for ($i=0; $i<$count; $i++) {
551                   $res[$i]['table'] = @fbsql_field_table ($id, $i);
552                   $res[$i]['name']  = @fbsql_field_name  ($id, $i);
553                   $res[$i]['type']  = @fbsql_field_type  ($id, $i);
554                   $res[$i]['len']   = @fbsql_field_len   ($id, $i);
555                   $res[$i]['flags'] = @fbsql_field_flags ($id, $i);
556                   if ($mode & DB_TABLEINFO_ORDER) {
557                       $res['order'][$res[$i]['name']] = $i;
558                   }
559                   if ($mode & DB_TABLEINFO_ORDERTABLE) {
560                       $res['ordertable'][$res[$i]['table']][$res[$i]['name']] = $i;
561                   }
562               }
563           }
564   
565           // free the result only if we were called on a table
566           if (is_string($result)) {
567               @fbsql_free_result($id);
568           }
569           return $res;
570       }
