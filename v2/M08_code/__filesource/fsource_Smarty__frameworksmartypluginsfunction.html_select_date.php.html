3      * Smarty plugin
4      * -------------------------------------------------------------
5      * Type:     function
6      * Name:     html_select_date
7      * Version:  1.3
8      * Purpose:  Prints the dropdowns for date selection.
9      * Author:   Andrei Zmievski
10     *
11     * ChangeLog: 1.0 initial release
12     *            1.1 added support for +/- N syntax for begin
13     *                and end year values. (Monte)
14     *            1.2 added support for yyyy-mm-dd syntax for
15     *                time value. (Jan Rosier)
16     *            1.3 added support for choosing format for
17     *                month values (Gary Loescher)
18     *            1.3.1 added support for choosing format for
19     *                day values (Marcus Bointon)
20     * -------------------------------------------------------------
21     *
22     * @package Smarty
23     */
24    function smarty_function_html_select_date($params, &$smarty)
25    {
26        require_once $smarty->_get_plugin_filepath('shared','make_timestamp');
27        require_once $smarty->_get_plugin_filepath('function','html_options');
28        /* Default values. */
29        $prefix          = "Date_";
30        $start_year      = strftime("%Y");
31        $end_year        = $start_year;
32        $display_days    = true;
33        $display_months  = true;
34        $display_years   = true;
35        $month_format    = "%B";
36        /* Write months as numbers by default  GL */
37        $month_value_format = "%m";
38        $day_format      = "%02d";
39        /* Write day values using this format MB */
40        $day_value_format = "%d";
41        $year_as_text    = false;
42        /* Display years in reverse order? Ie. 2000,1999,.... */
43        $reverse_years   = false;
44        /* Should the select boxes be part of an array when returned from PHP?
45           e.g. setting it to "birthday", would create "birthday[Day]",
46           "birthday[Month]" & "birthday[Year]". Can be combined with prefix */
47        $field_array     = null;
48        /* <select size>'s of the different <select> tags.
49           If not set, uses default dropdown. */
50        $day_size        = null;
51        $month_size      = null;
52        $year_size       = null;
53        /* Unparsed attributes common to *ALL* the <select>/<input> tags.
54           An example might be in the template: all_extra ='class ="foo"'. */
55        $all_extra       = null;
56        /* Separate attributes for the tags. */
57        $day_extra       = null;
58        $month_extra     = null;
59        $year_extra      = null;
60        /* Order in which to display the fields.
61           "D" -> day, "M" -> month, "Y" -> year. */
62        $field_order      = 'MDY';
63        /* String printed between the different fields. */
64        $field_separator = "\n";
65        $time = time();
66    
67    
68        extract($params);
69    
70          // If $time is not in format yyyy-mm-dd
71          if (!preg_match('/^\d{4}-\d{2}-\d{2}$/', $time)) {
72              // then $time is empty or unix timestamp or mysql timestamp
73              // using smarty_make_timestamp to get an unix timestamp and
74              // strftime to make yyyy-mm-dd
75              $time = strftime('%Y-%m-%d', smarty_make_timestamp($time));
76          }
77          // Now split this in pieces, which later can be used to set the select
78          $time = explode("-", $time);
79      
80          // make syntax "+N" or "-N" work with start_year and end_year
81          if (preg_match('!^(\+|\-)\s*(\d+)$!', $end_year, $match)) {
82              if ($match[1] == '+') {
83                  $end_year = strftime('%Y') + $match[2];
84              } else {
85                  $end_year = strftime('%Y') - $match[2];
86              }
87          }
88          if (preg_match('!^(\+|\-)\s*(\d+)$!', $start_year, $match)) {
89              if ($match[1] == '+') {
90                  $start_year = strftime('%Y') + $match[2];
91              } else {
92                  $start_year = strftime('%Y') - $match[2];
93              }
94          }
95      
96        $field_order = strtoupper($field_order);
97    
98        $html_result = $month_result = $day_result = $year_result = "";
99    
100       if ($display_months) {
101           $month_names = array();
102           $month_values = array();
103   
104           for ($i = 1; $i <= 12; $i++) {
105               $month_names[] = strftime($month_format, mktime(0, 0, 0, $i, 1, 2000));
106               $month_values[] = strftime($month_value_format, mktime(0, 0, 0, $i, 1, 2000));
107           }
108   
109           $month_result .= '<select name=';
110           if (null !== $field_array){
111               $month_result .= '"' . $field_array . '[' . $prefix . 'Month]"';
112           } else {
113               $month_result .= '"' . $prefix . 'Month"';
114           }
115           if (null !== $month_size){
116               $month_result .= ' size="' . $month_size . '"';
117           }
118           if (null !== $month_extra){
119               $month_result .= ' ' . $month_extra;
120           }
121           if (null !== $all_extra){
122               $month_result .= ' ' . $all_extra;
123           }
124           $month_result .= '>'."\n";
125           
126           $month_result .= smarty_function_html_options(array('output'     => $month_names,
127                                                               'values'     => $month_values,
128                                                               'selected'   => $month_values[$time[1]-1],
129                                                               'print_result' => false),
130                                                         $smarty);
131           
132           $month_result .= '</select>';
133       }
134   
135       if ($display_days) {
136           $days = array();
137           for ($i = 1; $i <= 31; $i++) {
138               $days[] = sprintf($day_format, $i);
139               $day_values[] = sprintf($day_value_format, $i);
140           }
141   
142           $day_result .= '<select name=';
143           if (null !== $field_array){
144               $day_result .= '"' . $field_array . '[' . $prefix . 'Day]"';
145           } else {
146               $day_result .= '"' . $prefix . 'Day"';
147           }
148           if (null !== $day_size){
149               $day_result .= ' size="' . $day_size . '"';
150           }
151           if (null !== $all_extra){
152               $day_result .= ' ' . $all_extra;
153           }
154           if (null !== $day_extra){
155               $day_result .= ' ' . $day_extra;
156           }
157           $day_result .= '>'."\n";
158           $day_result .= smarty_function_html_options(array('output'     => $days,
159                                                             'values'     => $day_values,
160                                                             'selected'   => $time[2],
161                                                             'print_result' => false),
162                                                       $smarty);
163           $day_result .= '</select>';
164       }
165   
166       if ($display_years) {
167           if (null !== $field_array){
168               $year_name = $field_array . '[' . $prefix . 'Year]';
169           } else {
170               $year_name = $prefix . 'Year';
171           }
172           if ($year_as_text) {
173               $year_result .= '<input type="text" name="' . $year_name . '" value="' . $time[0] . '" size="4" maxlength="4"';
174               if (null !== $all_extra){
175                   $year_result .= ' ' . $all_extra;
176               }
177               if (null !== $year_extra){
178                   $year_result .= ' ' . $year_extra;
179               }
180               $year_result .= '>';
181           } else {
182               $years = range((int)$start_year, (int)$end_year);
183               if ($reverse_years) {
184                   rsort($years, SORT_NUMERIC);
185               }
186   
187               $year_result .= '<select name="' . $year_name . '"';
188               if (null !== $year_size){
189                   $year_result .= ' size="' . $year_size . '"';
190               }
191               if (null !== $all_extra){
192                   $year_result .= ' ' . $all_extra;
193               }
194               if (null !== $year_extra){
195                   $year_result .= ' ' . $year_extra;
196               }
197               $year_result .= '>'."\n";
198               $year_result .= smarty_function_html_options(array('output' => $years,
199                                                                  'values' => $years,
200                                                                  'selected'   => $time[0],
201                                                                  'print_result' => false),
202                                                            $smarty);
203               $year_result .= '</select>';
204           }
205       }
206   
207       // Loop thru the field_order field
208       for ($i = 0; $i <= 2; $i++){
209         $c = substr($field_order, $i, 1);
210         switch ($c){
211           case 'D':
212               $html_result .= $day_result;
213               break;
214   
215           case 'M':
216               $html_result .= $month_result;
217               break;
218   
219           case 'Y':
220               $html_result .= $year_result;
221               break;
222         }
223         // Add the field seperator
224         if($i != 2) {
225             $html_result .= $field_separator;
226           }
227       }
228   
229       return $html_result;
230   }
