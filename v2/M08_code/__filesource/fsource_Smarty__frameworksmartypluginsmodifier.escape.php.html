13    function smarty_modifier_escape($string, $esc_type = 'html')
14    {
15        switch ($esc_type) {
16            case 'html':
17                return htmlspecialchars($string, ENT_QUOTES);
18    
19            case 'htmlall':
20                return htmlentities($string, ENT_QUOTES);
21    
22            case 'url':
23                return urlencode($string);
24    
25            case 'quotes':
26                // escape unescaped single quotes
27                return preg_replace("%(?<!\\\\)'%", "\\'", $string);
28    
29            case 'hex':
30                // escape every character into hex
31                $return = '';
32                for ($x=0; $x < strlen($string); $x++) {
33                    $return .= '%' . bin2hex($string[$x]);
34                }
35                return $return;
36                
37            case 'hexentity':
38                $return = '';
39                for ($x=0; $x < strlen($string); $x++) {
40                    $return .= '&#x' . bin2hex($string[$x]) . ';';
41                }
42                return $return;
43    
44            case 'javascript':
45                // escape quotes and backslashes and newlines
46                return str_replace(array('\\','\'',"\r","\n"), array("\\\\", "\\'",'\r','\r'), $string);
47    
48            default:
49                return $string;
50        }
51    }
