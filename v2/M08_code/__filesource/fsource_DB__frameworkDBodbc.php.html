97        /**
98         * Connect to a database and log in as the specified user.
99         *
100        * @param $dsn the data source name (see DB::parseDSN for syntax)
101        * @param $persistent (optional) whether the connection should
102        *         be persistent
103        *
104        * @return int DB_OK on success, a DB error code on failure
105        */
106       function connect($dsninfo, $persistent = false)
107       {
108           if (!DB::assertExtension('odbc'))
109               return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
110   
111           $this->dsn = $dsninfo;
112           if (!empty($dsninfo['dbsyntax'])) {
113               $this->dbsyntax = $dsninfo['dbsyntax'];
114           }
115           switch ($this->dbsyntax) {
116               case 'solid':
117                   $this->features = array(
118                       'prepare' => true,
119                       'pconnect' => true,
120                       'transactions' => true
121                   );
122                   $default_dsn = 'localhost';
123                   break;
124               case 'navision':
125                   // the Navision driver doesn't support fetch row by number
126                   $this->features['limit'] = false;
127                   break;
128               default:
129                   break;
130           }
131           $dbhost = $dsninfo['hostspec'] ? $dsninfo['hostspec'] : 'localhost';
132           $user = $dsninfo['username'];
133           $pw = $dsninfo['password'];
134           if ($this->provides('pconnect')) {
135               $connect_function = $persistent ? 'odbc_pconnect' : 'odbc_connect';
136           } else {
137               $connect_function = 'odbc_connect';
138           }
139           $conn = @$connect_function($dbhost, $user, $pw);
140           if (!is_resource($conn)) {
141               return $this->raiseError(DB_ERROR_CONNECT_FAILED, null, null,
142                                            null, $this->errorNative());
143           }
144           $this->connection = $conn;
145           return DB_OK;
146       }

161       /**
162        * Send a query to ODBC and return the results as a ODBC resource
163        * identifier.
164        *
165        * @param $query the SQL query
166        *
167        * @return int returns a valid ODBC result for successful SELECT
168        *  queries, DB_OK for other successful queries.  A DB error code
169        *  is returned on failure.
170        */
171       function simpleQuery($query)
172       {
173           $this->last_query = $query;
174           $query = $this->modifyQuery($query);
175           $result = @odbc_exec($this->connection, $query);
176           if (!$result) {
177               return $this->odbcRaiseError(); // XXX ERRORMSG
178           }
179           // Determine which queries that should return data, and which
180           // should return an error code only.
181           if (DB::isManip($query)) {
182               $this->manip_result = $result; // For affectedRows()
183               return DB_OK;
184           }
185           $this->row[$result] = 0;
186           $this->manip_result = 0;
187           return $result;
188       }

210       function fetchInto($result, &$row, $fetchmode, $rownum=null)
211       {
212           $row = array();
213           if ($rownum !== null) {
214               $rownum++; // ODBC first row is 1
215               if (version_compare(phpversion(), '4.2.0', 'ge')) {
216                   $cols = odbc_fetch_into($result, $row, $rownum);
217               } else {
218                   $cols = odbc_fetch_into($result, $rownum, $row);
219               }
220           } else {
221               $cols = odbc_fetch_into($result, $row);
222           }
223   
224           if (!$cols) {
225               /* XXX FIXME: doesn't work with unixODBC and easysoft
226                             (get corrupted $errno values)
227               if ($errno = odbc_error($this->connection)) {
228                   return $this->RaiseError($errno);
229               }*/
230               return null;
231           }
232           if ($fetchmode !== DB_FETCHMODE_ORDERED) {
233               for ($i = 0; $i < count($row); $i++) {
234                   $colName = odbc_field_name($result, $i+1);
235                   $a[$colName] = $row[$i];
236               }
237               if ($this->options['optimize'] == 'portability') {
238                   $a = array_change_key_case($a, CASE_LOWER);
239               }
240               $row = $a;
241           }
242           return DB_OK;
243       }

249       {
250           if (is_resource($result)) {
251               // Always return true
252               return odbc_free_result($result);
253           }
254           if (!isset($this->prepare_tokens[(int)$result])) {
255               return false;
256           }

258           unset($this->prepare_types[(int)$result]);
259           return true;

315       /**
316       * Quote the given string so it can be safely used within string delimiters
317       * in a query.
318       * @param $string mixed Data to be quoted
319       * @return mixed "NULL" string, quoted string or original data
320       */
321       function quote($str = null)
322       {
323           if (is_numeric($str)) {
324               return $str;
325           }
326           switch (strtolower(gettype($str))) {
327               case 'null':
328                   return 'NULL';
329               case 'boolean':
330                   return $str ? 'TRUE' : 'FALSE';
331               case 'string':
332               default:
333                   $str = str_replace("'", "''", $str);
334                   return "'$str'";
335           }
336       }

361       /**
362        * Get the next value in a sequence.  We emulate sequences
363        * for odbc. Will create the sequence if it does not exist.
364        *
365        * @access public
366        *
367        * @param $seq_name the name of the sequence
368        *
369        * @param $ondemand whether to create the sequence table on demand
370        *  (default is true)
371        *
372        * @return a sequence integer, or a DB error
373        */
374       function nextId($seq_name, $ondemand = true)
375       {
376           $seqname = $this->getSequenceName($seq_name);
377           $repeat = 0;
378           do {
379               $this->pushErrorHandling(PEAR_ERROR_RETURN);
380               $result = $this->query("update ${seqname} set id = id + 1");
381               $this->popErrorHandling();
382               if ($ondemand && DB::isError($result) &&
383                   $result->getCode() == DB_ERROR_NOSUCHTABLE) {
384                   $repeat = 1;
385                   $this->pushErrorHandling(PEAR_ERROR_RETURN);
386                   $result = $this->createSequence($seq_name);
387                   $this->popErrorHandling();
388                   if (DB::isError($result)) {
389                       return $this->raiseError($result);
390                   }
391                   $result = $this->query("insert into ${seqname} (id) values(0)");
392               } else {
393                   $repeat = 0;
394               }
395           } while ($repeat);
396   
397           if (DB::isError($result)) {
398               return $this->raiseError($result);
399           }
400   
401           $result = $this->query("select id from ${seqname}");
402           if (DB::isError($result)) {
403               return $result;
404           }
405   
406           $row = $result->fetchRow(DB_FETCHMODE_ORDERED);
407           if (DB::isError($row || !$row)) {
408               return $row;
409           }
410   
411           return $row[0];
412       }
