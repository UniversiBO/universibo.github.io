3     /**
4      * Smarty plugin
5      * -------------------------------------------------------------
6      * Type:     function
7      * Name:     html_image
8      * Version:  1.0
9      * Date:     Feb 24, 2003
10     * Author:	 Monte Ohrt <monte@ispi.net>
11     * Credits:  Duda <duda@big.hu> - wrote first image function
12     *           in repository, helped with lots of functionality
13     * Purpose:  format HTML tags for the image
14     * Input:    file = file (and path) of image (required)
15     *           border = border width (optional, default 0)
16     *           height = image height (optional, default actual height)
17     *           image =image width (optional, default actual width)
18     *           basedir = base directory for absolute paths, default
19     *                     is environment variable DOCUMENT_ROOT
20     * 
21     * Examples: {image file="images/masthead.gif"}
22     * Output:   <img src="images/masthead.gif" border=0 width=400 height=23>
23     * -------------------------------------------------------------
24     *
25     * @package Smarty
26     */
27    function smarty_function_html_image($params, &$smarty)
28    {    
29        require_once $smarty->_get_plugin_filepath('shared','escape_special_chars');
30    
31        $file = '';
32        $border = 0;
33        $height = '';
34        $width = '';
35        $extra = '';
36        $prefix = '';
37        $suffix = '';
38        $basedir = isset($GLOBALS['HTTP_SERVER_VARS']['DOCUMENT_ROOT'])
39                ? $GLOBALS['HTTP_SERVER_VARS']['DOCUMENT_ROOT'] : '/';
40        if(strstr($GLOBALS['HTTP_SERVER_VARS']['HTTP_USER_AGENT'], 'Mac')) {
41            $dpi_default = 72;
42        } else {
43            $dpi_default = 96;
44        }
45        
46        foreach($params as $_key => $_val) {    
47            switch($_key) {
48                case 'file':
49                    $file = $_val;
50                    break;
51                case 'border':
52                    $border = $_val;
53                    break;
54                case 'height':
55                    $height = $_val;
56                    break;
57                case 'width':
58                    $width = $_val;
59                    break;
60                case 'link':
61                    $prefix = '<a href="' . $_val . '">';
62                    $suffix = '</a>';
63                    break;
64                case 'dpi':
65                    $dpi = $_val;
66                    break;
67                default:
68                    $extra .= ' '.$_key.'="'.smarty_function_escape_special_chars($_val).'"';
69                    break;                    
70            }
71        }
72    
73        if (empty($file)) {
74            $smarty->trigger_error("html_image: missing 'file' parameter", E_USER_NOTICE);
75            return;
76        }
77    
78        if(substr($file,0,1) == DIR_SEP) {
79            $_image_path = $basedir . DIR_SEP . $file;
80        } else {
81            $_image_path = $file;
82        }
83        
84        if(!isset($params['width']) || !isset($params['height'])) {
85            if(!$_image_data = @getimagesize($_image_path)) {
86                if(!file_exists($_image_path)) {
87                    $smarty->trigger_error("html_image: unable to find '$_image_path'", E_USER_NOTICE);        
88                    return;
89                } else if(!is_readable($_image_path)) {
90                    $smarty->trigger_error("html_image: unable to read '$_image_path'", E_USER_NOTICE);        
91                    return;
92                } else {
93                    $smarty->trigger_error("html_image: '$_image_path' is not a valid image file", E_USER_NOTICE);
94                    return;
95                }
96            }
97            if(!$smarty->security && !$smarty->_is_secure('file', $_image_path)) {
98                $smarty->trigger_error("html_image: (secure) '$_image_path' not in secure directory", E_USER_NOTICE);        
99                return;
100           }    
101           
102           if(!isset($params['width'])) {
103               $width = $_image_data[0];
104           }
105           if(!isset($params['height'])) {
106               $height = $_image_data[1];
107           }
108           
109       }
110   
111       if(isset($params['dpi'])) {
112           $_resize = $dpi_default/$params['dpi'];
113           $width = round($width * $_resize);
114           $height = round($height * $_resize);
115       }
116               
117       return $prefix . '<img src="'.$file.'" border="'.$border.'" width="'.$width.'" height="'.$height.'"'.$extra.' />' . $suffix;
118   }
