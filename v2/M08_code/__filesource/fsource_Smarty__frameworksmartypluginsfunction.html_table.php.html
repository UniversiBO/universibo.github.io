11     * Purpose:  make an html table from an array of data
12     * Input:    loop = array to loop through
13     *           cols = number of columns
14     *           table_attr = table attributes
15     *           tr_attr = table row attributes (arrays are cycled)
16     *           td_attr = table cell attributes (arrays are cycled)
17     *           trailpad = value to pad trailing cells with
18     * 
19     * Examples: {table loop=$data}
20     *           {$table loop=$data cols=4 tr_attr='"bgcolor=red"'}
21     *           {$table loop=$data cols=4 tr_attr=$colors}
22     * -------------------------------------------------------------
23     *
24     * @package Smarty
25     */
26    function smarty_function_html_table($params, &$smarty)
27    {
28        $table_attr = 'border="1"';
29        $tr_attr = '';
30        $td_attr = '';
31        $cols = 3;
32        $trailpad = '&nbsp;';
33        
34        extract($params);
35    
36        if (!isset($loop)) {
37            $smarty->trigger_error("html_table: missing 'loop' parameter");
38            return;
39        }
40        
41        $output = "<table $table_attr>\n";
42        $output .= "<tr " . smarty_function_html_table_cycle('tr', $tr_attr) . ">\n";
43    
44        for($x = 0, $y = count($loop); $x < $y; $x++) {
45            $output .= "<td " . smarty_function_html_table_cycle('td', $td_attr) . ">" . $loop[$x] . "</td>\n";        
46            if((!(($x+1) % $cols)) && $x < $y-1) {
47                // go to next row
48                $output .= "</tr>\n<tr " . smarty_function_html_table_cycle('tr', $tr_attr) . ">\n";
49            }
50            if($x == $y-1) {
51                // last row, pad remaining cells
52                $cells = $cols - $y % $cols;
53                if($cells != $cols) {
54                    for($padloop = 0; $padloop < $cells; $padloop++) {
55                        $output .= "<td " . smarty_function_html_table_cycle('td', $td_attr) . ">$trailpad</td>\n";
56                    }
57                }
58                $output .= "</tr>\n";
59            }
60        }
61                
62        $output .= "</table>\n";
63        
64        return $output;
65    }

67    function smarty_function_html_table_cycle($name, $var) {
68        static $names = array();
69        
70        if(!is_array($var)) {
71            return $var;
72        }
73        
74        if(!isset($names[$name]) || $names[$name] == count($var)-1) {
75            $names[$name] = 0;
76            return $var[0];
77        }
78        
79        $names[$name]++;
80        return $var[$names[$name]];
81        
82    }
