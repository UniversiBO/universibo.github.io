6     /**
7     *   this decorator actually just adds the functionality to read the
8     *   test-suite classes from a given directory and instanciate them
9     *   automatically, use it as given in the example below.
10    *
11    *   usage example
12    *   <code>
13    *   $gui = new PHPUnit_GUI_SetupDecorator(new PHPUnit_GUI_HTML());

39        /**
40        *   Setup test suites that can be found in the given directory
41        *   Using the second parameter you can also choose a subsets of the files found
42        *   in the given directory. I.e. only all the files that contain '_UnitTest_',
43        *   in order to do this simply call it like this:
44        *   <code>getSuitesFromDir($dir,'.*_UnitTest_.*')</code>.
45        *   There you can already see that the pattern is built for the use within a regular expression.
46        *   
47        *
48        */
49        function getSuitesFromDir($dir,$filenamePattern='',$exclude=array())
50        {
51            // remove trailing DIRECTORY_SEPERATOR if missing
52            if ($dir{strlen($dir)-1} == DIRECTORY_SEPARATOR) {
53                $dir = substr($dir,0,-1);
54            }
55    
56            $files = $this->_getFiles($dir,$filenamePattern,$exclude,realpath($dir.'/..'));
57            asort($files);
58            foreach ($files as $className=>$aFile) {
59                include_once($aFile);
60                if (class_exists($className)) {
61                    $suites[] =& new PHPUnit_TestSuite($className);
62                } else {
63                    trigger_error("$className could not be found in $dir$aFile!");
64                }
65            }
66            
67            $this->_gui->addSuites($suites);
68        }
69        
70        function _getFiles($dir,$filenamePattern,$exclude,$rootDir)
71        {
72            $files = array();
73            if ($dp=opendir($dir)) {
74                while (false!==($file=readdir($dp))) {
75                    $filename = $dir.DIRECTORY_SEPARATOR.$file;
76                    $match = true;
77                    if ($filenamePattern && !preg_match("~$filenamePattern~",$file)) {
78                        $match = false;
79                    }
80                    if (sizeof($exclude)) {
81                        foreach ($exclude as $aExclude) {
82                            if (strpos($file,$aExclude)!==false) {
83                                $match = false;
84                                break;
85                            }
86                        }
87                    }
88                    if (is_file($filename) && $match) {
89                        $className = str_replace('/','_',substr(str_replace($rootDir,'',$filename),1));
90                        $className = basename($className,'.php');   // remove php-extension
91                        $files[$className] = $filename;
92                    }
93                    if ($file!='.' && $file!='..' && is_dir($filename)) {
94                        $files = array_merge($files,$this->_getFiles($filename,$filenamePattern,$exclude,$rootDir));
95                    }
96                }
97                closedir($dp);
98            }
99            return $files;        
100       }
