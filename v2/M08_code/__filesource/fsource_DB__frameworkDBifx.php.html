79        /**
80         * Connect to a database and log in as the specified user.
81         *
82         * @param $dsn the data source name (see DB::parseDSN for syntax)
83         * @param $persistent (optional) whether the connection should
84         *         be persistent
85         *
86         * @return int DB_OK on success, a DB error code on failure
87         */
88        function connect(&$dsninfo, $persistent = false)
89        {
90            if (!DB::assertExtension('informix') && !DB::assertExtension('Informix')) {
91                return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
92            }
93            $this->dsn = $dsninfo;
94            $dbhost = $dsninfo['hostspec'] ? '@' . $dsninfo['hostspec'] : '';
95            $dbname = $dsninfo['database'] ? $dsninfo['database'] . $dbhost : '';
96            $user = $dsninfo['username'] ? $dsninfo['username'] : '';
97            $pw = $dsninfo['password'] ? $dsninfo['password'] : '';
98    
99            $connect_function = $persistent ? 'ifx_pconnect' : 'ifx_connect';
100   
101           $this->connection = @$connect_function($dbname, $user, $pw);
102           if (!is_resource($this->connection)) {
103               return $this->ifxraiseError(DB_ERROR_CONNECT_FAILED);
104           }
105           return DB_OK;
106       }

124       /**
125        * Send a query to Informix and return the results as a
126        * Informix resource identifier.
127        *
128        * @param $query the SQL query
129        *
130        * @return int returns a valid Informix result for successful SELECT
131        *  queries, DB_OK for other successful queries.  A DB error code
132        *  is returned on failure.
133        */
134       function simpleQuery($query)
135       {
136           $this->last_query = $query;
137           if (preg_match('/(SELECT)/i', $query)) {    //TESTME: Use !DB::isManip()?
138               // the scroll is needed for fetching absolute row numbers
139               // in a select query result
140               $result = @ifx_query($query, $this->connection, IFX_SCROLL);
141           } else {
142               $result = @ifx_query($query, $this->connection);
143           }
144           if (!$result) {
145               return $this->ifxraiseError();
146           }
147           $this->affected = ifx_affected_rows ($result);
148           // Determine which queries that should return data, and which
149           // should return an error code only.
150           if (preg_match('/(SELECT)/i', $query)) {
151               return $result;
152           }
153           // Result has to be freeed even with a insert or update
154           ifx_free_result($result);
155   
156           return DB_OK;
157       }

191       /**
192        * Fetch a row and return as array.
193        *
194        * @param   $result     Informix result identifier
195        * @param   $row         (reference) array where data from the row is stored
196        * @param   $fetchmode      how the resulting array should be indexed
197        * @param   $rownum     the row number to fetch
198        *
199        * @return int an array on success, a DB error code on failure, NULL
200        *              if there is no more data
201        */
202       function fetchInto($result, &$row, $fetchmode, $rownum=null)
203       {
204           if (($rownum !== null) && ($rownum < 0)) {
205               return null;
206           }
207           // if $rownum is null, fetch row will return the next row
208           if (!$row = @ifx_fetch_row($result, $rownum)) {
209               return null;
210           }
211           if ($fetchmode !== DB_FETCHMODE_ASSOC) {
212               $i=0;
213               $order = array();
214               foreach ($row as $key => $val) {
215                   $order[$i++] = $val;
216               }
217               $row = $order;
218           }
219           return DB_OK;
220       }

249       /**
250        * Free the internal resources associated with $result.
251        *
252        * @param $result Informix result identifier
253        *
254        * @return bool TRUE on success, DB_error on error
255        */
256       function freeResult($result)
257       {
258           if (is_resource($result)) {
259               if (!@ifx_free_result($result)) {
260                   return $this->ifxraiseError();
261               }
262               return true;
263           }
264           if (!isset($this->prepare_tokens[(int)$result])) {
265               return false;
266           }
267           unset($this->prepare_tokens[(int)$result]);
268           unset($this->prepare_types[(int)$result]);
269           return true;
270       }

287       /**
288        * Map native error codes to DB's portable ones.  Requires that
289        * the DB implementation's constructor fills in the $errorcode_map
290        * property.
291        *
292        * @return int a portable DB error code, or DB_ERROR if this DB
293        *  implementation has no mapping for the given error code.
294        */
295   
296       function errorCode($nativecode)
297       {
298           if (ereg('SQLCODE=(.*)]', $nativecode, $match)) {
299               $code = $match[1];
300               if (isset($this->errorcode_map[$code])) {
301                   return $this->errorcode_map[$code];
302               }
303           }
304           return DB_ERROR;
305       }
