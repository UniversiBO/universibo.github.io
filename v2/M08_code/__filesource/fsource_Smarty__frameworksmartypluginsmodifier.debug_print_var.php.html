13    function smarty_modifier_debug_print_var($var, $depth = 0, $length = 100)
14    {
15        if (is_array($var)) {
16            $results = "<b>Array (".count($var).")</b>";
17            foreach ($var as $curr_key => $curr_val) {
18                $return = smarty_modifier_debug_print_var($curr_val, $depth+1, $length);
19                $results .= '<br>\r'.str_repeat('&nbsp;', $depth*2)."<b>$curr_key</b> =&gt; $return";
20            }
21            return $results;
22        } else if (is_object($var)) {
23            $object_vars = get_object_vars($var);
24            $results = "<b>".get_class($var)." Object (".count($object_vars).")</b>";
25            foreach ($object_vars as $curr_key => $curr_val) {
26                $return = smarty_modifier_debug_print_var($curr_val, $depth+1, $length);
27                $results .= '<br>\r'.str_repeat('&nbsp;', $depth*2)."<b>$curr_key</b> =&gt; $return";
28            }
29            return $results;
30        } else {
31            if (empty($var) && $var != "0") {
32                return '<i>empty</i>';
33            }
34            if (strlen($var) > $length ) {
35                $results = substr($var, 0, $length-3).'...';
36            } else {
37                $results = $var;
38            }
39            $results = preg_replace("![\r\t\n]!", " ", $results);
40            $results = htmlspecialchars($results);
41            return $results;
42        }
43    }
