12        *   the current implementation of PHPUnit is designed
13        *   this way that adding a suite to another suite only

46        function show()
47        {
48            $showPassed=false;
49            $submitted = @$_REQUEST['submitted'];
50            if ($submitted) {
51                $showPassed = @$_REQUEST['showOK']?true:false;
52            }
53            
54            $suiteResults = array();
55            foreach ($this->_suites as $aSuite) {
56                $aSuiteResult = array();
57                $aSuiteResult['name'] = $aSuite->getName();
58                if ($submitted && isset($_REQUEST[$aSuiteResult['name']])) {
59                    $result = PHPUnit::run($aSuite);
60    
61                    $aSuiteResult['counts']['run'] = $result->runCount();
62                    $aSuiteResult['counts']['error'] = $result->errorCount();
63                    $aSuiteResult['counts']['failure'] = $result->failureCount();
64    
65                    $aSuiteResult['results'] = $this->_prepareResult($result,$showPassed);
66    
67                    $per = 100/$result->runCount();
68                    $failed = ($per*$result->errorCount())+($per*$result->failureCount());
69                    $aSuiteResult['percent'] = round(100-$failed,2);
70                } else {
71                    $aSuiteResult['addInfo'] = 'NOT EXECUTED';
72                }
73                
74                $suiteResults[] = $aSuiteResult;
75            }
76    
77            $final['name'] = 'OVERALL RESULT';
78            $final['counts'] = array();
79            $final['percent'] = 0;
80            $numExecutedTests = 0;
81            foreach ($suiteResults as $aSuiteResult) {
82                if (sizeof(@$aSuiteResult['counts'])) {
83                    foreach ($aSuiteResult['counts'] as $key=>$aCount) {
84                        if (!isset($final['counts'][$key])) {
85                            $final['counts'][$key] = 0;
86                        }
87                        $final['counts'][$key] += $aCount;
88                    }
89                }
90            }
91            if (isset($final['counts']['run'])) {
92                $per = 100/$final['counts']['run'];
93                $failed = ($per*$final['counts']['error'])+($per*$final['counts']['failure']);
94                $final['percent'] = round(100-$failed,2);
95            } else {
96                $final['percent'] = 0;
97            }
98            array_unshift($suiteResults,$final);

103       function _prepareResult($result,$showPassed)
104       {
105           $ret = array();
106           $failures = $result->failures();
107           foreach($failures as $aFailure) {
108               $ret['failures'][] = $this->_prepareFailure($aFailure);
109           }
110           
111           $errors = $result->errors();
112           foreach($errors as $aError) {
113               $ret['errors'][] = $this->_prepareErrors($aError);
114           }
115           
116           if ($showPassed) {
117               $passed = $result->passedTests();
118               foreach($passed as $aPassed) {
119                   $ret['passed'][] = $this->_preparePassedTests($aPassed);
120               }
121           }
122           
123           return $ret;
124       }

126       function _prepareFailure($failure)
127       {
128           $test = $failure->failedTest();            
129           $ret['testName'] = $test->getName();
130   
131           $exception = $failure->thrownException();
132           // if there are () around it then we seem to have a serialized data
133           if (preg_match('/expected\s.*\), actual.*/',$exception)) {
134   // this might be unnecessary since it only works for arrays...        
135               ob_start();
136               print_r(unserialize(preg_replace('/expected\s(.*), actual.*/','$1',$exception)));
137               $ret['expected'] = ob_get_contents();
138               ob_clean();
139               print_r(unserialize(preg_replace('/expected\s.*, actual (.*)/','$1',$exception)));
140               $ret['actual'] = ob_get_contents();
141               ob_end_clean();
142           } else {
143               // spaces might make a diff, so we shall show them properly (since a user agent ignores them)
144               if (preg_match('/expected\s.*, actual.*/',$exception)) {
145                   $ret['expected'] = preg_replace('/expected\s(.*), actual.*/','$1',$exception);
146                   $ret['actual'] = preg_replace('/expected\s.*, actual (.*)/','$1',$exception);
147               } else {
148                   $ret['message'] = str_replace(' ','&nbsp;',$exception);
149               }
150           }
151           
152           return $ret;
153       }
