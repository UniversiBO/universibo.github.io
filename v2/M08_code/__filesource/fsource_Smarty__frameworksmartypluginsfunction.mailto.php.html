4      * Smarty plugin
5      * -------------------------------------------------------------
6      * Type:     function
7      * Name:     mailto
8      * Version:  1.2
9      * Date:     May 21, 2002
10     * Author:	 Monte Ohrt <monte@ispi.net>
11     * Credits:  Jason Sweat (added cc, bcc and subject functionality)
12     * Purpose:  automate mailto address link creation, and optionally
13     *           encode them.
14     * Input:    address = e-mail address
15     *           text = (optional) text to display, default is address
16     *           encode = (optional) can be one of:
17     *                 none : no encoding (default)
18     *                 javascript : encode with javascript
19     *                 hex : encode with hexidecimal (no javascript)
20     *           cc = (optional) address(es) to carbon copy
21     *           bcc = (optional) address(es) to blind carbon copy
22     *           subject = (optional) e-mail subject
23     *           newsgroups = (optional) newsgroup(s) to post to
24     *           followupto = (optional) address(es) to follow up to
25     *           extra = (optional) extra tags for the href link
26     * 
27     * Examples: {mailto address="me@domain.com"}
28     *           {mailto address="me@domain.com" encode="javascript"}
29     *           {mailto address="me@domain.com" encode="hex"}
30     *           {mailto address="me@domain.com" subject="Hello to you!"}
31     *           {mailto address="me@domain.com" cc="you@domain.com,they@domain.com"}
32     *           {mailto address="me@domain.com" extra='class="mailto"'}
33     * -------------------------------------------------------------
34     *
35     * @package Smarty
36     */
37    function smarty_function_mailto($params, &$smarty)
38    {
39        extract($params);
40    
41        if (empty($address)) {
42            $smarty->trigger_error("mailto: missing 'address' parameter");
43            return;
44        }
45        
46        if (empty($text)) {
47            $text = $address;
48        }
49        
50        // netscape and mozilla do not decode %40 (@) in BCC field (bug?)
51        // so, don't encode it.
52        
53        $mail_parms = array();
54        if (!empty($cc)) {
55            $mail_parms[] = 'cc='.str_replace('%40','@',rawurlencode($cc));
56        }
57        
58        if (!empty($bcc)) {
59            $mail_parms[] = 'bcc='.str_replace('%40','@',rawurlencode($bcc));
60        }
61        
62        if (!empty($subject)) {
63            $mail_parms[] = 'subject='.rawurlencode($subject);
64        }
65    
66        if (!empty($newsgroups)) {
67            $mail_parms[] = 'newsgroups='.rawurlencode($newsgroups);
68        }
69    
70        if (!empty($followupto)) {
71            $mail_parms[] = 'followupto='.str_replace('%40','@',rawurlencode($followupto));
72        }
73        
74        for ($i=0; $i<count($mail_parms); $i++) {
75            $mail_parm_vals .= (0==$i) ? '?' : '&';
76            $mail_parm_vals .= $mail_parms[$i];
77        }
78        $address .= $mail_parm_vals;
79        
80        if (empty($encode)) {
81            $encode = 'none';
82        } elseif (!in_array($encode,array('javascript','hex','none')) ) {
83            $smarty->trigger_error("mailto: 'encode' parameter must be none, javascript or hex");
84            return;        
85        }
86        
87        if ($encode == 'javascript' ) {
88            $string = 'document.write(\'<a href="mailto:'.$address.'" '.$extra.'>'.$text.'</a>\');';
89            
90            for ($x=0; $x < strlen($string); $x++) {
91                $js_encode .= '%' . bin2hex($string[$x]);
92            }
93        
94            return '<script type="text/javascript" language="javascript">eval(unescape(\''.$js_encode.'\'))</script>';
95            
96        } elseif ($encode == 'hex') {
97    
98            preg_match('!^(.*)(\?.*)$!',$address,$match);
99            if(!empty($match[2])) {
100               $smarty->trigger_error("mailto: hex encoding does not work with extra attributes. Try javascript.");
101               return;                        
102           }  
103           for ($x=0; $x < strlen($address); $x++) {
104               if(preg_match('!\w!',$address[$x])) {
105                   $address_encode .= '%' . bin2hex($address[$x]);
106               } else {
107                   $address_encode .= $address[$x];                
108               }
109           }
110           for ($x=0; $x < strlen($text); $x++) {
111               $text_encode .= '&#x' . bin2hex($text[$x]).';';
112           }
113           
114           return '<a href="mailto:'.$address_encode.'" '.$extra.'>'.$text_encode.'</a>';
115           
116       } else {
117           // no encoding		
118           return '<a href="mailto:'.$address.'" '.$extra.'>'.$text.'</a>';
119   
120       }
121       
122   }
