74        function connect($dsninfo, $persistent = false)
75        {
76            if (!DB::assertExtension('mssql') && !DB::assertExtension('sybase')
77                && !DB::assertExtension('sybase_ct'))
78            {
79                return $this->raiseError(DB_ERROR_EXTENSION_NOT_FOUND);
80            }
81            $this->dsn = $dsninfo;
82            $user = $dsninfo['username'];
83            $pw = $dsninfo['password'];
84            $dbhost = $dsninfo['hostspec'] ? $dsninfo['hostspec'] : 'localhost';
85            $port   = $dsninfo['port'] ? ':' . $dsninfo['port'] : '';
86            $dbhost .= $port;
87    
88            $connect_function = $persistent ? 'mssql_pconnect' : 'mssql_connect';
89    
90            if ($dbhost && $user && $pw) {
91                $conn = @$connect_function($dbhost, $user, $pw);
92            } elseif ($dbhost && $user) {
93                $conn = @$connect_function($dbhost, $user);
94            } else {
95                $conn = @$connect_function($dbhost);
96            }
97            if (!$conn) {
98                return $this->raiseError(DB_ERROR_CONNECT_FAILED, null, null,
99                                             null, mssql_get_last_message());
100           }
101           if ($dsninfo['database']) {
102               if (!@mssql_select_db($dsninfo['database'], $conn)) {
103                   return $this->raiseError(DB_ERROR_NODBSELECTED, null, null,
104                                            null, mssql_get_last_message());
105               }
106               $this->_db = $dsninfo['database'];
107           }
108           $this->connection = $conn;
109           return DB_OK;
110       }

125       function simpleQuery($query)
126       {
127           $ismanip = DB::isManip($query);
128           $this->last_query = $query;
129           if (!@mssql_select_db($this->_db, $this->connection)) {
130               return $this->mssqlRaiseError(DB_ERROR_NODBSELECTED);
131           }
132           $query = $this->modifyQuery($query);
133           if (!$this->autocommit && $ismanip) {
134               if ($this->transaction_opcount == 0) {
135                   $result = @mssql_query('BEGIN TRAN', $this->connection);
136                   if (!$result) {
137                       return $this->mssqlRaiseError();
138                   }
139               }
140               $this->transaction_opcount++;
141           }
142           $result = @mssql_query($query, $this->connection);
143           if (!$result) {
144               return $this->mssqlRaiseError();
145           }
146           // Determine which queries that should return data, and which
147           // should return an error code only.
148           return $ismanip ? DB_OK : $result;
149       }

172       function fetchInto($result, &$ar, $fetchmode, $rownum=null)
173       {
174           if ($rownum !== null) {
175               if (!@mssql_data_seek($result, $rownum)) {
176                   return null;
177               }
178           }
179           if ($fetchmode & DB_FETCHMODE_ASSOC) {
180               $ar = @mssql_fetch_array($result, MSSQL_ASSOC);
181           } else {
182               $ar = @mssql_fetch_row($result);
183           }
184           if (!$ar) {
185               /* This throws informative error messages,
186                  don't use it for now
187               if ($msg = mssql_get_last_message()) {
188                   return $this->raiseError($msg);
189               }
190               */
191               return null;
192           }
193           return DB_OK;
194       }

199       function freeResult($result)
200       {
201           if (is_resource($result)) {
202               return @mssql_free_result($result);
203           }
204           if (!isset($this->prepare_tokens[$result])) {
205               return false;
206           }
207           unset($this->prepare_tokens[$result]);
208           unset($this->prepare_types[$result]);
209           return true;
210       }

254        * Commit the current transaction.
255        */
256       function commit()
257       {
258           if ($this->transaction_opcount > 0) {
259               if (!@mssql_select_db($this->_db, $this->connection)) {
260                   return $this->mssqlRaiseError(DB_ERROR_NODBSELECTED);
261               }
262               $result = @mssql_query('COMMIT TRAN', $this->connection);
263               $this->transaction_opcount = 0;
264               if (!$result) {
265                   return $this->mssqlRaiseError();
266               }
267           }
268           return DB_OK;
269       }

278       {
279           if ($this->transaction_opcount > 0) {
280               if (!@mssql_select_db($this->_db, $this->connection)) {
281                   return $this->mssqlRaiseError(DB_ERROR_NODBSELECTED);

284               $this->transaction_opcount = 0;

296        * Gets the number of rows affected by the last query.
297        * if the last query was a select, returns 0.
298        *
299        * @return number of rows affected by the last query or DB_ERROR
300        */
301       function affectedRows()
302       {
303           if (DB::isManip($this->last_query)) {
304               $res = @mssql_query('select @@rowcount', $this->connection);
305               if (!$res) {
306                   return $this->mssqlRaiseError();
307               }
308               $ar = @mssql_fetch_row($res);
309               if (!$ar) {
310                   $result = 0;
311               } else {
312                   @mssql_free_result($res);
313                   $result = $ar[0];
314               }
315           } else {
316               $result = 0;
317           }
318           return $result;
319       }

324       /**
325        * Get the next value in a sequence.  We emulate sequences
326        * for MSSQL.  Will create the sequence if it does not exist.
327        *
328        * @access public
329        *
330        * @param $seq_name the name of the sequence
331        *
332        * @param $ondemand whether to create the sequence table on demand
333        *  (default is true)
334        *
335        * @return a sequence integer, or a DB error
336        */
337       function nextId($seq_name, $ondemand = true)
338       {
339           $seqname = $this->getSequenceName($seq_name);
340           if (!@mssql_select_db($this->_db, $this->connection)) {
341               return $this->mssqlRaiseError(DB_ERROR_NODBSELECTED);
342           }
343           $repeat = 0;
344           do {
345               $this->pushErrorHandling(PEAR_ERROR_RETURN);
346               $result = $this->query("INSERT INTO $seqname (vapor) VALUES (0)");
347               $this->popErrorHandling();
348               if ($ondemand && DB::isError($result) &&
349                   ($result->getCode() == DB_ERROR || $result->getCode() == DB_ERROR_NOSUCHTABLE))
350               {
351                   $repeat = 1;
352                   $result = $this->createSequence($seq_name);
353                   if (DB::isError($result)) {
354                       return $this->raiseError($result);
355                   }
356               } elseif (!DB::isError($result)) {
357                   $result = $this->query("SELECT @@IDENTITY FROM $seqname");
358                   $repeat = 0;
359               } else {
360                   $repeat = false;
361               }
362           } while ($repeat);
363           if (DB::isError($result)) {
364               return $this->raiseError($result);
365           }
366           $result = $result->fetchRow(DB_FETCHMODE_ORDERED);
367           return $result[0];
368       }

404       function mssqlRaiseError($code = null)
405       {
406           $native_msg = mssql_get_last_message();
407           $native_code = $this->errorCode();
408           if ($code === null) {
409               if (isset($this->errorcode_map[$native_code])) {
410                   $code = $this->errorcode_map[$native_code];
411               } else {
412                   $code = DB_ERROR;
413               }
414           }
415           return $this->raiseError($code, null, null, null, $native_code . ' - ' . $native_msg);
416       }

421     /**
422        * Returns information about a table or a result set
423        *
424        * NOTE: doesn't support table name and flags if called from a db_result
425        *
426        * @param  mixed $resource SQL Server result identifier or table name
427        * @param  int $mode A valid tableInfo mode (DB_TABLEINFO_ORDERTABLE or
428        *                    DB_TABLEINFO_ORDER)
429        *
430        * @return array An array with all the information
432        */
433   
434       function tableInfo($result, $mode = null)
435       {
436   
437           $count = 0;
438           $id    = 0;
439           $res   = array();
440   
441           /*
442            * depending on $mode, metadata returns the following values:
443            *
444            * - mode is false (default):
445            * $result[]:
446            *   [0]["table"]  table name
447            *   [0]["name"]   field name
448            *   [0]["type"]   field type
449            *   [0]["len"]    field length
450            *   [0]["flags"]  field flags
451            *
452            * - mode is DB_TABLEINFO_ORDER
453            * $result[]:
454            *   ["num_fields"] number of metadata records
455            *   [0]["table"]  table name
456            *   [0]["name"]   field name
457            *   [0]["type"]   field type
458            *   [0]["len"]    field length
459            *   [0]["flags"]  field flags
460            *   ["order"][field name]  index of field named "field name"
461            *   The last one is used, if you have a field name, but no index.
462            *   Test:  if (isset($result['meta']['myfield'])) { ...
463            *
464            * - mode is DB_TABLEINFO_ORDERTABLE
465            *    the same as above. but additionally
466            *   ["ordertable"][table name][field name] index of field
467            *      named "field name"
468            *
469            *      this is, because if you have fields from different
470            *      tables with the same field name * they override each
471            *      other with DB_TABLEINFO_ORDER
472            *
473            *      you can combine DB_TABLEINFO_ORDER and
474            *      DB_TABLEINFO_ORDERTABLE with DB_TABLEINFO_ORDER |
475            *      DB_TABLEINFO_ORDERTABLE * or with DB_TABLEINFO_FULL
476            */
477   
478           // if $result is a string, then we want information about a
479           // table without a resultset
480   
481           if (is_string($result)) {
482               if (!@mssql_select_db($this->_db, $this->connection)) {
483                   return $this->mssqlRaiseError(DB_ERROR_NODBSELECTED);
484               }
485               $id = mssql_query("SELECT * FROM $result", $this->connection);
486               if (empty($id)) {
487                   return $this->mssqlRaiseError();
488               }
489           } else { // else we want information about a resultset
490               $id = $result;
491               if (empty($id)) {
492                   return $this->mssqlRaiseError();
493               }
494           }
495   
496           $count = @mssql_num_fields($id);
497   
498           // made this IF due to performance (one if is faster than $count if's)
499           if (empty($mode)) {
500   
501               for ($i=0; $i<$count; $i++) {
502                   $res[$i]['table'] = (is_string($result)) ? $result : '';
503                   $res[$i]['name']  = @mssql_field_name($id, $i);
504                   $res[$i]['type']  = @mssql_field_type($id, $i);
505                   $res[$i]['len']   = @mssql_field_length($id, $i);
506                   $res[$i]['flags'] = '';
507               }
508   
509           } else { // full
510               $res['num_fields']= $count;
511   
512               for ($i=0; $i<$count; $i++) {
513                   $res[$i]['table'] = (is_string($result)) ? $result : '';
514                   $res[$i]['name']  = @mssql_field_name($id, $i);
515                   $res[$i]['type']  = @mssql_field_type($id, $i);
516                   $res[$i]['len']   = @mssql_field_length($id, $i);
517                   $res[$i]['flags'] = '';
518                   if ($mode & DB_TABLEINFO_ORDER) {
519                       $res['order'][$res[$i]['name']] = $i;
520                   }
521                   if ($mode & DB_TABLEINFO_ORDERTABLE) {
522                       $res['ordertable'][$res[$i]['table']][$res[$i]['name']] = $i;
523                   }
524               }
525           }
526   
527           // free the result only if we were called on a table
528           if (is_string($result)) {
529               @mssql_free_result($id);
530           }
531           return $res;
532       }
