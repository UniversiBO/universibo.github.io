108   	 * Genera una password casuale
109   	 *
110   	 * @static
111   	 * @param string $password stringa della password da verificare
112   	 * @return boolean 
113   	 */
114       function generateRandomPassword()
115       {
116           $chars = array( 'a', 'A', 'b', 'B', 'c', 'C', 'd', 'D', 'e', 'E', 'f', 'F', 'g', 'G', 'h', 'H', 'i', 'I', 'j', 'J',  'k', 'K', 'l', 'L', 'm', 'M', 'n', 'N', 'o', 'O', 'p', 'P', 'q', 'Q', 'r', 'R', 's', 'S', 't', 'T',  'u', 'U', 'v', 'V', 'w', 'W', 'x', 'X', 'y', 'Y', 'z', 'Z', '1', '2', '3', '4', '5', '6', '7', '8', '9', '0');
117           $max_chars = count($chars) - 1;
118           
119           $hash = md5(microtime());
120           $loWord = substr($hash, -8);
121           $seed = hexdec($loWord);
122           $seed &= 0x7fffffff;
123           
124           mt_srand( $seed );
125       
126           $rand_str = '';
127           for($i = 0; $i < 8; $i++)
128           {
129               $rand_str = $rand_str . $chars[mt_rand(0, $max_chars)];
130           }
131   
132           return $rand_str;
133       }
