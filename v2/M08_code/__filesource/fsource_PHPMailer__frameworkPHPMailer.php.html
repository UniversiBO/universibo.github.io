339       /**
340        * Creates message and assigns Mailer. If the message is
341        * not sent successfully then it returns false.  Use the ErrorInfo
342        * variable to view description of the error.
343        * @return bool 
344        */
345       function Send() {
346           $header = "";
347           $body = "";
348   
349           if((count($this->to) + count($this->cc) + count($this->bcc)) < 1)
350           {
351               $this->SetError($this->Lang("provide_address"));
352               return false;
353           }
354   
355           // Set whether the message is multipart/alternative
356           if(!empty($this->AltBody))
357               $this->ContentType = "multipart/alternative";
358   
359           $this->SetMessageType();
360           $header .= $this->CreateHeader();
361           $body = $this->CreateBody();
362   
363           if($body == "") { return false; }
364   
365           // Choose the mailer
366           if($this->Mailer == "sendmail")
367           {
368             if(!$this->SendmailSend($header, $body))
369                 return false;
370           }
371           elseif($this->Mailer == "mail")
372           {
373             if(!$this->MailSend($header, $body))
374                 return false;
375           }
376           elseif($this->Mailer == "smtp")
377           {
378             if(!$this->SmtpSend($header, $body))
379                 return false;
380           }
381           else
382           {
383               $this->SetError($this->Mailer . $this->Lang("mailer_not_supported"));
384               return false;
385           }
386   
387           return true;
388       }

395       function SendmailSend($header, $body) {
396           if ($this->Sender != "")
397               $sendmail = sprintf("%s -oi -f %s -t", $this->Sendmail, $this->Sender);
398           else
399               $sendmail = sprintf("%s -oi -t", $this->Sendmail);
400   
401           if(!@$mail = popen($sendmail, "w"))
402           {
403               $this->SetError($this->Lang("execute") . $this->Sendmail);
404               return false;
405           }
406   
407           fputs($mail, $header);
408           fputs($mail, $body);
409           
410           $result = pclose($mail) >> 8 & 0xFF;
411           if($result != 0)
412           {
413               $this->SetError($this->Lang("execute") . $this->Sendmail);
414               return false;
415           }
416   
417           return true;

421        * Sends mail using the PHP mail() function.
422        * @access private
423        * @return bool 
424        */
425       function MailSend($header, $body) {
426           // Cannot add Bcc's to the $to
427           $to = $this->to[0][0]; // no extra comma
428           for($i = 1; $i < count($this->to); $i++)
429               $to .= sprintf(",%s", $this->to[$i][0]);
430   
431           if ($this->Sender != "" && PHP_VERSION >= "4.0")
432           {
433               $old_from = ini_get("sendmail_from");
434               ini_set("sendmail_from", $this->Sender);
435           }
436   
437           if ($this->Sender != "" && PHP_VERSION >= "4.0.5")
438           {
439               $params = sprintf("-oi -f %s", $this->Sender);
440               $rt = @mail($to, $this->EncodeHeader($this->Subject), $body, $header, $params);
441           }
442           else
443               $rt = @mail($to, $this->EncodeHeader($this->Subject), $body, $header);
444   
445           if (isset($old_from))
446               ini_set("sendmail_from", $old_from);
447   
448           if(!$rt)
449           {
450               $this->SetError($this->Lang("instantiate"));
451               return false;
452           }
453   
454           return true;
455       }

457       /**
458        * Sends mail via SMTP using PhpSMTP (Author:
459        * Chris Ryan).  Returns bool.  Returns false if there is a
460        * bad MAIL FROM, RCPT, or DATA input.
461        * @access private
462        * @return bool 
463        */
464       function SmtpSend($header, $body) {
465           // Include SMTP class code, but not twice
466           include_once($this->PluginDir . "Smtp.php");
467           $error = "";
468           $bad_rcpt = array();
469   
470           if($this->smtp == NULL)
471           {
472               if(!$this->SmtpConnect())
473                   return false;
474           }
475   
476           // Must perform HELO before authentication
477           if ($this->Helo != '')
478               $this->smtp->Hello($this->Helo);
479           else
480               $this->smtp->Hello($this->ServerHostname());
481   
482           // If user requests SMTP authentication
483           if($this->SMTPAuth)
484           {
485               if(!$this->smtp->Authenticate($this->Username, $this->Password))
486               {
487                   $this->SetError($this->Lang("authenticate"));
488                   $this->smtp->Reset();
489                   return false;
490               }
491           }
492   
493           $smtp_from = ($this->Sender == "") ? $this->From : $this->Sender;
494           if(!$this->smtp->Mail($smtp_from))
495           {
496               $error = $this->Lang("from_failed") . $smtp_from;
497               $this->SetError($error);
498               $this->smtp->Reset();
499               return false;
500           }
501   
502           // Attempt to send attach all recipients
503           for($i = 0; $i < count($this->to); $i++)
504           {
505               if(!$this->smtp->Recipient($this->to[$i][0]))
506                   $bad_rcpt[] = $this->to[$i][0];
507           }
508           for($i = 0; $i < count($this->cc); $i++)
509           {
510               if(!$this->smtp->Recipient($this->cc[$i][0]))
511                   $bad_rcpt[] = $this->cc[$i][0];
512           }
513           for($i = 0; $i < count($this->bcc); $i++)
514           {
515               if(!$this->smtp->Recipient($this->bcc[$i][0]))
516                   $bad_rcpt[] = $this->bcc[$i][0];
517           }
518   
519           // Create error message
520           if(count($bad_rcpt) > 0)
521           {
522               for($i = 0; $i < count($bad_rcpt); $i++)
523               {
524                   if($i != 0) { $error .= ", "; }
525                   $error .= $bad_rcpt[$i];
526               }
527               $error = $this->Lang("recipients_failed") . $error;
528               $this->SetError($error);
529               $this->smtp->Reset();
530               return false;
531           }
532   
533           if(!$this->smtp->Data($header . $body))
534           {
535               $this->SetError($this->Lang("data_not_accepted"));
536               $this->smtp->Reset();
537               return false;
538           }
539           if($this->SMTPKeepAlive == true)
540               $this->smtp->Reset();
541           else
542               $this->SmtpClose();
543   
544           return true;
545       }

547       /**
548        * Initiates a connection to an SMTP server.  Returns false if the
549        * operation failed.
550        * @access private
551        * @return bool 
552        */
553       function SmtpConnect() {
554           if($this->smtp == NULL) { $this->smtp = new SMTP(); }
555   
556           $this->smtp->do_debug = $this->SMTPDebug;
557           $hosts = explode(";", $this->Host);
558           $index = 0;
559           $connection = ($this->smtp->Connected()); 
560   
561           // Retry while there is no connection
562           while($index < count($hosts) && $connection == false)
563           {
564               if(strstr($hosts[$index], ":"))
565                   list($host, $port) = explode(":", $hosts[$index]);
566               else
567               {
568                   $host = $hosts[$index];
569                   $port = $this->Port;
570               }
571   
572               if($this->smtp->Connect($host, $port, $this->Timeout))
573                   $connection = true;
574               $index++;
575           }
576           if(!$connection)
577               $this->SetError($this->Lang("connect_host"));
578   
579           return $connection;
580       }

659       /**
660        * Wraps message for use with mailers that do not
661        * automatically perform wrapping and for quoted-printable.
662        * Original written by philippe.
663        * @access private
664        * @return string 
665        */
666       function WrapText($message, $length, $qp_mode = false) {
667           $soft_break = ($qp_mode) ? sprintf(" =%s", $this->LE) : $this->LE;
668   
669           $message = $this->FixEOL($message);
670           if (substr($message, -1) == $this->LE)
671               $message = substr($message, 0, -1);
672   
673           $line = explode($this->LE, $message);
674           $message = "";
675           for ($i=0 ;$i < count($line); $i++)
676           {
677             $line_part = explode(" ", $line[$i]);
678             $buf = "";
679             for ($e = 0; $e<count($line_part); $e++)
680             {
681                 $word = $line_part[$e];
682                 if ($qp_mode and (strlen($word) > $length))
683                 {
684                   $space_left = $length - strlen($buf) - 1;
685                   if ($e != 0)
686                   {
687                       if ($space_left > 20)
688                       {
689                           $len = $space_left;
690                           if (substr($word, $len - 1, 1) == "=")
691                             $len--;
692                           elseif (substr($word, $len - 2, 1) == "=")
693                             $len -= 2;
694                           $part = substr($word, 0, $len);
695                           $word = substr($word, $len);
696                           $buf .= " " . $part;
697                           $message .= $buf . sprintf("=%s", $this->LE);
698                       }
699                       else
700                       {
701                           $message .= $buf . $soft_break;
702                       }
703                       $buf = "";
704                   }
705                   while (strlen($word) > 0)
706                   {
707                       $len = $length;
708                       if (substr($word, $len - 1, 1) == "=")
709                           $len--;
710                       elseif (substr($word, $len - 2, 1) == "=")
711                           $len -= 2;
712                       $part = substr($word, 0, $len);
713                       $word = substr($word, $len);
714   
715                       if (strlen($word) > 0)
716                           $message .= $part . sprintf("=%s", $this->LE);
717                       else
718                           $buf = $part;
719                   }
720                 }
721                 else
722                 {
723                   $buf_o = $buf;
724                   $buf .= ($e == 0) ? $word : (" " . $word); 
725   
726                   if (strlen($buf) > $length and $buf_o != "")
727                   {
728                       $message .= $buf_o . $soft_break;
729                       $buf = $word;
730                   }
731                 }
732             }
733             $message .= $buf . $this->LE;
734           }
735   
736           return $message;
737       }

761       /**
762        * Assembles message header.
763        * @access private
764        * @return string 
765        */
766       function CreateHeader() {
767           $result = "";
768           
769           // Set the boundaries
770           $uniq_id = md5(uniqid(time()));
771           $this->boundary[1] = "b1_" . $uniq_id;
772           $this->boundary[2] = "b2_" . $uniq_id;
773   
774           $result .= $this->Received();
775           $result .= $this->HeaderLine("Date", $this->RFCDate());
776           if($this->Sender == "")
777               $result .= $this->HeaderLine("Return-Path", trim($this->From));
778           else
779               $result .= $this->HeaderLine("Return-Path", trim($this->Sender));
780           
781           // To be created automatically by mail()
782           if($this->Mailer != "mail")
783           {
784               if(count($this->to) > 0)
785                   $result .= $this->AddrAppend("To", $this->to);
786               else if (count($this->cc) == 0)
787                   $result .= $this->HeaderLine("To", "undisclosed-recipients:;");
788           }
789   
790           $from = array();
791           $from[0][0] = trim($this->From);
792           $from[0][1] = $this->FromName;
793           $result .= $this->AddrAppend("From", $from); 
794   
795           if(count($this->cc) > 0)
796               $result .= $this->AddrAppend("Cc", $this->cc);
797   
798           // sendmail and mail() extract Bcc from the header before sending
799           if((($this->Mailer == "sendmail") || ($this->Mailer == "mail")) && (count($this->bcc) > 0))
800               $result .= $this->AddrAppend("Bcc", $this->bcc);
801   
802           if(count($this->ReplyTo) > 0)
803               $result .= $this->AddrAppend("Reply-to", $this->ReplyTo);
804   
805           // mail() sets the subject itself
806           if($this->Mailer != "mail")
807               $result .= $this->HeaderLine("Subject", $this->EncodeHeader(trim($this->Subject)));
808   
809           $result .= sprintf("Message-ID: <%s@%s>%s", $uniq_id, $this->ServerHostname(), $this->LE);
810           $result .= $this->HeaderLine("X-Priority", $this->Priority);
811           $result .= $this->HeaderLine("X-Mailer", "PHPMailer [version " . $this->Version . "]");
812           
813           if($this->ConfirmReadingTo != "")
814           {
815               $result .= $this->HeaderLine("Disposition-Notification-To", 
816                          "<" . trim($this->ConfirmReadingTo) . ">");
817           }
818   
819           // Add custom headers
820           for($index = 0; $index < count($this->CustomHeader); $index++)
821           {
822               $result .= $this->HeaderLine(trim($this->CustomHeader[$index][0]), 
823                          $this->EncodeHeader(trim($this->CustomHeader[$index][1])));
824           }
825           $result .= $this->HeaderLine("MIME-Version", "1.0");
826   
827           switch($this->message_type)
828           {
829               case "plain":
830                   $result .= $this->HeaderLine("Content-Transfer-Encoding", $this->Encoding);
831                   $result .= sprintf("Content-Type: %s; charset=\"%s\"",
832                                       $this->ContentType, $this->CharSet);
833                   break;
834               case "attachments":
835                   // fall through
836               case "alt_attachments":
837                   if($this->EmbeddedImageCount() > 0)
838                   {
839                       $result .= sprintf("Content-Type: %s;%s\ttype=\"text/html\";%s\tboundary=\"%s\"%s", 
840                                       "multipart/related", $this->LE, $this->LE, 
841                                       $this->boundary[1], $this->LE);
842                   }
843                   else
844                   {
845                       $result .= $this->HeaderLine("Content-Type", "multipart/mixed;");
846                       $result .= $this->TextLine("\tboundary=\"" . $this->boundary[1] . '"');
847                   }
848                   break;
849               case "alt":
850                   $result .= $this->HeaderLine("Content-Type", "multipart/alternative;");
851                   $result .= $this->TextLine("\tboundary=\"" . $this->boundary[1] . '"');
852                   break;
853           }
854   
855           if($this->Mailer != "mail")
856               $result .= $this->LE.$this->LE;
857   
858           return $result;
859       }

861       /**
862        * Assembles the message body.  Returns an empty string on failure.
863        * @access private
864        * @return string 
865        */
866       function CreateBody() {
867           $result = "";
868   
869           $this->SetWordWrap();
870   
871           switch($this->message_type)
872           {
873               case "alt":
874                   $result .= $this->GetBoundary($this->boundary[1], "", 
875                                                 "text/plain", "");
876                   $result .= $this->EncodeString($this->AltBody, $this->Encoding);
877                   $result .= $this->LE.$this->LE;
878                   $result .= $this->GetBoundary($this->boundary[1], "", 
879                                                 "text/html", "");
880                   
881                   $result .= $this->EncodeString($this->Body, $this->Encoding);
882                   $result .= $this->LE.$this->LE;
883       
884                   $result .= $this->EndBoundary($this->boundary[1]);
885                   break;
886               case "plain":
887                   $result .= $this->EncodeString($this->Body, $this->Encoding);
888                   break;
889               case "attachments":
890                   $result .= $this->GetBoundary($this->boundary[1], "", "", "");
891                   $result .= $this->EncodeString($this->Body, $this->Encoding);
892                   $result .= $this->LE;
893        
894                   $result .= $this->AttachAll();
895                   break;
896               case "alt_attachments":
897                   $result .= sprintf("--%s%s", $this->boundary[1], $this->LE);
898                   $result .= sprintf("Content-Type: %s;%s" .
899                                      "\tboundary=\"%s\"%s",
900                                      "multipart/alternative", $this->LE, 
901                                      $this->boundary[2], $this->LE.$this->LE);
902       
903                   // Create text body
904                   $result .= $this->GetBoundary($this->boundary[2], "", 
905                                                 "text/plain", "") . $this->LE;
906   
907                   $result .= $this->EncodeString($this->AltBody, $this->Encoding);
908                   $result .= $this->LE.$this->LE;
909       
910                   // Create the HTML body
911                   $result .= $this->GetBoundary($this->boundary[2], "", 
912                                                 "text/html", "") . $this->LE;
913       
914                   $result .= $this->EncodeString($this->Body, $this->Encoding);
915                   $result .= $this->LE.$this->LE;
916   
917                   $result .= $this->EndBoundary($this->boundary[2]);
918                   
919                   $result .= $this->AttachAll();
920                   break;
921           }
922           if($this->IsError())
923               $result = "";
924   
925           return $result;
926       }

1032      /**
1033       * Attaches all fs, string, and binary attachments to the message.
1034       * Returns an empty string on failure.
1035       * @access private
1036       * @return string 
1037       */
1038      function AttachAll() {
1039          // Return text of body
1040          $mime = array();
1041  
1042          // Add all attachments
1043          for($i = 0; $i < count($this->attachment); $i++)
1044          {
1045              // Check for string attachment
1046              $bString = $this->attachment[$i][5];
1047              if ($bString)
1048                  $string = $this->attachment[$i][0];
1049              else
1050                  $path = $this->attachment[$i][0];
1051  
1052              $filename    = $this->attachment[$i][1];
1053              $name        = $this->attachment[$i][2];
1054              $encoding    = $this->attachment[$i][3];
1055              $type        = $this->attachment[$i][4];
1056              $disposition = $this->attachment[$i][6];
1057              $cid         = $this->attachment[$i][7];
1058              
1059              $mime[] = sprintf("--%s%s", $this->boundary[1], $this->LE);
1060              $mime[] = sprintf("Content-Type: %s; name=\"%s\"%s", $type, $name, $this->LE);
1061              $mime[] = sprintf("Content-Transfer-Encoding: %s%s", $encoding, $this->LE);
1062  
1063              if($disposition == "inline")
1064                  $mime[] = sprintf("Content-ID: <%s>%s", $cid, $this->LE);
1065  
1066              $mime[] = sprintf("Content-Disposition: %s; filename=\"%s\"%s", 
1067                                $disposition, $name, $this->LE.$this->LE);
1068  
1069              // Encode as string attachment
1070              if($bString)
1071              {
1072                  $mime[] = $this->EncodeString($string, $encoding);
1073                  if($this->IsError()) { return ""; }
1074                  $mime[] = $this->LE.$this->LE;
1075              }
1076              else
1077              {
1078                  $mime[] = $this->EncodeFile($path, $encoding);                
1079                  if($this->IsError()) { return ""; }
1080                  $mime[] = $this->LE.$this->LE;
1081              }
1082          }
1083  
1084          $mime[] = sprintf("--%s--%s", $this->boundary[1], $this->LE);
1085  
1086          return join("", $mime);
1087      }

1108      /**
1109       * Encodes string to requested format. Returns an
1110       * empty string on failure.
1111       * @access private
1112       * @return string 
1113       */
1114      function EncodeString ($str, $encoding = "base64") {
1115          $encoded = "";
1116          switch(strtolower($encoding)) {
1117            case "base64":
1118                // chunk_split is found in PHP >= 3.0.6
1119                $encoded = chunk_split(base64_encode($str), 76, $this->LE);
1120                break;
1121  
1122            case "7bit":
1123            case "8bit":
1124                $encoded = $this->FixEOL($str);
1125                if (substr($encoded, -(strlen($this->LE))) != $this->LE)
1126                  $encoded .= $this->LE;
1127                break;
1128  
1129            case "binary":
1130                $encoded = $str;
1131                break;
1132  
1133            case "quoted-printable":
1134                $encoded = $this->EncodeQP($str);
1135                break;
1136  
1137            default:
1138                $this->SetError($this->Lang("encoding") . $encoding);
1139                break;
1140          }
1141          return $encoded;
1142      }

1144      /**
1145       * Encode a header string to best of Q, B, quoted or none.
1146       * @access private
1147       * @return string 
1148       */
1149      function EncodeHeader ($str, $position = 'text') {
1150        $x = 0;
1151        
1152        switch (strtolower($position)) {
1153          case 'phrase':
1154            if (!preg_match('/[\200-\377]/', $str)) {
1155              // Can't use addslashes as we don't know what value has magic_quotes_sybase.
1156              $encoded = addcslashes($str, "\0..\37\177\\\"");
1157  
1158              if (($str == $encoded) && !preg_match('/[^A-Za-z0-9!#$%&\'*+\/=?^_`{|}~ -]/', $str))
1159                return ($encoded);
1160              else
1161                return ("\"$encoded\"");
1162            }
1163            $x = preg_match_all('/[^\040\041\043-\133\135-\176]/', $str, $matches);
1164            break;
1165          case 'comment':
1166            $x = preg_match_all('/[()"]/', $str, $matches);
1167            // Fall-through
1168          case 'text':
1169          default:
1170            $x += preg_match_all('/[\000-\010\013\014\016-\037\177-\377]/', $str, $matches);
1171            break;
1172        }
1173  
1174        if ($x == 0)
1175          return ($str);
1176  
1177        $maxlen = 75 - 7 - strlen($this->CharSet);
1178        // Try to select the encoding which should produce the shortest output
1179        if (strlen($str)/3 < $x) {
1180          $encoding = 'B';
1181          $encoded = base64_encode($str);
1182          $maxlen -= $maxlen % 4;
1183          $encoded = trim(chunk_split($encoded, $maxlen, "\n"));
1184        } else {
1185          $encoding = 'Q';
1186          $encoded = $this->EncodeQ($str, $position);
1187          $encoded = $this->WrapText($encoded, $maxlen, true);
1188          $encoded = str_replace("=".$this->LE, "\n", trim($encoded));
1189        }
1190  
1191        $encoded = preg_replace('/^(.*)$/m', " =?".$this->CharSet."?$encoding?\\1?=", $encoded);
1192        $encoded = trim(str_replace("\n", $this->LE, $encoded));
1193        
1194        return $encoded;
1195      }

1197      /**
1198       * Encode string to quoted-printable.
1199       * @access private
1200       * @return string 
1201       */
1202      function EncodeQP ($str) {
1203          $encoded = $this->FixEOL($str);
1204          if (substr($encoded, -(strlen($this->LE))) != $this->LE)
1205              $encoded .= $this->LE;
1206  
1207          // Replace every high ascii, control and = characters
1208          $encoded = preg_replace('/([\000-\010\013\014\016-\037\075\177-\377])/e',
1209                    "'='.sprintf('%02X', ord('\\1'))", $encoded);
1210          // Replace every spaces and tabs when it's the last character on a line
1211          $encoded = preg_replace("/([\011\040])".$this->LE."/e",
1212                    "'='.sprintf('%02X', ord('\\1')).'".$this->LE."'", $encoded);
1213  
1214          // Maximum line length of 76 characters before CRLF (74 + space + '=')
1215          $encoded = $this->WrapText($encoded, 74, true);
1216  
1217          return $encoded;
1218      }

1220      /**
1221       * Encode string to q encoding.
1222       * @access private
1223       * @return string 
1224       */
1225      function EncodeQ ($str, $position = "text") {
1226          // There should not be any EOL in the string
1227          $encoded = preg_replace("[\r\n]", "", $str);
1228  
1229          switch (strtolower($position)) {
1230            case "phrase":
1231              $encoded = preg_replace("/([^A-Za-z0-9!*+\/ -])/e", "'='.sprintf('%02X', ord('\\1'))", $encoded);
1232              break;
1233            case "comment":
1234              $encoded = preg_replace("/([\(\)\"])/e", "'='.sprintf('%02X', ord('\\1'))", $encoded);
1235            case "text":
1236            default:
1237              // Replace every high ascii, control =, ? and _ characters
1238              $encoded = preg_replace('/([\000-\011\013\014\016-\037\075\077\137\177-\377])/e',
1239                    "'='.sprintf('%02X', ord('\\1'))", $encoded);
1240              break;
1241          }
1242          
1243          // Replace every spaces to _ (more readable than =20)
1244          $encoded = str_replace(" ", "_", $encoded);
1245  
1246          return $encoded;
1247      }

1422      /**
1423       * Returns Received header for message tracing. Returns string.
1424       * @access private
1425       * @return string 
1426       */
1427      function Received() {
1428          if ($this->ServerVar('SERVER_NAME') != '')
1429          {
1430              $protocol = ($this->ServerVar('HTTPS') == 'on') ? 'HTTPS' : 'HTTP';
1431              $remote = $this->ServerVar('REMOTE_HOST');
1432              if($remote == "")
1433                  $remote = 'phpmailer';
