PHPUnit_TestListener
A Listener for test progress.


Page Details:

A Listener for test progress.

Here is an example:

<?php require_once 'PHPUnit.php'; require_once 'PHPUnit/TestListener.php';

class MathTest extends PHPUnit_TestCase { var $fValue1; var $fValue2;

function MathTest($name) { $this->PHPUnit_TestCase($name); }

function setUp() { $this->fValue1 = 2; $this->fValue2 = 3; }

function testAdd() { $this->assertTrue($this->fValue1 + $this->fValue2 == 4); } }

class MyListener extends PHPUnit_TestListener { function addError(&$test, &$t) { echo "MyListener::addError() called.\n"; }

function addFailure(&$test, &$t) { echo "MyListener::addFailure() called.\n"; }

function endTest(&$test) { echo "MyListener::endTest() called.\n"; }

function startTest(&$test) { echo "MyListener::startTest() called.\n"; } }

$suite = new PHPUnit_TestSuite; $suite->addTest(new MathTest('testAdd'));

$result = new PHPUnit_TestResult; $result->addListener(new MyListener);

$suite->run($result); echo $result->toString(); ?>
